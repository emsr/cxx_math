52  /**
53  * Return the reperiodized sine of argument x:
54  * @f[
55  * \mathrm{sin_\pi}(x) = \sin(\pi x)
56  * @f]
57  */
58  template<typename Tp>
59  Tp
60  sin_pi(Tp x)
61  {
62  using _Val = emsr::num_traits_t<Tp>;
63  const auto s_pi = emsr::pi_v<_Val>;
64  if (std::isnan(x))
65  return emsr::quiet_NaN(x);
66  else if (x < Tp{0})
67  return -sin_pi(-x);
68  else if (x < Tp{0.5L})
69  return std::sin(x * s_pi);
70  else if (x < Tp{1})
71  return std::sin((Tp{1} - x) * s_pi);
72  else
73  {
74  auto nu = std::floor(x);
75  auto arg = x - nu;
76  auto sign = (int(nu) & 1) == 1 ? -1 : +1;
77  auto sinval = (arg < Tp{0.5L})
78  ? sin_pi(arg)
79  : sin_pi(Tp{1} - arg);
80  return sign * sinval;
81  }
82  }

185  /**
186  * Return the reperiodized sine of complex argument z:
187  * @f[
188  * \mathrm{sin_\pi}(z) = \sin(\pi z)
189  * = \mathrm{sin_\pi}(x) \mathrm{cosh_\pi}(y)
190  * + i \mathrm{cos_\pi}(x) \mathrm{sinh_\pi}(y)
191  * @f]
192  */
193  template<typename Tp>
194  std::complex<Tp>
195  sin_pi(std::complex<Tp> z)
196  {
197  using _Val = Tp;
198  using _Real = emsr::num_traits_t<_Val>;
199  const auto s_pi = emsr::pi_v<_Real>;
200  const auto s_i = std::complex<Tp>{0, 1};
201  auto x = std::real(z);
202  auto y = std::imag(z);
203  return sin_pi(x) * std::cosh(s_pi * y)
204  + s_i * cos_pi(x) * std::sinh(s_pi * y);
205  }

273  /**
274  * Return the reperiodized tangent of complex argument z:
275  * @f[
276  * \mathrm{tan_\pi}(z) = \tan(\pi z)
277  * = \frac{\mathrm{tan_\pi}(x) + i \mathrm{tanh_\pi}(y)}
278  * {1 - i \mathrm{tan_\pi}(x) \mathrm{tanh_\pi}(y)}
279  * @f]
280  */
281  template<typename Tp>
282  std::complex<Tp>
283  tan_pi(std::complex<Tp> z)
284  {
285  using _Val = Tp;
286  using _Real = emsr::num_traits_t<_Val>;
287  const auto s_pi = emsr::pi_v<_Real>;
288  const auto s_i = std::complex<Tp>{0, 1};
289  auto x = std::real(z);
290  auto y = std::imag(z);
291  auto tan = tan_pi(x);
292  auto tanh = std::tanh(s_pi * y);
293  return (tan + s_i * tanh) / (_Val{1} - s_i * tan * tanh);
294  }

296  /**
297  * Return the reperiodized hyperbolic tangent of complex argument z:
298  * @f[
299  * \mathrm{tanh_\pi}(z) = \tanh(\pi z)
300  * = \frac{\mathrm{tanh_\pi}(x) + i \mathrm{tan_\pi}(y)}
301  * {1 + i \mathrm{tanh_\pi}(x) \mathrm{tan_\pi}(y)}
302  * @f]
303  */
304  template<typename Tp>
305  std::complex<Tp>
306  tanh_pi(std::complex<Tp> z)
307  {
308  using _Val = Tp;
309  using _Real = emsr::num_traits_t<_Val>;
310  const auto s_pi = emsr::pi_v<_Real>;
311  const auto s_i = std::complex<Tp>{0, 1};
312  auto x = std::real(z);
313  auto y = std::imag(z);
314  auto tanh = std::tanh(s_pi * x);
315  auto tan = tan_pi(y);
316  return (tanh + s_i * tan) / (_Val{1} + s_i * tanh * tan);
317  }

363  /**
364  * Reperiodized sincos.
365  */
366  template<typename Tp>
367  emsr::sincos_t<Tp>
368  sincos_pi(Tp x)
369  {
370  const auto s_pi = emsr::pi_v<Tp>;
371  const auto s_NaN = emsr::quiet_NaN(x);
372  if (std::isnan(x))
373  return emsr::sincos_t<Tp>{s_NaN, s_NaN};
374  else if (x < Tp{0})
375  {
376  emsr::sincos_t<Tp> tempsc = sincos_pi(-x);
377  return emsr::sincos_t<Tp>{-tempsc.sin_v, tempsc.cos_v};
378  }
379  else if (x < Tp{0.5L})
380  return sincos(s_pi * x);
381  else if (x < Tp{1})
382  {
383  emsr::sincos_t<Tp> tempsc = sincos(s_pi * (Tp{1} - x));
384  return emsr::sincos_t<Tp>{tempsc.sin_v, -tempsc.cos_v};
385  }
386  else
387  {
388  auto nu = std::floor(x);
389  auto arg = x - nu;
390  auto sign = (int(nu) & 1) == 1 ? Tp{-1} : Tp{+1};
391 
392  auto sinval = (arg < Tp{0.5L})
393  ? std::sin(s_pi * arg)
394  : std::sin(s_pi * (Tp{1} - arg));
395  auto cosval = std::cos(s_pi * arg);
396  return emsr::sincos_t<Tp>{sign * sinval,
397  sign * cosval};
398  }
