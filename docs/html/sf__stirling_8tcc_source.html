48  /**
49  * Return the Stirling number of the second kind from lookup
50  * or by series expansion.
51  *
52  * The series is:
53  * @f[
54  * \newcommand{\stirling}[2]{\genfrac{\{}{\}}{0pt}{0}{#1}{#2}}
55  *
56  * \sigma_n^{(m)} = \stirling{n}{m}
57  * = \sum_{k=0}^{m}\frac{(-1)^{m-k}k^n}{(m-k)!k!}
58  * @f]
59  * The Stirling number of the second kind is denoted by other symbols
60  * in the literature:
61  * @f$ \sigma_n^{(m)} @f$, @f$ \textit{S}_n^{(m)} @f$ and others.
62  *
63  * @todo Find a way to predict the maximum Stirling number for a type.
64  */
65  template<typename _Tp>
66  _Tp
67  __stirling_2_series(unsigned int __n, unsigned int __m)
68  {
69  if (__m > _S_num_factorials<_Tp>)
70  {
71  auto _S2 = _Tp{0};
72  for (auto __k = 0u; __k <= __m; ++__k)
73  {
74  auto __lf1 = __log_factorial<_Tp>(__k);
75  auto __lf2 = __log_factorial<_Tp>(__m - __k);
76  _S2 += ((__m - __k) & 1 ? _Tp{-1} : _Tp{1})
77  * std::exp(__n * std::log(__k) - __lf1 - __lf2);
78  }
79  return _S2;
80  }
81  else
82  {
83  auto _S2 = _Tp{0};
84  for (auto __k = 0u; __k <= __m; ++__k)
85  {
86  _S2 += ((__m - __k) & 1 ? _Tp{-1} : _Tp{1})
87  * std::pow(__k, __n)
88  / __factorial<_Tp>(__k)
89  / __factorial<_Tp>(__m - __k);
90  }
91  // @todo Only round if the sum is less than
92  // the maximum representable integer.
93  // Find or make a tool for this.
94  return std::nearbyint(_S2);
95  }
96  }

98  /**
99  * Return the Stirling number of the second kind by recursion.
100  * The recursion is
101  * @f[
102  * \newcommand{\stirling}[2]{\genfrac{\{}{\}}{0pt}{0}{#1}{#2}}
103  * \stirling{n}{m} = m \stirling{n-1}{m} + \stirling{n-1}{m-1}
104  * @f]
105  * with starting values
106  * @f[
107  * \newcommand{\stirling}[2]{\genfrac{\{}{\}}{0pt}{0}{#1}{#2}}
108  * \stirling{0}{0\rightarrow m} = {1, 0, 0, ..., 0}
109  * @f]
110  * and
111  * @f[
112  * \newcommand{\stirling}[2]{\genfrac{\{}{\}}{0pt}{0}{#1}{#2}}
113  * \stirling{0\rightarrow n}{0} = {1, 0, 0, ..., 0}
114  * @f]
115  *
116  * The Stirling number of the second kind is denoted by other symbols
117  * in the literature:
118  * @f$ \sigma_n^{(m)} @f$, @f$ \textit{S}_n^{(m)} @f$ and others.
119  */
120  template<typename _Tp>
121  _Tp
122  __stirling_2_recur(unsigned int __n, unsigned int __m)
123  {
124  if (__n == 0)
125  return _Tp(__m == 0);
126  else if (__m == 0)
127  return _Tp(__n == 0);
128  else
129  {
130  std::vector<_Tp> __sigold(__m + 1), __signew(__m + 1);
131  __sigold[1] = _Tp{1};
132  if (__n == 1)
133  return __sigold[__m];
134  for (auto __in = 1u; __in <= __n; ++__in)
135  {
136  __signew[1] = __sigold[1];
137  for (auto __im = 2u; __im <= __m; ++__im)
138  __signew[__im] = __im * __sigold[__im] + __sigold[__im - 1];
139  std::swap(__sigold, __signew);
140  }
141  return __signew[__m];
142  }
143  }

247  * @f]

250  _Tp
251  __stirling_1_recur(unsigned int __n, unsigned int __m)
252  {
253  if (__n == 0)
254  return _Tp(__m == 0);
255  else if (__m == 0)
256  return _Tp(__n == 0);
257  else
258  {
259  std::vector<_Tp> _Sold(__m + 1), _Snew(__m + 1);
260  _Sold[1] = _Tp{1};
261  if (__n == 1)
262  return _Sold[__m];
263  for (auto __in = 1u; __in <= __n; ++__in)
264  {
265  for (auto __im = 1u; __im <= __m; ++__im)
266  _Snew[__im] = _Sold[__im - 1] - __in * _Sold[__im];
267  std::swap(_Sold, _Snew);
268  }
269  return _Snew[__m];
270  }
