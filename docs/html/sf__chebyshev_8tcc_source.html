45  /**
46  * Return a Chebyshev polynomial of non-negative order @f$ n @f$
47  * and real argument @f$ x @f$ by the recursion
48  * @f[
49  * C_n(x) = 2xC_{n-1} - C_{n-2}
50  * @f]
51  *
52  * @tparam _Tp The real type of the argument
53  * @param __n The non-negative integral order
54  * @param __x The real argument @f$ -1 <= x <= +1 @f$
55  * @param _C0 The value of the zeroth-order Chebyshev polynomial at @f$ x @f$
56  * @param _C1 The value of the first-order Chebyshev polynomial at @f$ x @f$
57  */
58  template<typename _Tp>
59  std::tuple<_Tp, _Tp, _Tp>
60  __chebyshev_recur(unsigned int __n, _Tp __x, _Tp _C0, _Tp _C1)
61  {
62  auto _Ck = _Tp{2} * __x * _C1 - _C0;
63  for (unsigned int __j = 2; __j < __n; ++__j)
64  {
65  _C0 = _C1;
66  _C1 = _Ck;
67  _Ck = _Tp{2} * __x * _C1 - _C0;
68  }
69  return std::make_tuple(_Ck, _C1, _C0);
70  }

72  /**
73  * Return the Chebyshev polynomial of the first kind @f$ T_n(x) @f$
74  * of non-negative order @f$ n @f$ and real argument @f$ x @f$.
75  *
76  * The Chebyshev polynomial of the first kind is defined by:
77  * @f[
78  * T_n(x) = \cos(n \theta)
79  * @f]
80  * where @f$ \theta = \arccos(x) @f$, @f$ -1 <= x <= +1 @f$.
81  *
82  * @tparam _Tp The real type of the argument
83  * @param __n The non-negative integral order
84  * @param __x The real argument @f$ -1 <= x <= +1 @f$
85  */
86  template<typename _Tp>
87  __gnu_cxx::__chebyshev_t_t<_Tp>
88  __chebyshev_t(unsigned int __n, _Tp __x)
89  {
90  auto _T0 = _Tp{1};
91  if (__n == 0)
92  return {__n, __x, _T0, _Tp{0}, _Tp{0}};
93 
94  auto _T1 = __x;
95  if (__n == 1)
96  return {__n, __x, _T1, _T0, _Tp{0}};
97 
98  auto _Ts = __chebyshev_recur(__n, __x, _T0, _T1);
99  return {__n, __x, std::get<0>(_Ts), std::get<1>(_Ts), std::get<2>(_Ts)};
100  }
