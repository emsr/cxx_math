41  /**
42  * Return a Chebyshev polynomial of non-negative order @f$ n @f$
43  * and real argument @f$ x @f$ by the recursion
44  * @f[
45  * C_n(x) = 2xC_{n-1} - C_{n-2}
46  * @f]
47  *
48  * @tparam Tp The real type of the argument
49  * @param n The non-negative integral order
50  * @param x The real argument @f$ -1 <= x <= +1 @f$
51  * @param C0 The value of the zeroth-order Chebyshev polynomial at @f$ x @f$
52  * @param C1 The value of the first-order Chebyshev polynomial at @f$ x @f$
53  */
54  template<typename Tp>
55  std::tuple<Tp, Tp, Tp>
56  chebyshev_recur(unsigned int n, Tp x, Tp C0, Tp C1)
57  {
58  auto Ck = Tp{2} * x * C1 - C0;
59  for (unsigned int j = 2; j < n; ++j)
60  {
61  C0 = C1;
62  C1 = Ck;
63  Ck = Tp{2} * x * C1 - C0;
64  }
65  return std::make_tuple(Ck, C1, C0);
66  }

68  /**
69  * Return the Chebyshev polynomial of the first kind @f$ T_n(x) @f$
70  * of non-negative order @f$ n @f$ and real argument @f$ x @f$.
71  *
72  * The Chebyshev polynomial of the first kind is defined by:
73  * @f[
74  * T_n(x) = \cos(n \theta)
75  * @f]
76  * where @f$ \theta = \arccos(x) @f$, @f$ -1 <= x <= +1 @f$.
77  *
78  * @tparam Tp The real type of the argument
79  * @param n The non-negative integral order
80  * @param x The real argument @f$ -1 <= x <= +1 @f$
81  */
82  template<typename Tp>
83  emsr::chebyshev_t_t<Tp>
84  chebyshev_t(unsigned int n, Tp x)
85  {
86  auto T0 = Tp{1};
87  if (n == 0)
88  return {n, x, T0, Tp{0}, Tp{0}};
89 
90  auto T1 = x;
91  if (n == 1)
92  return {n, x, T1, T0, Tp{0}};
93 
94  auto Ts = chebyshev_recur(n, x, T0, T1);
95  return {n, x, std::get<0>(Ts), std::get<1>(Ts), std::get<2>(Ts)};
96  }

98  /**
99  * Return the Chebyshev polynomial of the second kind @f$ U_n(x) @f$
100  * of non-negative order @f$ n @f$ and real argument @f$ x @f$.
101  *
102  * The Chebyshev polynomial of the second kind is defined by:
103  * @f[
104  * U_n(x) = \frac{\sin \left[(n+1)\theta \right]}{\sin(\theta)}
105  * @f]
106  * where @f$ \theta = \arccos(x) @f$, @f$ -1 <= x <= +1 @f$.
107  *
108  * @tparam Tp The real type of the argument
109  * @param n The non-negative integral order
110  * @param x The real argument @f$ -1 <= x <= +1 @f$
111  */
112  template<typename Tp>
113  emsr::chebyshev_u_t<Tp>
114  chebyshev_u(unsigned int n, Tp x)
115  {
116  auto U0 = Tp{1};
117  if (n == 0)
118  return {n, x, U0, Tp{0}, Tp{0}};
119 
120  auto U1 = Tp{2} * x;
121  if (n == 1)
122  return {n, x, U1, U0, Tp{0}};
123 
124  auto Us = chebyshev_recur(n, x, U0, U1);
125  return {n, x, std::get<0>(Us), std::get<1>(Us), std::get<2>(Us)};
126  }

128  /**
129  * Return the Chebyshev polynomial of the third kind @f$ V_n(x) @f$
130  * of non-negative order @f$ n @f$ and real argument @f$ x @f$.
131  *
132  * The Chebyshev polynomial of the third kind is defined by:
133  * @f[
134  * V_n(x) = \frac{\cos \left[ \left(n+\frac{1}{2}\right)\theta \right]}
135  * {\cos \left(\frac{\theta}{2}\right)}
136  * @f]
137  * where @f$ \theta = \arccos(x) @f$, @f$ -1 <= x <= +1 @f$.
138  *
139  * @tparam Tp The real type of the argument
140  * @param n The non-negative integral order
141  * @param x The real argument @f$ -1 <= x <= +1 @f$
142  */
143  template<typename Tp>
144  emsr::chebyshev_v_t<Tp>
145  chebyshev_v(unsigned int n, Tp x)
146  {
147  auto V0 = Tp{1};
148  if (n == 0)
149  return {n, x, V0, Tp{0}, Tp{0}};
150 
151  auto V1 = Tp{2} * x - Tp{1};
152  if (n == 1)
153  return {n, x, V1, V0, Tp{0}};
154 
155  auto Vs = chebyshev_recur(n, x, V0, V1);
156  return {n, x, std::get<0>(Vs), std::get<1>(Vs), std::get<2>(Vs)};
157  }

161  * of non-negative order @f$ n @f$ and real argument @f$ x @f$.
162  *
163  * The Chebyshev polynomial of the fourth kind is defined by:
164  * @f[
165  * W_n(x) = \frac{\sin \left[ \left(n+\frac{1}{2}\right)\theta \right]}

167  * @f]
168  * where @f$ \theta = \arccos(x) @f$, @f$ -1 <= x <= +1 @f$.
169  *
170  * @tparam Tp The real type of the argument
171  * @param n The non-negative integral order
172  * @param x The real argument @f$ -1 <= x <= +1 @f$
173  */

175  emsr::chebyshev_w_t<Tp>
