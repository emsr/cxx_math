63  /**
64  * @brief Return the Legendre polynomial by upward recursion
65  * on degree @f$ l @f$.
66  *
67  * The Legendre function of degree @f$ l @f$ and argument @f$ x @f$,
68  * @f$ P_l(x) @f$, is defined by:
69  * @f[
70  * P_l(x) = \frac{1}{2^l l!}\frac{d^l}{dx^l}(x^2 - 1)^{l}
71  * @f]
72  * This can be expressed as a series:
73  * @f[
74  * P_l(x) = \frac{1}{2^l l!}\sum_{k=0}^{\lfloor l/2 \rfloor}
75  * \frac{(-1)^k(2l-2k)!}{k!(l-k)!(l-2k)!}x^{l-2k}
76  * @f]
77  *
78  * @param l The degree of the Legendre polynomial. @f$ l >= 0 @f$.
79  * @param x The argument of the Legendre polynomial.
80  */
81  template<typename Tp>
82  emsr::legendre_p_t<Tp>
83  legendre_p(unsigned int l, Tp x)
84  {
85  using _Real = emsr::num_traits_t<Tp>;
86  using ret_t = emsr::legendre_p_t<Tp>;
87 
88  const auto lge1 = l >= 1 ? Tp{+1} : Tp{0};
89  const auto lge2 = l >= 2 ? Tp{+1} : Tp{0};
90  const auto s_NaN = emsr::quiet_NaN(_Real{});
91 
92  if (std::isnan(x))
93  return {l, s_NaN, s_NaN, s_NaN, s_NaN};
94  else if (x == _Real{+1})
95  return {l, x, Tp{+1}, lge1, lge2};
96  else if (x == _Real{-1})
97  return l % 2 == 1
98  ? ret_t{l, x, Tp{-1}, +lge1, -lge2}
99  : ret_t{l, x, Tp{+1}, -lge1, +lge2};
100  else
101  {
102  auto _P_lm2 = Tp{1};
103  if (l == 0)
104  return {l, x, _P_lm2, Tp{0}, Tp{0}};
105 
106  auto _P_lm1 = x;
107  if (l == 1)
108  return {l, x, _P_lm1, _P_lm2, Tp{0}};
109 
110  auto _P_l = Tp{2} * x * _P_lm1 - _P_lm2
111  - (x * _P_lm1 - _P_lm2) / Tp{2};
112  for (unsigned int ll = 3; ll <= l; ++ll)
113  {
114  _P_lm2 = _P_lm1;
115  _P_lm1 = _P_l;
116  // This arrangement is supposed to be better for roundoff
117  // protection, Arfken, 2nd Ed, Eq 12.17a.
118  _P_l = Tp{2} * x * _P_lm1 - _P_lm2
119  - (x * _P_lm1 - _P_lm2) / Tp(ll);
120  }
121 
122  return {l, x, _P_l, _P_lm1, _P_lm2};
123  }
124  }

126  /**
127  * Legendre q series.
128  */
129  template<typename Tp>
130  Tp
131  legendre_q_series(unsigned int l, Tp x)
132  {
133  const auto num1 = Tp(l + 1) / Tp{2};
134  const auto num2 = Tp(l + 2) / Tp{2};
135  const auto den = Tp(2 * l + 3) / Tp{2};
136  const auto rx = Tp{1} / x;
137  const auto rx2 = rx * rx;
138 
139  const auto sum = emsr::hyperg(num1, num2, den, rx2);
140 
141  auto fact = rx;
142  for (unsigned int k = 1; k <= l; ++k)
143  fact *= Tp(k) * rx / Tp(2 * k - 1);
144 
145  return fact * sum;
146  }

148  /**
149  * @brief Return the Legendre function of the second kind
150  * by upward recursion on degree @f$ l @f$.
151  *
152  * The Legendre function of the second kind of degree @f$ l @f$
153  * and argument @f$ x @f$, @f$ Q_l(x) @f$, is defined by:
154  * @f[
155  * Q_l(x) = \frac{1}{2^l l!}\frac{d^l}{dx^l}(x^2 - 1)^{l}
156  * @f]
157  *
158  * @param l The degree of the Legendre function. @f$ l >= 0 @f$.
159  * @param x The argument of the Legendre function. @f$ |x| <= 1 @f$.
160  */
161  template<typename Tp>
162  emsr::legendre_q_t<Tp>
163  legendre_q(unsigned int l, Tp x)
164  {
165  using _Real = emsr::num_traits_t<Tp>;
166  const auto s_eps = emsr::epsilon(_Real{});
167  const auto s_inf = emsr::infinity(_Real{});
168  if (std::isnan(x))
169  {
170  const auto s_NaN = emsr::quiet_NaN(_Real{});
171  return {l, x, s_NaN, s_NaN, s_NaN};
172  }
173  else if (std::abs(x - _Real{1}) < s_eps)
174  return {l, x, s_inf, s_inf, s_inf};
175  else if (std::abs(x + _Real{1}) < s_eps)
176  {
177  const auto sgn = (l & 1 ? +1 : -1);
178  return {l, x, sgn * s_inf, -sgn * s_inf, sgn * s_inf};
179  }
180  else if (std::abs(x) < _Real{1})
181  {
182  auto _Q_lm2 = Tp{0.5L} * std::log((Tp{1} + x) / (Tp{1} - x));
183  if (l == 0)
184  return {l, x, _Q_lm2, Tp{0}, Tp{0}};
185  auto _Q_lm1 = x * _Q_lm2 - Tp{1};
186  if (l == 1)
187  return {l, x, _Q_lm1, _Q_lm2, Tp{0}};
188  auto _Q_l = Tp{2} * x * _Q_lm1 - _Q_lm2
189  - (x * _Q_lm1 - _Q_lm2) / Tp{2};
190  for (unsigned int ll = 3; ll <= l; ++ll)
191  {
192  _Q_lm2 = _Q_lm1;
193  _Q_lm1 = _Q_l;
194  // This arrangement is supposed to be better for roundoff
195  // protection, Arfken, 2nd Ed, Eq 12.17a.
196  // Is this still true for Q?
197  _Q_l = Tp{2} * x * _Q_lm1 - _Q_lm2
198  - (x * _Q_lm1 - _Q_lm2) / Tp(ll);
199  }
200 
201  return {l, x, _Q_l, _Q_lm1, _Q_lm2};
202  }
203  else
204  {
205  // FIXME: Knock these out with one series.
206  const auto _Q_l = legendre_q_series(l, x);
207  if (l == 0)
208  return {l, x, _Q_l, Tp{0}, Tp{0}};
209  const auto _Q_lm1 = legendre_q_series(l - 1, x);
210  if (l == 1)
211  return {l, x, _Q_l, _Q_lm1, Tp{0}};
212  const auto _Q_lm2 = (Tp(2 * l - 1) * x * _Q_lm1
213  - Tp(l) * _Q_l) / Tp(l - 1);
214  return {l, x, _Q_l, _Q_lm1, _Q_lm2};
215  }
216  }

218  /**
219  * @brief Return the associated Legendre function by recursion
220  * on @f$ l @f$ and downward recursion on m.
221  *
222  * The associated Legendre function is derived from the Legendre function
223  * @f$ P_l(x) @f$ by the Rodrigues formula:
224  * @f[
225  * P_l^m(x) = (1 - x^2)^{m/2}\frac{d^m}{dx^m}P_l(x)
226  * @f]
227  * @note The Condon-Shortley phase factor @f$ (-1)^m @f$ is absent
228  * by default.
229  * @note @f$ P_l^m(x) = 0 @f$ if @f$ m > l @f$.
230  *
231  * @param l The degree of the associated Legendre function.
232  * @f$ l >= 0 @f$.
233  * @param m The order of the associated Legendre function.
234  * @param x The argument of the associated Legendre function.
235  * @param phase The phase of the associated Legendre function.
236  * Use -1 for the Condon-Shortley phase convention.
237  */
238  template<typename Tp>
239  emsr::assoc_legendre_p_t<Tp>
240  assoc_legendre_p(unsigned int l, unsigned int m, Tp x,
241  Tp phase = Tp{+1})
242  {
243  using _Real = emsr::num_traits_t<Tp>;
244  if (m > l)
245  return {l, m, x, Tp{0}, Tp{0}, Tp{0}};
246  else if (std::isnan(x))
247  {
248  const auto _NaN = emsr::quiet_NaN(_Real{});
249  return {l, m, x, _NaN, _NaN, _NaN, phase};
250  }
251  else if (m == 0)
252  {
253  const auto _P_l = legendre_p(l, x);
254  return {l, m, x, _P_l.P_l, _P_l.P_lm1, _P_l.P_lm2, phase};
255  }
256  else
257  {
258  auto _P_mm = Tp{1};
259  if (m > 0)
260  {
261  // Two square roots seem more accurate more of the time
262  // than just one.
263  auto root = std::sqrt(Tp{1} - x) * std::sqrt(Tp{1} + x);
264  auto fact = Tp{1};
265  for (unsigned int i = 1; i <= m; ++i)
266  {
267  // N. B. Condon-Shortley would use phase = -1 here.
268  _P_mm *= phase * fact * root;
269  fact += Tp{2};
270  }
271  }
272  if (l == m)
273  return {l, m, x, _P_mm, Tp{0}, Tp{0}, phase};
274 
275  auto _P_mp1m = Tp(2 * m + 1) * x * _P_mm;
276  if (l == m + 1)
277  return {l, m, x, _P_mp1m, _P_mm, Tp{0}, phase};
278 
279  auto _P_lm2m = _P_mm;
280  auto _P_lm1m = _P_mp1m;
281  auto _P_lm = (Tp(2 * m + 3) * x * _P_lm1m
282  - Tp(2 * m + 1) * _P_lm2m) / Tp{2};
283  for (unsigned int j = m + 3; j <= l; ++j)
284  {
285  _P_lm2m = _P_lm1m;
286  _P_lm1m = _P_lm;
287  _P_lm = (Tp(2 * j - 1) * x * _P_lm1m
288  - Tp(j + m - 1) * _P_lm2m) / Tp(j - m);
289  }
290 
291  return {l, m, x, _P_lm, _P_lm1m, _P_lm2m};
292  }
293  }

295  template<typename Tp>
296  emsr::assoc_legendre_q_t<Tp>
297  assoc_legendre_q(unsigned int l, unsigned int m, Tp x,
298  Tp phase = Tp{+1})
299  {
300  using _Real = emsr::num_traits_t<Tp>;
301  if (std::isnan(x))
302  {
303  const auto _NaN = emsr::quiet_NaN(_Real{});
304  return {l, m, x, _NaN, _NaN, _NaN, phase};
305  }
306  else if (std::abs(x) < _Real{1})
307  {
308  // Find Q_l^0 and Q_l^1 by upward recurrence on l.
309  const auto fact = (Tp{1} - x) * (Tp{1} + x);
310  const auto root = std::sqrt(Tp{1} - x) * std::sqrt(Tp{1} + x);
311  const auto log = std::log((Tp{1} + x) / (Tp{1} - x)) / Tp{2};
312 
313  const auto _Q_00 = log;
314  const auto _Q_01 = phase / root;
315  if (l == 0)
316  {
317  if (m == 0)
318  return {l, m, x, _Q_00, Tp{0}, Tp{0}, phase}; // FIXME?
319  else if (m == 1)
320  return {l, m, x, _Q_01, _Q_00, Tp{0}, phase}; // FIXME?
321  }
322 
323  const auto _Q_10 = x * _Q_00 - Tp{1};
324  const auto _Q_11 = phase * root * (log + x / fact);
325  if (l == 1)
326  {
327  if (m == 0)
328  return {l, m, x, _Q_10, Tp{0}, Tp{0}, phase}; // FIXME?
329  else if (m == 1)
330  return {l, m, x, _Q_11, _Q_10, Tp{0}, phase}; // FIXME?
331  }
332 
333  auto _Q_lm20 = _Q_00;
334  auto _Q_lm10 = _Q_10;
335  auto _Q_lm21 = _Q_01;
336  auto _Q_lm11 = _Q_11;
337  auto _Q_l0 = (Tp{3} * x * _Q_lm10 - Tp{1} * _Q_lm20) / Tp{2};
338  auto _Q_l1 = (Tp{3} * x * _Q_lm11 - Tp{2} * _Q_lm21);
339  for (unsigned int k = 3; k <= l; ++k)
340  {
341  _Q_lm20 = _Q_lm10;
342  _Q_lm10 = _Q_l0;
343  _Q_l0 = (Tp(2 * k - 1) * x * _Q_lm10
344  - Tp(k - 1) * _Q_lm20) / Tp(k);
345 
346  _Q_lm21 = _Q_lm11;
347  _Q_lm11 = _Q_l1;
348  _Q_l1 = (Tp(2 * k - 1) * x * _Q_lm11
349  - Tp(k) * _Q_lm21) / Tp(k - 1);
350  }
351  if (m == 0)
352  return {l, m, x, _Q_l0, Tp{0}, Tp{0}, phase}; // FIXME?
353  else if (m == 1)
354  return {l, m, x, _Q_l1, _Q_l0, Tp{0}, phase}; // FIXME?
355 
356  // Find Q_l^m by upward recurrence on m.
357  auto _Q_lmm2 = _Q_l0;
358  auto _Q_lmm1 = _Q_l1;
359  auto _Q_lm = phase * (Tp{2} * x * _Q_lmm1 / root
360  + Tp(l + 1) * Tp(l) * _Q_l0);
361  for (unsigned int k = 3; k <= m; ++k)
362  {
363  _Q_lmm2 = _Q_lmm1;
364  _Q_lmm1 = _Q_lm;
365  _Q_lm = phase * (Tp(2 * (k - 1)) * x * _Q_lmm1 / root
366  - Tp(l + k - 1) * Tp(l - k + 2) * _Q_l0);
367  }
368 
369  return {l, m, x, _Q_lm, _Q_lmm1, _Q_lmm2, phase};
370  }
371  else
372  { // FIXME!
373  return {l, m, x, Tp{0}, Tp{0}, Tp{0}, phase};
374  }
375  }

377  /**
378  * @brief Return the spherical associated Legendre function.
379  *
380  * The spherical associated Legendre function of @f$ l @f$, @f$ m @f$,
381  * and @f$ \theta @f$ is defined as @f$ Y_l^m(\theta,0) @f$ where
382  * @f[
383  * Y_l^m(\theta,\phi) = (-1)^m[\frac{(2l+1)}{4\pi}
384  * \frac{(l-m)!}{(l+m)!}]
385  * P_l^m(\cos\theta) \exp^{im\phi}
386  * @f]
387  * is the spherical harmonic function and @f$ P_l^m(x) @f$ is the
388  * associated Legendre function.
389  *
390  * This function differs from the associated Legendre function by
391  * argument (@f$x = \cos(\theta)@f$) and by a normalization factor
392  * but this factor is rather large for large @f$ l @f$ and @f$ m @f$
393  * and so this function is stable for larger differences of @f$ l @f$
394  * and @f$ m @f$.
395  * @note Unlike the case for assoc_legendre_p the Condon-Shortley
396  * phase factor @f$ (-1)^m @f$ is present here.
397  * @note @f$ Y_l^m(\theta) = 0 @f$ if @f$ m > l @f$.
398  *
399  * @param l The degree of the spherical associated Legendre function.
400  * @f$ l >= 0 @f$.
401  * @param m The order of the spherical associated Legendre function.
402  * @param theta The radian polar angle argument
403  * of the spherical associated Legendre function.
404  */
405  template<typename Tp>
406  Tp
407  sph_legendre(unsigned int l, unsigned int m, Tp theta)
408  {
409  if (std::isnan(theta))
410  return emsr::quiet_NaN(theta);
411 
412  const auto x = std::cos(theta);
413 
414  if (m > l)
415  return Tp{0};
416  else if (m == 0)
417  {
418  auto _P_l = legendre_p(l, x).P_l;
419  Tp fact = std::sqrt(Tp(2 * l + 1)
420  / (Tp{4} * emsr::pi_v<Tp>));
421  _P_l *= fact;
422  return _P_l;
423  }
424  else if (x == Tp{1} || x == -Tp{1})
425  return Tp{0}; // m > 0 here
426  else
427  {
428  // m > 0 and |x| < 1 here
429 
430  // Starting value for recursion.
431  // Y_m^m(x) = \sqrt{ (2m+1)/(4pi m) \Gamma(m+1/2)/\Gamma(m) }
432  // (-1)^m (1-x^2)^(m/2) / \pi^(1/4)
433  const auto sgn = (m % 2 == 1 ? -Tp{1} : Tp{1});
434  const auto _Y_mp1m_factor = x * std::sqrt(Tp(2 * m + 3));
435  const auto lncirc = std::log1p(-x * x);
436  // Gamma(m+1/2) / Gamma(m)
437  const auto lnpoch = log_gamma(Tp(m + 0.5L))
438  - log_gamma(Tp(m));
439  const auto lnpre_val =
440  -Tp{0.25L} * emsr::lnpi_v<Tp>
441  + Tp{0.5L} * (lnpoch + m * lncirc);
442  const auto sr = std::sqrt((Tp{2} + Tp{1} / m)
443  / (Tp{4} * emsr::pi_v<Tp>));
444  auto _Y_mm = sgn * sr * std::exp(lnpre_val);
445  auto _Y_mp1m = _Y_mp1m_factor * _Y_mm;
446 
447  if (l == m)
448  return _Y_mm;
449  else if (l == m + 1)
450  return _Y_mp1m;
451  else
452  {
453  auto _Y_lm = Tp{0};
454 
455  // Compute Y_l^m, l > m+1, upward recursion on l.
456  for (auto ll = m + 2; ll <= l; ++ll)
457  {
458  const auto rat1 = Tp(ll - m) / Tp(ll + m);
459  const auto rat2 = Tp(ll - m - 1) / Tp(ll + m - 1);
460  const auto fact1 = std::sqrt(rat1 * Tp(2 * ll + 1)
461  * Tp(2 * ll - 1));
462  const auto fact2 = std::sqrt(rat1 * rat2
463  * Tp(2 * ll + 1)
464  / Tp(2 * ll - 3));
465  _Y_lm = (x * _Y_mp1m * fact1
466  - Tp(ll + m - 1) * _Y_mm * fact2)
467  / Tp(ll - m);
468  _Y_mm = _Y_mp1m;
469  _Y_mp1m = _Y_lm;
470  }
471 
472  return _Y_lm;
473  }
474  }
475  }

478  /**
479  * @brief Return the spherical harmonic function.
480  *
481  * The spherical harmonic function of @f$ l @f$, @f$ m @f$,
482  * and @f$ \theta @f$, @f$ \phi @f$ is defined by:
483  * @f[
484  * Y_l^m(\theta,\phi) = (-1)^m[\frac{(2l+1)}{4\pi}
485  * \frac{(l-m)!}{(l+m)!}]
486  * P_l^{|m|}(\cos\theta) \exp^{im\phi}
487  * @f]
488  * @note @f$ Y_l^m(\theta,\phi) = 0 @f$ if @f$ |m| > l @f$.
489  *
490  * @param l The degree of the spherical harmonic function.
491  * @f$ l >= 0 @f$.
492  * @param m The order of the spherical harmonic function.
493  * @param theta The radian polar angle argument
494  * of the spherical harmonic function.
495  * @param phi The radian azimuthal angle argument
496  * of the spherical harmonic function.
497  */
498  template<typename Tp>
499  std::complex<Tp>
500  sph_harmonic(unsigned int l, int m, Tp theta, Tp phi)
501  {
502  const auto s_NaN = emsr::quiet_NaN(theta);
503  if (std::isnan(theta) || std::isnan(phi))
504  return std::complex<Tp>{s_NaN, s_NaN};
505  else if (std::abs(m) > l)
506  return std::complex<Tp>{0, 0};
507  else
508  return sph_legendre(l, std::abs(m), theta)
509  * std::polar(Tp{1}, Tp(m) * phi);
510  }

513  /**
514  * Build a list of zeros and weights for the Gauss-Legendre integration rule
515  * for the Legendre polynomial of degree @c l.
516  */
517  template<typename Tp>
518  std::vector<emsr::QuadraturePoint<Tp>>
519  legendre_zeros(unsigned int l, Tp proto = Tp{})
520  {
521  const auto s_eps = emsr::epsilon(proto);
522  const auto s_pi = emsr::pi_v<Tp>;
523  const unsigned int s_maxit = 1000u;
524 
525  std::vector<emsr::QuadraturePoint<Tp>> pt(l);
526 
527  auto m = l / 2;
528 
529  // Treat the central zero for odd degree specially.
530  // Be careful to avoid overflow of the factorials.
531  // An alternative would be to proceed with the recursion
532  // for large degree.
533  if (l & 1)
534  {
535  const auto lm = l - 1;
536  const auto mm = lm / 2;
537  auto _Am = Tp{1};
538  for (auto m = 1u; m <= mm; ++m)
539  _Am *= -Tp(2 * m - 1) / Tp(2 * m);
540  auto Plm1 = _Am;
541  auto Ppl = l * Plm1;
542  pt[m].point = Tp{0};
543  pt[m].weight = Tp{2} / Ppl / Ppl;
544  }
545 
546  for (auto i = 1u; i <= m; ++i)
547  {
548  // Clever approximation of root.
549  auto z = std::cos(s_pi * (i - Tp{1} / Tp{4})
550  / (l + Tp{1} / Tp{2}));
551  auto z1 = z;
552  auto w = Tp{0};
553  for (auto its = 0u; its < s_maxit; ++its)
554  {
555  // Compute P, P1, and P2 the Legendre polynomials of degree
556  // l, l-1, l-2 respectively by iterating through the recursion
557  // relation for the Legendre polynomials.
558  // Compute Pp the derivative of the Legendre polynomial
559  // of degree l.
560  auto P1 = Tp{0};
561  auto P = Tp{1};
562  for (auto k = 1u; k <= l; ++k)
563  {
564  auto P2 = P1;
565  P1 = P;
566  // Recursion for Legendre polynomials.
567  P = ((Tp{2} * k - Tp{1}) * z * P1
568  - (k - Tp{1}) * P2) / k;
569  }
570  // Recursion for the derivative of The Legendre polynomial.
571  auto Pp = l * (z * P - P1) / (z * z - Tp{1});
572  z1 = z;
573  // Converge on root by Newton's method.
574  z = z1 - P / Pp;
575  if (std::abs(z - z1) < s_eps)
576  {
577  w = Tp{2} / ((Tp{1} - z * z) * Pp * Pp);
578  break;
579  }
580  if (its > s_maxit)
581  throw std::logic_error("legendre_zeros: Too many iterations");
582  }
583 
584  pt[i - 1].point = -z;
585  pt[l - i].point = z;
586  pt[i - 1].weight = w;
587  pt[l - i].weight = w;
588  }
589 
590  return pt;
591  }
