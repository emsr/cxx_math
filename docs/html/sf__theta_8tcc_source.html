46  /**
47  * Compute and return the exponential @f$ \theta_2 @f$ function
48  * by series expansion:
49  * @f[
50  * \theta_2(\nu, x) = \frac{1}{\sqrt{\pi x}}
51  * \sum_{k=-\infty}^{\infty}(-1)^k e^{-(\nu+k)^2/x}
52  * @f]
53  */
54  template<typename _Tp>
55  _Tp
56  __theta_2_sum(_Tp __nu, _Tp __x)
57  {
58  using _Real = __num_traits_t<_Tp>;
59  const auto _S_eps = __gnu_cxx::__epsilon(std::abs(__x));
60  const auto _S_NaN = __gnu_cxx::__quiet_NaN(std::abs(__x));
61  const auto _S_pi = __gnu_cxx::__const_pi(std::abs(__x));
62  auto __sum = std::exp(-__nu * __nu / __x);
63  auto __sign = _Tp{-1};
64  for (auto __k = 1; __k < 20; ++__k)
65  {
66  const auto __nup = __nu + _Tp(__k);
67  const auto __termp = __sign * std::exp(-__nup * __nup / __x);
68  const auto __num = __nu - _Tp(__k);
69  const auto __termm = __sign * std::exp(-__num * __num / __x);
70  __sum += __termp + __termm;
71  __sign = -__sign;
72  if (std::abs(__termp) < _S_eps * std::abs(__sum)
73  && std::abs(__termm) < _S_eps * std::abs(__sum))
74  break;
75  }
76  return __sum / std::sqrt(_S_pi * __x);
77  }

301  /**
302  * Use the arithmetic-geometric mean to calculate the elliptic nome
303  * given the elliptic argument k.
304  * @f[
305  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
306  * @f]
307  * where @f$ k' = \sqrt{1 - k^2} @f$ is the complementary elliptic argument
308  * and @f$ @f$ is the Legendre elliptic integral of the first kind.
309  */
310  template<typename _Tp>
311  _Tp
312  __ellnome_k(_Tp __k)
313  {
314  const auto _S_pi = _Tp{3.1415926535897932384626433832795029L};
315  const auto __kp = std::sqrt(_Tp{1} - __k * __k);
316  const auto __K = __comp_ellint_1(__k);
317  const auto __Kp = __comp_ellint_1(__kp);
318  return std::exp(-_S_pi * __Kp / __K);
319  }
