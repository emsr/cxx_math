47  /**
48  * Compute and return the exponential @f$ \theta_2 @f$ function
49  * by series expansion:
50  * @f[
51  * \theta_2(\nu, x) = \frac{1}{\sqrt{\pi x}}
52  * \sum_{k=-\infty}^{\infty}(-1)^k e^{-(\nu+k)^2/x}
53  * @f]
54  */
55  template<typename Tp>
56  Tp
57  theta_2_sum(Tp nu, Tp x)
58  {
59  using Real = emsr::num_traits_t<Tp>;
60  const auto s_eps = emsr::epsilon(std::abs(x));
61  const auto s_pi = emsr::pi_v<Real>;
62  auto sum = std::exp(-nu * nu / x);
63  auto sign = Tp{-1};
64  for (auto k = 1; k < 20; ++k)
65  {
66  const auto nup = nu + Tp(k);
67  const auto termp = sign * std::exp(-nup * nup / x);
68  const auto num = nu - Tp(k);
69  const auto termm = sign * std::exp(-num * num / x);
70  sum += termp + termm;
71  sign = -sign;
72  if (std::abs(termp) < s_eps * std::abs(sum)
73  && std::abs(termm) < s_eps * std::abs(sum))
74  break;
75  }
76  return sum / std::sqrt(s_pi * x);
77  }

109  /**
110  * Compute and return the exponential @f$ \theta_2 @f$ function
111  * by asymptotic series expansion:
112  * @f[
113  * \theta_2(\nu, x) = 2\sum_{k=0}^{\infty} e^{-((k+1/2)\pi)^2 x}
114  * \cos((2k+1)\nu\pi)
115  * @f]
116  */
117  template<typename Tp>
118  Tp
119  theta_2_asymp(Tp nu, Tp x)
120  {
121  using Real = emsr::num_traits_t<Tp>;
122  const auto s_eps = emsr::epsilon(std::abs(x));
123  const auto s_pi = emsr::pi_v<Real>;
124  auto sum = Tp{0};
125  for (auto k = 0; k < 20; ++k)
126  {
127  const auto thing = Tp(2 * k + 1) * s_pi;
128  const auto cosarg = nu * thing;
129  const auto exparg = thing * thing * x / Tp{4};
130  const auto term = std::exp(-exparg) * std::cos(cosarg);
131  sum += term;
132  if (std::abs(term) < s_eps * std::abs(sum))
133  break;
134  }
135  return Tp{2} * sum;
136  }

273  /**
274  * Use MacLaurin series to calculate the elliptic nome
275  * given the elliptic argument k.
276  * @f[
277  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
278  * @f]
279  * where @f$ k' = \sqrt{1 - k^2} @f$ is the complementary elliptic argument
280  * and @f$ @f$ is the Legendre elliptic integral of the first kind.
281  */
282  template<typename Tp>
283  Tp
284  ellnome_series(Tp k)
285  {
286  const auto m = k * k;
287  return m * ((Tp{1} / Tp{16})
288  + m * ((Tp{1} / Tp{32})
289  + m * ((Tp{21} / Tp{1024})
290  + m * ((Tp{31} / Tp{2048})
291  + m * (Tp{6257} / Tp{524288})))));
292  }

294  /**
295  * Use the arithmetic-geometric mean to calculate the elliptic nome
296  * given the elliptic argument k.
297  * @f[
298  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
299  * @f]
300  * where @f$ k' = \sqrt{1 - k^2} @f$ is the complementary elliptic argument
301  * and @f$ @f$ is the Legendre elliptic integral of the first kind.
302  */
303  template<typename Tp>
304  Tp
305  ellnome_k(Tp k)
306  {
307  const auto s_pi = Tp{3.1415926535897932384626433832795029L};
308  const auto kp = std::sqrt(Tp{1} - k * k);
309  const auto K = emsr::comp_ellint_1(k);
310  const auto Kp = emsr::comp_ellint_1(kp);
311  return std::exp(-s_pi * Kp / K);
312  }
313 
314  /**
315  * Return the elliptic nome given the modulus @c k.
316  * @f[
317  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
318  * @f]
319  */
320  template<typename Tp>
321  Tp
322  ellnome(Tp k)
323  {
324  const auto s_eps = std::numeric_limits<Tp>::epsilon();
325  if (std::isnan(k))
326  return std::numeric_limits<Tp>::quiet_NaN();
327  else if (std::abs(k) > Tp{1})
328  throw std::domain_error("ellnome: argument k out of range");
329  else if (k < std::pow(Tp{67} * s_eps, Tp{0.125L}))
330  return ellnome_series(k);
331  else
332  return ellnome_k(k);
333  }

335  /**
336  * Return the Neville @f$ \theta_s @f$ function
337  * @f[
338  * \theta_s(k,x) = \sqrt{\frac{\pi}{2 k k' K(k)}}
339  * \theta_1\left(q(k),\frac{\pi x}{2K(k)}\right)
340  * @f]
341  */
342  template<typename Tp>
343  Tp
344  theta_s(Tp k, Tp x)
345  {
346  using Real = emsr::num_traits_t<Tp>;
347  const auto s_NaN = emsr::quiet_NaN(std::abs(x));
348  const auto s_pi_2 = emsr::pi_v<Real> / Real{2};
349 
350  if (std::isnan(k) || std::isnan(x))
351  return s_NaN;
352  else if (std::abs(k) > Tp{1})
353  throw std::domain_error("theta_s: argument k out of range");
354  else
355  {
356  const auto kc = std::sqrt(Tp{1} - k * k);
357  const auto _Kk = emsr::comp_ellint_1(k);
358  const auto q = ellnome(k);
359  return std::sqrt(s_pi_2 / (k * kc * _Kk))
360  * theta_1(q, s_pi_2 * x / _Kk);
361  }
362  }

512  "Lattice frequencies must be linearly independent.");

589  /**
590  * Reduce the argument to the fundamental lattice parallelogram
591  * @f$ (0, 2\pi, 2\pi (1 + \tau), 2\pi \tau) @f$.
592  * This is sort of like a 2D lattice remquo.
593  *
594  * @param z The argument to be reduced.
595  * @return A struct containing the argument reduced to the interior
596  * of the fundamental parallelogram and two integers indicating
597  * the number of periods in the 'real' and 'tau' directions.
598  */
599  template<typename _Tp1, typename _Tp3>
600  typename jacobi_lattice_t<_Tp1, _Tp3>::arg_t
601  jacobi_lattice_t<_Tp1, _Tp3>::
602  reduce(const typename jacobi_lattice_t<_Tp1, _Tp3>::Cmplx& z) const
603  {
604  const auto s_pi = emsr::pi_v<Real>;
605 
606  const auto tau = this->tau().val;
607  const auto tau_r = std::real(tau);
608  const auto tau_i = std::imag(tau);
609  const auto z_r = std::real(z);
610  const auto z_i = std::imag(z);
611 
612  // Solve z = (z_r, z_i) = pi a (1, 0) + pi b (tau_r, tau_i).
613  const auto b = z_i / tau_i / s_pi;
614  const int n = std::floor(b);
615  const auto nu = b - n;
616  const auto a = (z_r - b * tau_r * s_pi) / s_pi;
617  const int m = std::floor(a);
618  const auto mu = a - m;
619 
620  return {m, n,

661  {

663  const auto q = lattice.ellnome();
664  const auto s_eps = emsr::epsilon(std::abs(q));
665 
666  const auto fact = Real{2} * std::pow(q, Real{0.25L});
667  this->th1p = fact;
668  this->th2 = fact;
669  this->th3 = Real{1};
670  this->th4 = Real{1};
671  this->th1ppp = Real{0};
672  this->th2pp = Real{0};
673  this->th3pp = Real{0};
674  this->th4pp = Real{0};
675  auto q2n = _Type{1};
676  for (std::size_t n = 1; n < s_max_iter; ++n)
677  {

682  this->th4 *= tm * tm;

982  return jacobi_theta_1_prod(q, x);

1010  fact *= phase / fact2;
