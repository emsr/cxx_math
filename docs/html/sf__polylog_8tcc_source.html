55  /**
56  * This class manages the termination of series.
57  * Termination conditions involve both a maximum iteration count
58  * and a relative precision.
59  */
60  template<typename Tp>
61  class _Terminator
62  {
63  private:
64 
65  using _Real = emsr::num_traits_t<Tp>;
66  const std::size_t _M_max_iter;
67  std::size_t _M_curr_iter;
68  _Real _M_toler;
69 
70  public:
71 
72  _Terminator(std::size_t max_iter, _Real mul = _Real{1})
73  : _M_max_iter(max_iter), _M_curr_iter{0},
74  _M_toler(std::abs(mul) * std::numeric_limits<_Real>::epsilon())
75  { }
76 
77  /// Return the current number of terms summed.
78  std::size_t
79  num_terms() const
80  { return this->_M_curr_iter; }
81 
82  /// Detect if the sum should terminate either because the incoming term
83  /// is small enough or the maximum number of terms has been reached.
84  bool
85  operator()(Tp term, Tp sum)
86  {
87  if (this->_M_curr_iter >= this->_M_max_iter
88  || ++this->_M_curr_iter == this->_M_max_iter)
89  return true;
90  else if (std::abs(term) < this->_M_toler * std::abs(sum))
91  return true;
92  else
93  return false;
94  }
95  };

97  /**
98  * This class manages the termination of asymptotic series.
99  * In particular, this termination watches for the growth of the sequence
100  * of terms to stop the series.
101  *
102  * Termination conditions involve both a maximum iteration count
103  * and a relative precision.
104  */
105  template<typename Tp>
106  class _AsympTerminator
107  {
108  private:
109 
110  using _Real = emsr::num_traits_t<Tp>;
111  const std::size_t _M_max_iter;
112  std::size_t _M_curr_iter;
113  _Real _M_toler;
114  _Real _M_prev_term = std::numeric_limits<_Real>::max();
115  bool _M_stop_asymp = false;
116 
117  public:
118 
119  _AsympTerminator(std::size_t max_iter, _Real mul = _Real{1})
120  : _M_max_iter(max_iter), _M_curr_iter{0},
121  _M_toler(std::abs(mul) * std::numeric_limits<_Real>::epsilon())
122  { }
123 
124  /// Filter a term before applying it to the sum.
125  Tp
126  operator<<(Tp term)
127  {
128  if (std::abs(term) > this->_M_prev_term)
129  {
130  this->_M_stop_asymp = true;
131  return Tp{0};
132  }
133  else
134  return term;
135  }
136 
137  /// Return the current number of terms summed.
138  std::size_t
139  num_terms() const
140  { return this->_M_curr_iter; }
141 
142  /// Detect if the sum should terminate either because the incoming term
143  /// is small enough or because the terms are starting to grow or
144  // the maximum number of terms has been reached.
145  bool
146  operator()(Tp term, Tp sum)
147  {
148  if (this->_M_stop_asymp)
149  return true;
150  else
151  {
152  const auto aterm = std::abs(term);
153  this->_M_stop_asymp = (aterm > this->_M_prev_term);
154  this->_M_prev_term = aterm;
155  if (this->_M_curr_iter >= this->_M_max_iter
156  || ++this->_M_curr_iter == this->_M_max_iter)
157  return true;
158  else if (aterm < this->_M_toler * std::abs(sum))
159  return true;
160  else if (this->_M_stop_asymp)
161  return true;
162  else
163  return false;
164  }
165  }
166  };

168  template<typename Tp>
169  std::complex<Tp>
170  clamp_pi(std::complex<Tp> z)
171  {
172  using _Real = emsr::num_traits_t<Tp>;
173  const auto s_pi = emsr::pi_v<_Real>;
174  const auto s_i2pi = std::complex<Tp>{0, Tp{2} * s_pi};
175  while (z.imag() > s_pi)
176  z -= s_i2pi;
177  while (z.imag() <= -s_pi)
178  z += s_i2pi;
179  return z;
180  }

195  /**
196  * This function treats the cases of positive integer index s
197  * for complex argument w.
198  *
199  * @f[
200  * Li_s(e^w) = \sum_{k=0, k != s-1} \zeta(s-k) \frac{w^k}{k!}
201  * + \left[H_{s-1} - \log(-w)\right] \frac{w^{s-1}}{(s-1)!}
202  * @f]
203  * The radius of convergence is @f$ |w| < 2 \pi @f$.
204  * Note that this series involves a @f$ \log(-x) @f$.
205  * gcc and Mathematica differ in their implementation
206  * of @f$ \log(e^{i\pi}) @f$:
207  * gcc: @f$ \log(e^{+-i\pi}) = +-i\pi @f$
208  * whereas Mathematica doesn't preserve the sign in this case:
209  * @f$ \log(e^{+- i\pi}) = +i \pi @f$
210  *
211  * @param s the positive integer index.
212  * @param w the argument.
213  * @return the value of the polylogarithm.
214  */
215  template<typename Tp>
216  std::complex<Tp>
217  polylog_exp_pos(unsigned int s, std::complex<Tp> w)
218  {
219  using _Real = emsr::num_traits_t<Tp>;
220  const auto s_2pi = emsr::tau_v<_Real>;
221  const auto s_pi = emsr::pi_v<_Real>;
222  const auto s_pipio6 = emsr::pi_sqr_div_6_v<_Real>;
223  std::complex<Tp> res = emsr::detail::riemann_zeta<Tp>(s);
224  auto wk = w;
225  auto fact = Tp{1};
226  auto harmonicN = Tp{1}; // HarmonicNumber_1
227  for (unsigned int k = 1; k <= s - 2; ++k)
228  {
229  res += fact * emsr::detail::riemann_zeta<Tp>(s - k) * wk;
230  wk *= w;
231  const auto temp = Tp{1} / Tp(1 + k);
232  fact *= temp;
233  harmonicN += temp;
234  }
235  // harmonicN now contains H_{s-1}.
236  // fact should be 1/(s-1)!
237  res += (harmonicN - std::log(-w)) * wk * fact;
238  wk *= w;
239  fact /= s; // 1/s!
240  res -= wk * fact / Tp{2};
241  wk *= w;
242  // Now comes the remainder of the series.
243  const auto pref = wk / s_pi / s_2pi;
244  fact /= Tp(s + 1); // 1/(s+1)!
245  // Subtract the zeroth order term.
246  res -= s_pipio6 * fact * pref;
247  fact *= Tp{2} / Tp(s + 2) * Tp{3} / Tp(s + 3);
248  const auto wbar = w / s_2pi;
249  const auto w2 = -wbar * wbar;
250  auto w2k = w2;
251  auto rzarg = Tp{2};
252  const unsigned int maxit = 200;
253  _Terminator<std::complex<Tp>> done(maxit);
254  while (true)
255  {
256  rzarg += Tp{2};
257  const auto rzeta = emsr::detail::riemann_zeta(rzarg);
258  const auto term = pref * fact * rzeta * w2k;
259  res -= term;
260  if (done(term, res))
261  break;
262  w2k *= w2;
263  fact *= Tp(rzarg) / Tp(s + rzarg)
264  * Tp(rzarg + 1) / Tp(s + rzarg + 1);
265  }
266  return res;
267  }

269  /**
270  * This function treats the cases of positive integer index s
271  * for real argument w.
272  *
273  * This specialization is worthwhile to catch the differing behaviour
274  * of log(x).
275  * @f[
276  * Li_s(e^w) = \sum_{k=0, k != s-1} \zeta(s-k) \frac{w^k}{k!}
277  * + \left[H_{s-1} - \log(-w)\right] \frac{w^{s-1}}{(s-1)!}
278  * @f]
279  * The radius of convergence is @f$ |w| < 2 \pi @f$.
280  * Note that this series involves a @f$ \log(-x) @f$.
281  * gcc and Mathematica differ in their implementation
282  * of @f$ \log(e^{i\pi}) @f$:
283  * gcc: @f$ \log(e^{+-i\pi}) = +-i\pi @f$
284  * whereas Mathematica doesn't preserve the sign in this case:
285  * @f$ \log(e^{+- i\pi}) = +i\pi @f$
286  *
287  * @param s the positive integer index.
288  * @param w the argument.
289  * @return the value of the polylogarithm.
290  */
291  template<typename Tp>
292  std::complex<Tp>
293  polylog_exp_pos(unsigned int s, Tp w)
294  {
295  const auto s_2pi = emsr::tau_v<Tp>;
296  const auto s_pi = emsr::pi_v<Tp>;
297  const auto s_pipio6 = emsr::pi_sqr_div_6_v<Tp>;
298  auto res = emsr::detail::riemann_zeta<Tp>(s);
299  auto wk = w;
300  auto fact = Tp{1};
301  auto harmonicN = Tp{1}; // HarmonicNumber_1
302  for (unsigned int k = 1; k <= s - 2; ++k)
303  {
304  res += fact * emsr::detail::riemann_zeta<Tp>(s - k) * wk;
305  wk *= w;
306  const auto temp = Tp{1} / Tp(1 + k);
307  fact *= temp;
308  harmonicN += temp;
309  }
310  // harmonicN now contains H_{s-1}
311  // fact should be 1/(s-1)!
312  const auto imagtemp = fact * wk
313  * (harmonicN - std::log(std::complex<Tp>(-w)));
314  res += std::real(imagtemp);
315  wk *= w;
316  fact /= s; // 1/s!
317  res -= wk * fact / Tp{2};
318  wk *= w;
319  // Now comes the remainder of the series.
320  const auto pref = wk / s_pi / s_2pi;
321  fact /= Tp(s + 1); // 1/(s+1)!
322  // Subtract the zeroth order term.
323  res -= s_pipio6 * fact * pref;
324  fact *= Tp{2} / Tp(s + 2) * Tp{3} / Tp(s + 3);
325  const auto wbar = w / s_2pi;
326  const auto w2 = -wbar * wbar;
327  auto w2k = w2;
328  auto rzarg = Tp{2};
329  const unsigned int maxit = 200;
330  _Terminator<Tp> done(maxit);
331  while (true)
332  {
333  rzarg += Tp{2};
334  const auto rzeta = emsr::detail::riemann_zeta(rzarg);
335  const auto term = pref * fact * rzeta * w2k;
336  res -= term;
337  if (done(term, res))
338  break;
339  w2k *= w2;
340  fact *= Tp(rzarg) / Tp(s + rzarg)
341  * Tp(rzarg + 1) / Tp(s + rzarg + 1);
342  }
343  return std::complex<Tp>(res, std::imag(imagtemp));
344  }

346  /**
347  * This function treats the cases of negative real index s.
348  * Theoretical convergence is present for @f$ |w| < 2\pi @f$.
349  * We use an optimized version of
350  * @f[
351  * Li_s(e^w) = \Gamma(1-s)(-w)^{s-1} + \frac{(2\pi)^{-s}}{\pi} A_p(w)
352  * @f]
353  * @f[
354  * A_p(w) = \sum_k \frac{\Gamma(1+k-s)}{k!}
355  * \sin\left(\frac{\pi}{2} (s-k)\right)
356  * \left(\frac{w}{2\pi}\right)^k \zeta(1+k-s)
357  * @f]
358  * @param s The negative real index
359  * @param w The complex argument
360  * @return The value of the polylogarithm.
361  */
362  template<typename Tp>
363  std::complex<Tp>
364  polylog_exp_neg(Tp s, std::complex<Tp> w)
365  {
366  const auto s_i = std::complex<Tp>{0, 1};
367  const auto s_2pi = emsr::tau_v<Tp>;
368  // Basic general loop, but s is a negative quantity here
369  // FIXME Large s makes problems.
370  // The series should be rearrangeable so that we only need
371  // the ratio Gamma(1-s)/(2 pi)^s
372  auto ls = log_gamma(Tp{1} - s);
373  auto res = std::exp(ls - (Tp{1} - s) * std::log(-w));
374  const auto wup = w / s_2pi;
375  auto w2k = wup;
376  const auto pref = Tp{2} * std::pow(s_2pi, -(Tp{1} - s));
377  // Here we factor up the ratio of Gamma(1 - s + k)/k! .
378  // This ratio should be well behaved even for large k in the series
379  // afterwards
380  // Note that we have a problem for large s.
381  // Since s is negative we evaluate the Gamma Function
382  // on the positive real axis where it is real.
383  auto gam = std::exp(ls);
384 
385  const auto phase = emsr::polar_pi(Tp{1}, s / Tp{2});
386  const auto cp = std::real(phase);
387  const auto sp = std::imag(phase);
388  // Here we add the expression that would result from ignoring
389  // the zeta function in the series.
390  const auto p = s_2pi - s_i * w;
391  const auto q = s_2pi + s_i * w;
392  // This can be optimized for real values of w
393  res += s_i * gam * (std::conj(phase) * std::pow(p, s - Tp{1})
394  - phase * std::pow(q, s - Tp{1}));
395  // The above expression is the result of
396  // sum_k Gamma(1+k-s)/k! * sin(pi (s-k)/2) (w/2/pi)^k
397  // Therefore we only need to sample values of zeta(n) on the real axis
398  // that really differ from one
399  std::complex<Tp> sum = sp * gam * emsr::detail::riemann_zeta_m_1(Tp{1} - s);
400  unsigned int j = 1;
401  gam *= (Tp{1} - s);
402  constexpr unsigned int maxit = 200;
403  _Terminator<std::complex<Tp>> done(maxit);
404  while (true)
405  {
406  const auto rzarg = Tp(1 + j) - s;
407  const auto rz = emsr::detail::riemann_zeta_m_1(rzarg);
408  Tp sine;
409  // Save repeated recalculation of the sines.
410  if (j & 1)
411  { // odd
412  sine = cp;
413  if (!((j - 1) / 2 & 1))
414  sine = -sine;
415  }
416  else
417  { // even
418  sine = sp;
419  if ((j / 2) & 1)
420  sine = -sine;
421  }
422  const auto term = w2k * (gam * sine * rz);
423  w2k *= wup;
424  ++j;
425  gam *= rzarg / Tp(j); // == 1/(j+1) we incremented j above.
426  sum += term;
427  if (done(term, sum))
428  break;
429  }
430  res += pref * sum;
431  return res;
432  }

434  /**
435  * Compute the polylogarithm for negative integer order.
436  * @f[
437  * Li_{-p}(e^w) = p!(-w)^{-(p+1)}
438  * - \sum_{k=0}^{\infty} \frac{B_{p+2k+q+1}}{(p+2k+q+1)!}
439  * \frac{(p+2k+q)!}{(2k+q)!}w^{2k+q}
440  * @f]
441  * where @f$ q = (p+1) mod 2 @f$.
442  *
443  * @param n the negative integer index @f$ n = -p @f$.
444  * @param w the argument w.
445  * @return the value of the polylogarithm.
446  */
447  template<typename Tp>
448  std::complex<Tp>
449  polylog_exp_neg(int n, std::complex<Tp> w)
450  {
451  const auto s_inf = std::numeric_limits<Tp>::infinity();
452  if (emsr::fp_is_zero(w))
453  return std::complex<Tp>{0};
454  else if (emsr::fp_is_equal(w, Tp{1}))
455  return std::complex<Tp>{s_inf, Tp{0}};
456  else
457  {
458  const int p = -n;
459  const int pp = 1 + p;
460  const int q = p & 1 ? 0 : 1;
461  const auto w2 = w * w;
462  auto wp = p & 1 ? std::complex<Tp>{1} : w;
463  unsigned int __2k = q;
464  auto gam = factorial<Tp>(p + __2k);
465  const auto pfact = factorial<Tp>(p);
466  auto res = pfact * std::pow(-w, Tp(-pp));
467  auto sum = std::complex<Tp>{};
468  constexpr unsigned int maxit = 300;
469  _Terminator<std::complex<Tp>> done(maxit);
470  while (true)
471  {
472  const auto id = (p + __2k + 1) / 2;
473  if (id == emsr::detail::Num_Euler_Maclaurin_zeta)
474  break;
475  const auto term = gam * wp
476  * Tp(emsr::detail::Euler_Maclaurin_zeta[id]);
477  sum += term;
478  if (done(term, sum))
479  break;
480  gam *= Tp(p + __2k + 1) / Tp(__2k + 1)
481  * Tp(p + __2k + 2) / Tp(__2k + 2);
482  wp *= w2;
483  __2k += 2;
484  }
485  res -= sum;
486  return res;
487  }
488  }

490  /**
491  * This function treats the cases of positive real index s.
492  *
493  * The defining series is
494  * @f[
495  * Li_s(e^w) = A_s(w) + B_s(w) + \Gamma(1-s)(-w)^{s-1}
496  * @f]
497  * with
498  * @f[
499  * A_s(w) = \sum_{k=0}^{m} \zeta(s-k)w^k/k!
500  * @f]
501  * @f[
502  * B_s(w) = \sum_{k=m+1}^{\infty} \sin(\pi/2(s-k))
503  * \Gamma(1-s+k)\zeta(1-s+k) (w/2/\pi)^k/k!
504  * @f]
505  *
506  * @param s the positive real index s.
507  * @param w The complex argument w.
508  * @return the value of the polylogarithm.
509  */
510  template<typename Tp>
511  std::complex<Tp>
512  polylog_exp_pos(Tp s, std::complex<Tp> w)
513  { // positive s
514  const auto s_2pi = emsr::tau_v<Tp>;
515  const auto s_pi = emsr::pi_v<Tp>;
516  std::complex<Tp> res = emsr::detail::riemann_zeta(s);
517  auto wk = w;
518  const auto phase = emsr::polar_pi(Tp{1}, s / Tp{2});
519  const auto cp = std::real(phase);
520  const auto sp = std::imag(phase);
521  // This is \Gamma(1-s)(-w)^{s-1}
522  res += s_pi / (Tp{2} * sp * cp)
523  * std::exp(-log_gamma(s) + (s - Tp{1}) * std::log(-w));
524  auto fact = Tp{1};
525  const auto m = static_cast<unsigned int>(std::floor(s));
526  for (unsigned int k = 1; k <= m; ++k)
527  {
528  res += wk * fact
529  * emsr::detail::riemann_zeta(s - Tp(k));
530  wk *= w;
531  fact /= Tp(1 + k);
532  }
533  // fac should now be 1/(m+1)!
534  const auto pref = Tp{2} * std::pow(s_2pi, s - Tp{1});
535  // Factor this out for now so we can compare with sum.
536  res /= pref;
537  // Now comes the remainder of the series
538  unsigned int j = 0;
539  constexpr unsigned int maxit = 100;
540  _Terminator<std::complex<Tp>> done(maxit);
541  auto wup = w / s_2pi;
542  auto wbark = std::pow(wup, Tp(m + 1));
543  // It is 1 < 2 - s + m < 2 => Gamma(2-s+m) will not overflow
544  // Here we factor up the ratio of Gamma(1 - s + k) / k!.
545  // This ratio should be well behaved even for large k
546  auto gam = gamma(Tp(2 + m) - s) * fact;
547  std::complex<Tp> sum{};
548  while (true)
549  {
550  const auto idx = m + 1 + j;
551  const auto zetaarg = Tp(1 + idx) - s;
552  const auto rz = emsr::detail::riemann_zeta(zetaarg);
553  auto sine = cp;
554  if (idx & 1) // Save the repeated calculation of the sines.
555  { // odd
556  sine = cp;
557  if (!((idx - 1) / 2 & 1))
558  sine = -sine;
559  }
560  else
561  { // even
562  sine = sp;
563  if ((idx / 2) & 1)
564  sine = -sine;
565  }
566  const auto term = wbark * sine * gam * rz;
567  wbark *= wup;
568  gam *= zetaarg / Tp(1 + idx);
569  ++j;
570  sum += term;
571  if (done(term, res + sum))
572  break;
573  }
574  res += sum;
575  return pref * res;
576  }

578  /**
579  * This function implements the asymptotic series for the polylog.
580  * It is given by
581  * @f[
582  * 2 \sum_{k=0}^{\infty} \zeta(2k) w^{s-2k}/\Gamma(s-2k+1)
583  * -i \pi w^{s-1}/\Gamma(s)
584  * @f]
585  * for @f$ Re(w) >> 1 @f$
586  *
587  * Don't check this against Mathematica 8.
588  * For real w the imaginary part of the polylog is given by
589  * @f$ Im(Li_s(e^w)) = -\pi w^{s-1}/\Gamma(s) @f$.
590  * Check this relation for any benchmark that you use.
591  *
592  * @param s the real index s.
593  * @param w the large complex argument w.
594  * @return the value of the polylogarithm.
595  */
596  template<typename Tp>
597  std::complex<Tp>
598  polylog_exp_asymp(Tp s, std::complex<Tp> w)
599  {
600  const auto s_pi = emsr::pi_v<Tp>;
601  // wgamma = w^{s-1} / Gamma(s)
602  auto wgamma = std::pow(w, s - Tp{1}) * gamma_reciprocal(s);
603  auto res = std::complex<Tp>(Tp{0}, -s_pi) * wgamma;
604  // wgamma = w^s / Gamma(s+1)
605  wgamma *= w / s;
606  constexpr unsigned int maxiter = 100;
607  _AsympTerminator<std::complex<Tp>> done(maxiter);
608  // zeta(0) w^s / Gamma(s + 1)
609  std::complex<Tp> oldterm = -Tp{0.5L} * wgamma;
610  res += Tp{2} * oldterm;
611  std::complex<Tp> term;
612  auto wq = Tp{1} / (w * w);
613  int k = 1;
614  while (true)
615  {
616  wgamma *= wq * (s + Tp(1 - 2 * k)) * (s + Tp(2 - 2 * k));
617  term = emsr::detail::riemann_zeta<Tp>(2 * k) * wgamma;
618  res += done << Tp{2} * term;
619  if (done(Tp{2} * term, res))
620  break;
621  oldterm = term;
622  ++k;
623  }
624  return res;
625  }

627  /**
628  * Theoretical convergence for Re(w) < 0.
629  *
630  * Seems to beat the other expansions for @f$ Re(w) < -\pi/2 - \pi/5 @f$.
631  * Note that this is an implementation of the basic series:
632  * @f[
633  * Li_s(e^z) = \sum_{k=1}^{\infty} e^{kz} k^{-s}
634  * @f]
635  *
636  * @param s is an arbitrary type, integral or float.
637  * @param w something with a negative real part.
638  * @return the value of the polylogarithm.
639  */
640  template<typename _PowTp, typename Tp>
641  Tp
642  polylog_exp_sum(_PowTp s, Tp w)
643  {
644  auto ew = std::exp(w);
645  const auto up = ew;
646  auto res = ew;
647  unsigned int maxiter = 500;
648  _Terminator<Tp> done(maxiter);
649  bool terminate = false;
650  unsigned int k = 2;
651  while (!terminate)
652  {
653  ew *= up;
654  Tp temp = std::pow(k, s); // This saves us a type conversion
655  const auto term = ew / temp;
656  res += term;
657  terminate = done(term, res);
658  ++k;
659  }
660  return res;
661  }

1019  *

1021  * @f[
1022  * Li_s(x) = \sum_{k=1}^{\infty} \frac{x^k}{k^s}
1023  * @f]
1024  *
1025  * @param s The real order.
1026  * @param x The real argument.
1027  * @return The complex value of the polylogarithm.
1028  */
1029  template<typename Tp>
1030  Tp
1031  polylog(Tp s, Tp x)
1032  {
1033  if (std::isnan(s) || std::isnan(x))
1034  return emsr::quiet_NaN(s);
1035  else if (emsr::fp_is_zero(x))
1036  return Tp{0};
1037  else
1038  {
1039  const auto n = emsr::fp_is_integer(s, Tp{5});
1040  if (n && n() == 1)
1041  return -std::log(Tp{1} - x);
1042  else if (n && n() == 0)

1045  // Prevent blowups caused by reflecting the branch point.
1046  return std::real(polylog_exp(s, Tp{0})

1048  else if (x < Tp{0})
1049  { // Use the reflection formula to access negative values.
1050  const auto y = std::log(-x);
1051  return std::real(polylog_exp(s, Tp{2} * y)
1052  * std::pow(Tp{2}, Tp{1} - s)
1053  - polylog_exp(s, y));
1054  }
1055  else
1056  {
1057  const auto y = std::log(x);
1058  return std::real(polylog_exp(s, y));
1059  }
1060  }
1061  }
