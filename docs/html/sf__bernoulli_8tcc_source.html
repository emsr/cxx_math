39  /**
40  * @brief This returns Bernoulli numbers from a table or by summation
41  * for larger values.
42  * @f[
43  * B_{2n} = (-1)^{n+1} 2\frac{(2n)!}{(2\pi)^{2n}} \zeta(2n)
44  * @f]
45  *
46  * Note that
47  * @f[
48  * \zeta(2n) - 1 = (-1)^{n+1} \frac{(2\pi)^{2n}}{(2n)!} B_{2n} - 2
49  * @f]
50  * are small and rapidly decreasing finctions of n.
51  *
52  * @param n the order n of the Bernoulli number.
53  * @return The Bernoulli number of order n.
54  */
55  template<typename Tp>
56  constexpr Tp
57  bernoulli_series(unsigned int n)
58  {
59  constexpr unsigned long s_num_bern_tab = 12;
60  constexpr Tp
61  s_bernoulli_2n[s_num_bern_tab]
62  {
63  Tp{1ULL},
64  Tp{1ULL} / Tp{6ULL},
65  -Tp{1ULL} / Tp{30ULL},
66  Tp{1ULL} / Tp{42ULL},
67  -Tp{1ULL} / Tp{30ULL},
68  Tp{5ULL} / Tp{66ULL},
69  -Tp{691ULL} / Tp{2730ULL},
70  Tp{7ULL} / Tp{6ULL},
71  -Tp{3617ULL} / Tp{510ULL},
72  Tp{43867ULL} / Tp{798ULL},
73  -Tp{174611ULL} / Tp{330ULL},
74  Tp{854513ULL} / Tp{138ULL}
75  };
76 
77  if (n == 0)
78  return Tp{1};
79  else if (n == 1)
80  return -Tp{1} / Tp{2};
81  // Take care of the rest of the odd ones.
82  else if (n % 2 == 1)
83  return Tp{0};
84  // Take care of some small evens that are painful for the series.
85  else if (n / 2 < s_num_bern_tab)
86  return s_bernoulli_2n[n / 2];
87  else
88  {
89  constexpr auto s_2pi = emsr::tau_v<Tp>;
90  auto fact = Tp{1};
91  if ((n / 2) % 2 == 0)
92  fact *= -Tp{1};
93  for (unsigned int k = 1; k <= n; ++k)
94  fact *= k / s_2pi;
95  fact *= Tp{2};
96 
97  // Riemann zeta function minus-1 for even integer argument.
98  auto sum = Tp{0};
99  for (unsigned int i = 2; i < 1000; ++i)
100  {
101  auto term = std::pow(Tp(i), -Tp(n));
102  sum += term;
103  if (term < emsr::epsilon<Tp>() * sum)
104  break;
105  }
106 
107  return fact + fact * sum;
108  }
109  }

116  */
117  template<typename Tp>

123  * @brief This returns Bernoulli number @f$ B_2n @f$ at even integer

127  * @return The Bernoulli number of order 2n.

134  /**
135  * Return the Bernoulli polynomial @f$ B_n(x) @f$ of order n at argument x.
136  *
137  * The values at 0 and 1 are equal to the corresponding Bernoulli number:
138  * @f[
139  * B_n(0) = B_n(1) = B_n
140  * @f]
141  *
142  * The derivative is proportional to the previous polynomial:
143  * @f[
144  * B_n'(x) = n * B_{n-1}(x)
145  * @f]
146  *
147  * The series expansion is:
148  * @f[
149  * B_n(x) = \sum_{k=0}^{n} B_k binom{n}{k} x^{n-k}
150  * @f]
151  *
152  * A useful argument promotion is:
153  * @f[
154  * B_n(x+1) - B_n(x) = n * x^{n-1}
155  * @f]
156  */
157  template<typename Tp>
158  Tp
159  bernoulli(unsigned int n, Tp x)
160  {
161  if (std::isnan(x))
162  return std::numeric_limits<Tp>::quiet_NaN();
163  else
164  {
165  auto _B_n = bernoulli<Tp>(0);
166  auto binomial = Tp{1};
167  for (auto k = 1u; k <= n; ++k)
168  {
169  binomial *= Tp(n + 1 - k) / Tp(k);
170  _B_n = x * _B_n + binomial
171  * bernoulli<Tp>(k);
172  }
173  return _B_n;
174  }
175  }
