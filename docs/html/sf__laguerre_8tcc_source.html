58  /**
59  * @brief This routine returns the associated Laguerre polynomial
60  * of degree @f$ n @f$, order @f$ \alpha > -1 @f$ for large n.
61  * Abramowitz & Stegun, 13.5.21
62  *
63  * @tparam _Tpa The type of the degree.
64  * @tparam _Tp The type of the parameter.
65  * @param __n The degree of the Laguerre function.
66  * @param __alpha1 The order of the Laguerre function.
67  * @param __x The argument of the Laguerre function.
68  * @return The value of the Laguerre function of degree n,
69  * order @f$ \alpha @f$, and argument x.
70  *
71  * This is from the GNU Scientific Library.
72  */
73  template<typename _Tpa, typename _Tp>
74  _Tp
75  __laguerre_large_n(unsigned __n, _Tpa __alpha1, _Tp __x)
76  {
77  const auto __a = -_Tp(__n);
78  const auto __b = _Tp(__alpha1) + _Tp{1};
79  const auto __eta = _Tp{2} * __b - _Tp{4} * __a;
80  const auto __cos2th = __x / __eta;
81  const auto __sin2th = _Tp{1} - __cos2th;
82  const auto __th = std::acos(std::sqrt(__cos2th));
83  const auto __pre_h = __gnu_cxx::__const_pi_half(__x)
84  * __gnu_cxx::__const_pi_half(__x)
85  * __eta * __eta * __cos2th * __sin2th;
86 
87  const auto __lg_b = __log_gamma(_Tp(__n) + __b);
88  const auto __lnfact = __log_gamma(_Tp(__n + 1));
89 
90  const auto __pre_term1 = _Tp{0.5L} * (_Tp{1} - __b)
91  * std::log(_Tp{0.25L} * __x * __eta);
92  const auto __pre_term2 = _Tp{0.25L} * std::log(__pre_h);
93  const auto __lnpre = __lg_b - __lnfact + _Tp{0.5L} * __x
94  + __pre_term1 - __pre_term2;
95  const auto __ser_term1 = __sin_pi(__a);
96  const auto __ser_term2 = std::sin(_Tp{0.25L} * __eta
97  * (_Tp{2} * __th - std::sin(_Tp{2} * __th))
98  + __gnu_cxx::__const_pi_quarter(__x));
99  const auto __ser = __ser_term1 + __ser_term2;
100 
101  return std::exp(__lnpre) * __ser;
102  }

105  /**
106  * @brief Evaluate the polynomial based on the confluent hypergeometric
107  * function in a safe way, with no restriction on the arguments.
108  *
109  * The associated Laguerre function is defined by
110  * @f[
111  * L_n^{(\alpha)}(x) = \frac{(\alpha + 1)_n}{n!}
112  * {}_1F_1(-n; \alpha + 1; x)
113  * @f]
114  * where @f$ (\alpha)_n @f$ is the Pochhammer symbol and
115  * @f$ {}_1F_1(a; c; x) @f$ is the confluent hypergeometric function.
116  *
117  * This function assumes x != 0.
118  *
119  * This is from the GNU Scientific Library.
120  *
121  * @tparam _Tpa The type of the degree.
122  * @tparam _Tp The type of the parameter.
123  * @param __n The degree of the Laguerre function.
124  * @param __alpha1 The order of the Laguerre function.
125  * @param __x The argument of the Laguerre function.
126  * @return The value of the Laguerre function of degree n,
127  * order @f$ \alpha @f$, and argument x.
128  */
129  template<typename _Tpa, typename _Tp>
130  _Tp
131  __laguerre_hyperg(unsigned int __n, _Tpa __alpha1, _Tp __x)
132  {
133  const auto __b = _Tp(__alpha1) + _Tp{1};
134  const auto __mx = -__x;
135  const auto __tc_sgn = (__x < _Tp{0} ? _Tp{1}
136  : ((__n % 2 == 1) ? -_Tp{1} : _Tp{1}));
137  // Get |x|^n/n!
138  auto __tc = _Tp{1};
139  const auto __ax = std::abs(__x);
140  for (unsigned int __k = 1; __k <= __n; ++__k)
141  __tc *= (__ax / __k);
142 
143  auto __term = __tc * __tc_sgn;
144  auto __sum = __term;
145  for (int __k = int(__n) - 1; __k >= 0; --__k)
146  {
147  __term *= ((__b + _Tp(__k)) / _Tp{int(__n) - __k})
148  * _Tp(__k + 1) / __mx;
149  __sum += __term;
150  }
151 
152  return __sum;
153  }

156  /**
157  * @brief This routine returns the associated Laguerre polynomial
158  * of degree @c n, order @c @f$ \alpha @f$: @f$ L_n^{(\alpha)}(x) @f$
159  * by recursion.
160  *
161  * The associated Laguerre function is defined by
162  * @f[
163  * L_n^{(\alpha)}(x) = \frac{(\alpha + 1)_n}{n!}
164  * {}_1F_1(-n; \alpha + 1; x)
165  * @f]
166  * where @f$ (\alpha)_n @f$ is the Pochhammer symbol and
167  * @f$ {}_1F_1(a; c; x) @f$ is the confluent hypergeometric function.
168  *
169  * The associated Laguerre polynomial is defined for integral
170  * order @f$ \alpha = m @f$ by:
171  * @f[
172  * L_n^{(m)}(x) = (-1)^m \frac{d^m}{dx^m} L_{n + m}(x)
173  * @f]
174  * where the Laguerre polynomial is defined by:
175  * @f[
176  * L_n(x) = \frac{e^x}{n!} \frac{d^n}{dx^n} (x^ne^{-x})
177  * @f]
178  *
179  * @tparam _Tpa The type of the degree.
180  * @tparam _Tp The type of the parameter.
181  * @param __n The degree of the Laguerre function.
182  * @param __alpha1 The order of the Laguerre function.
183  * @param __x The argument of the Laguerre function.
184  * @return The value of the Laguerre function of order n,
185  * degree @f$ \alpha @f$, and argument x.
186  */
187  template<typename _Tpa, typename _Tp>
188  __gnu_cxx::__laguerre_t<_Tpa, _Tp>
189  __laguerre_recur(unsigned int __n, _Tpa __alpha1, _Tp __x)
190  {
191  // Compute L_0.
192  auto __L_0 = _Tp{1};
193  if (__n == 0)
194  return {__n, __alpha1, __x, __L_0, _Tp{0}, _Tp{0}};
195 
196  // Compute L_1^{(alpha)}.
197  auto __L_1 = -__x + _Tp{1} + _Tp(__alpha1);
198  if (__n == 1)
199  return {__n, __alpha1, __x, __L_1, __L_0, _Tp{0}};
200 
201  // Compute L_n^{(alpha)} by recursion on n.
202  auto __L_nm2 = __L_0;
203  auto __L_nm1 = __L_1;
204  auto __L_n = (_Tp{3} + _Tp(__alpha1) - __x) * __L_nm1 / _Tp{2}
205  - (_Tp{1} + _Tp(__alpha1)) * __L_nm2 / _Tp{2};
206  for (unsigned int __nn = 3; __nn <= __n; ++__nn)
207  {
208  __L_nm2 = __L_nm1;
209  __L_nm1 = __L_n;
210  __L_n = (_Tp(2 * __nn - 1) + _Tp(__alpha1) - __x)
211  * __L_nm1 / _Tp(__nn)
212  - (_Tp(__nn - 1) + _Tp(__alpha1)) * __L_nm2 / _Tp(__nn);
213  }
214 
215  // Derivative.
216  //auto __Lp_n = (_Tp(__n) * __L_nm1 - _Tp(__n + __alpha1) * __L_nm2) / __x;
217  return {__n, __alpha1, __x, __L_n, __L_nm1, __L_nm2};
218  }

220  /**
221  * Return an array of abscissae and weights for the Gauss-Laguerre rule.
222  */
223  template<typename _Tp>
224  std::vector<__gnu_cxx::__quadrature_point_t<_Tp>>
225  __laguerre_zeros(unsigned int __n, _Tp __alpha1)
226  {
227  const auto _S_eps = __gnu_cxx::__epsilon(__alpha1);
228  const unsigned int _S_maxit = 1000;
229 
230  std::vector<__gnu_cxx::__quadrature_point_t<_Tp>> __pt(__n);
231 
232  for (auto __i = 1u; __i <= __n; ++__i)
233  {
234  auto __z = _Tp{0};
235  auto __w = _Tp{0};
236  // Clever approximations for roots.
237  if (__i == 1)
238  __z += (1.0 + __alpha1)
239  * (3.0 + 0.92 * __alpha1) / (1.0 + 2.4 * __n + 1.8 * __alpha1);
240  else if (__i == 2)
241  __z += (15.0 + 6.25 * __alpha1) / (1.0 + 2.5 * __n + 0.9 * __alpha1);
242  else
243  {
244  auto __ai = __i - 2;
245  __z += ((1.0 + 2.55 * __ai) / (1.9 * __ai)
246  + 1.26 * __ai * __alpha1 / (1.0 + 3.5 * __ai))
247  * (__z - __pt[__i - 3].__point) / (1.0 + 0.3 * __alpha1);
248  }
249  // Iterate TTRR for polynomial values
250  for (auto __its = 1u; __its <= _S_maxit; ++__its)
251  {
252  auto __L2 = _Tp{0};
253  auto __L1 = _Tp{1};
254  for (auto __j = 1u; __j <= __n; ++__j)
255  {
256  auto __L3 = __L2;
257  __L2 = __L1;
258  __L1 = ((_Tp(2 * __j - 1 + __alpha1) - __z) * __L2
259  - (_Tp(__j - 1 + __alpha1)) * __L3) / _Tp(__j);
260  }
261  // Derivative.
262  auto __Lp = (_Tp(__n) * __L1 - _Tp(__n + __alpha1) * __L2) / __z;
263  // Newton's rule for root.
264  auto __z1 = __z;
265  __z = __z1 - __L1 / __Lp;
266  if (std::abs(__z - __z1) <= _S_eps)
267  {
268  auto __exparg = std::lgamma(_Tp(__alpha1 + __n))
269  - std::lgamma(_Tp(__n));
270  __w = -std::exp(__exparg) / (__Lp * __n * __L2);
271  break;
272  }
273  if (__its > _S_maxit)
274  std::__throw_logic_error("__laguerre_zeros: "
275  "Too many iterations");
276  }
277  __pt[__i - 1].__point = __z;
278  __pt[__i - 1].__weight = __w;
279  }
280  return __pt;
281  }

284  /**
285  * @brief This routine returns the associated Laguerre polynomial
286  * of degree n, order @f$ \alpha @f$: @f$ L_n^{(\alpha)}(x) @f$.
287  *
288  * The associated Laguerre function is defined by
289  * @f[
290  * L_n^{(\alpha)}(x) = \frac{(\alpha + 1)_n}{n!}
291  * {}_1F_1(-n; \alpha + 1; x)
292  * @f]
293  * where @f$ (\alpha)_n @f$ is the Pochhammer symbol and
294  * @f$ {}_1F_1(a; c; x) @f$ is the confluent hypergeometric function.
295  *
296  * The associated Laguerre polynomial is defined for integral
297  * order @f$ \alpha = m @f$ by:
298  * @f[
299  * L_n^{(m)}(x) = (-1)^m \frac{d^m}{dx^m} L_{n + m}(x)
300  * @f]
301  * where the Laguerre polynomial is defined by:
302  * @f[
303  * L_n(x) = \frac{e^x}{n!} \frac{d^n}{dx^n} (x^ne^{-x})
304  * @f]
305  *
306  * @tparam _Tpa The type of the degree.
307  * @tparam _Tp The type of the parameter.
308  * @param __n The degree of the Laguerre function.
309  * @param __alpha1 The order of the Laguerre function.
310  * @param __x The argument of the Laguerre function.
311  * @return The value of the Laguerre function of degree n,
312  * order @f$ \alpha @f$, and argument x.
313  */
314  template<typename _Tpa, typename _Tp>
315  _Tp
316  __laguerre(unsigned int __n, _Tpa __alpha1, _Tp __x)
317  {
318  const unsigned int __max_iter = 10000000;
319  if (std::isnan(__x))
320  return __gnu_cxx::__quiet_NaN(__x);
321  else if (__n == 0)
322  return _Tp{1};
323  else if (__n == 1)
324  return _Tp{1} + _Tp(__alpha1) - __x;
325  else if (__x == _Tp{0})
326  {
327  auto __prod = _Tp(__alpha1) + _Tp{1};
328  for (unsigned int __k = 2; __k <= __n; ++__k)
329  __prod *= (_Tp(__alpha1) + _Tp(__k)) / _Tp(__k);
330  return __prod;
331  }
332  else if (__n > __max_iter && _Tp(__alpha1) > -_Tp{1}
333  && __x < _Tp{2} * (_Tp(__alpha1) + _Tp{1}) + _Tp(4 * __n))
334  return __laguerre_large_n(__n, __alpha1, __x);
335  else if (_Tp(__alpha1) >= _Tp{0}
336  || (__x > _Tp{0} && _Tp(__alpha1) < -_Tp(__n + 1)))
337  return __laguerre_recur(__n, __alpha1, __x).__L_n;
338  else
339  return __laguerre_hyperg(__n, __alpha1, __x);
340  }
