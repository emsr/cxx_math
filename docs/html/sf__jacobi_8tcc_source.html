44  /**
45  * Compute the Jacobi polynomial by recursion on @c n:
46  * @f[
47  * 2 n(\alpha + \beta + n) (\alpha + \beta + 2n - 2)
48  * P^{(\alpha, \beta)}_{n}(x)
49  * = (\alpha + \beta + 2n - 1)
50  * ((\alpha^2 - \beta^2)
51  * + x(\alpha + \beta + 2n - 2)(\alpha + \beta + 2n))
52  * P^{(\alpha, \beta)}_{n-1}(x)
53  * - 2 (\alpha + n - 1)(\beta + n - 1)(\alpha + \beta + 2n)
54  * P^{(\alpha, \beta)}_{n-2}(x)
55  * @f]
56  * This works for @f$ \alpha,\beta > -1 @f$
57  *
58  * @tparam Tp The real type of the argument
59  * @param[in] n The degree of the Jacobi polynomial
60  * @param[in] alpha1 The first order parameter of the Jacobi polynomial
61  * @param[in] beta1 The second order parameter of the Jacobi polynomial
62  * @param[in] x The argument
63  */
64  template<typename Tp>
65  emsr::jacobi_t<Tp>
66  jacobi_recur(unsigned int n, Tp alpha1, Tp beta1, Tp x)
67  {
68  const auto s_NaN = emsr::quiet_NaN(x);
69 
70  if (std::isnan(alpha1) || std::isnan(beta1) || std::isnan(x))
71  return {n, alpha1, beta1, x, s_NaN, s_NaN, s_NaN};
72 
73  auto P_nm2 = Tp{1};
74  if (n == 0)
75  return {n, alpha1, beta1, x, P_nm2, Tp{0}, Tp{0}};
76 
77  const auto apb = alpha1 + beta1;
78  const auto amb = alpha1 - beta1;
79  auto P_nm1 = (amb + (apb + Tp{2}) * x) / Tp{2};
80  if (n == 1)
81  return {n, alpha1, beta1, x, P_nm1, P_nm2, Tp{0}};
82 
83  const auto a2mb2 = amb * apb;
84  const auto bah = ((apb + Tp{2}) + Tp{2});
85  const auto poo = (bah - Tp{1});
86  auto P_n = (((poo * a2mb2)
87  + ((poo - Tp{1}) * poo * bah) * x)
88  * P_nm1 - (Tp{2} * (alpha1 + Tp{1})
89  * (beta1 + Tp{1}) * bah) * P_nm2)
90  / (Tp{4} * (apb + Tp{2}) * (poo - Tp{1}));
91  for (auto k = 3u; k <= n; ++k )
92  {
93  const auto apbpk = apb + Tp(k);
94  const auto apbp2k = apbpk + Tp(k);
95  const auto apbp2km1 = apbp2k - Tp{1};
96  const auto apbp2km2 = apbp2km1 - Tp{1};
97  const auto d = Tp{2} * k * apbpk * apbp2km2;
98  const auto a = apbp2km2 * apbp2km1 * apbp2k;
99  const auto b = apbp2km1 * a2mb2;
100  const auto c = Tp{2} * (alpha1 + Tp(k - 1))
101  * (beta1 + Tp(k - 1)) * apbp2k;
102  if (d == Tp{0})
103  throw std::runtime_error("jacobi_recur: Failure in recursion");
104  P_nm2 = P_nm1;
105  P_nm1 = P_n;
106  P_n = ((b + a * x) * P_nm1 - c * P_nm2) / d;
107  }
108  //auto Pp_n = (n * (alpha1 - beta1 - apbp2k * x) * P_nm1
109  // + Tp{2} * (n + alpha1) * (n + beta1) * P_nm2)
110  // / (apbp2k * (Tp{1} - x * x));
111  return {n, alpha1, beta1, x, P_n, P_nm1, P_nm2};
112  }

114 
115  P^{(a,b)}_{k}(z) =
116 
117  (a+b+2k-1)((a-b)(a+b) + z(a+b+2k-2)(a+b+2k))
118  -------------------------------------------- P^{(a,b)}_{k-1}(z)
119  2 k(a+b+k)(a+b+2k-2)
120 
121  2(a+k-1)(b+k-1)(a+b+2k)
122  - ----------------------- P^{(a,b)}_{k-2}(z)
123  2 k(a+b+k)(a+b+2k-2)
124 
125 */

127  /**
128  * Return a vector containing the zeros of the Jacobi polynomial
129  * @f$ P_n^{(\alpha,\beta)}(x) @f$.
130  * Thias works for @f$ \alpha, \beta > -1 @f$.
131  *
132  * @tparam Tp The real type of the parameters
133  * @param[in] n The degree of the Jacobi polynomial
134  * @param[in] alpha1 The first order parameter of the Jacobi polynomial
135  * @param[in] beta1 The second order parameter of the Jacobi polynomial
136  */
137  template<typename Tp>
138  std::vector<emsr::QuadraturePoint<Tp>>
139  jacobi_zeros(unsigned int n, Tp alpha1, Tp beta1)
140  {
141  const auto s_eps = emsr::epsilon(alpha1);
142  const unsigned int s_maxit = 1000u;
143 
144  std::vector<emsr::QuadraturePoint<Tp>> pt(n);
145 
146  Tp z;
147  Tp w;
148  for (auto i = 1u; i <= n; ++i)
149  {
150  if (i == 1)
151  {
152  const auto an = alpha1 / n;
153  const auto bn = beta1 / n;
154  const auto r1 = (1.0 + alpha1) * (2.78 / (4.0 + n * n)
155  + 0.768 * an / n);
156  const auto r2 = 1.0 + 1.48 * an + 0.96 * bn
157  + 0.452 * an * an + 0.83 * an * bn;
158  z = 1.0 - r1 / r2;
159  }
160  else if (i == 2)
161  {
162  const auto r1 = (4.1 + alpha1)
163  / ((1.0 + alpha1) * (1.0 + 0.156 * alpha1));
164  const auto r2 = 1.0
165  + 0.06 * (n - 8.0) * (1.0 + 0.12 * alpha1) / n;
166  const auto r3 = 1.0
167  + 0.012 * beta1 * (1.0 + 0.25 * std::abs(alpha1))
168  / n;
169  z -= (1.0 - z) * r1 * r2 * r3;
170  }
171  else if (i == 3)
172  {
173  const auto r1 = (1.67 + 0.28 * alpha1)
174  / (1.0 + 0.37 * alpha1);
175  const auto r2 = 1.0 + 0.22 * (n - 8.0) / n;
176  const auto r3 = 1.0 + 8.0 * beta1
177  / ((6.28 + beta1) * n * n);
178  z -= (pt[0].point - z) * r1 * r2 * r3;
179  }
180  else if (i == n - 1)
181  {
182  const auto r1 = (1.0 + 0.235 * beta1)
183  / (0.766 + 0.119 * beta1);
184  const auto r2 = 1.0 / (1.0 + 0.639 * (n - 4.0)
185  / (1.0 + 0.71 * (n - 4.0)));
186  const auto r3 = 1.0 / (1.0 + 20.0 * alpha1
187  / ((7.5 + alpha1) * n * n));
188  z += (z - pt[n - 4].point) * r1 * r2 * r3;
189  }
190  else if (i == n)
191  {
192  const auto r1 = (1.0 + 0.37 * beta1)
193  / (1.67 + 0.28 * beta1);
194  const auto r2 = 1.0 / (1.0 + 0.22 * (n - 8.0) / n);
195  const auto r3 = 1.0 / (1.0 + 8.0 * alpha1
196  / ((6.28 + alpha1) * n * n));
197  z += (z - pt[n - 3].point) * r1 * r2 * r3;
198  }
199  else
200  {
201  z = 3.0 * pt[i - 2].point
202  - 3.0 * pt[i - 3].point + pt[i - 4].point;
203  }
204 
205  auto alphabeta = alpha1 + beta1;
206  for (auto its = 1u; its <= s_maxit; ++its)
207  {
208  auto temp = Tp{2} + alphabeta;
209  auto P1 = (alpha1 - beta1 + temp * z) / Tp{2};
210  auto P2 = Tp{1};
211  for (auto j = 2u; j <= n; ++j)
212  {
213  auto P3 = P2;
214  P2 = P1;
215  temp = Tp{2} * j + alphabeta;
216  auto a = Tp{2} * j * (j + alphabeta)
217  * (temp - Tp{2});
218  auto b = (temp - Tp{1})
219  * (alpha1 * alpha1 - beta1 * beta1
220  + temp * (temp - Tp{2}) * z);
221  auto c = Tp{2} * (j - 1 + alpha1)
222  * (j - 1 + beta1) * temp;
223  P1 = (b * P2 - c * P3) / a;
224  }
225  auto Pp = (n * (alpha1 - beta1 - temp * z) * P1
226  + Tp{2} * (n + alpha1) * (n + beta1) * P2)
227  / (temp * (Tp{1} - z * z));
228  auto z1 = z;
229  z = z1 - P1 / Pp;
230  if (std::abs(z - z1) <= s_eps)
231  {
232  w = std::exp(std::lgamma(alpha1 + Tp(n))
233  + std::lgamma(beta1 + Tp(n))
234  - std::lgamma(Tp(n + 1))
235  - std::lgamma(Tp(n + 1) + alphabeta))
236  * temp * std::pow(Tp{2}, alphabeta) / (Pp * P2);
237  break;
238  }
239  if (its > s_maxit)
240  throw std::logic_error("jacobi_zeros: Too many iterations");
241  }
242  pt[i - 1].point = z;
243  pt[i - 1].weight = w;
244  }
245 
246  return pt;
247  }

249  /**
250  * Return the radial polynomial @f$ R_n^m(\rho) @f$ for non-negative
251  * @f$ n and m @f$, and real radial argument @f$ \rho @f$
252  * is a polynomial of degree @f$ m + 2n @f$ in @f$ \rho @f$.
253  *
254  * The radial polynomials are defined by
255  * @f[
256  * R_n^m(\rho) = \sum_{k=0}^{\frac{n-m}{2}}
257  * \frac{(-1)^k(n-k)!}{k!(\frac{n+m}{2}-k)!(\frac{n-m}{2}-k)!}
258  * \rho^{n-2k}
259  * @f]
260  * for @f$ n - m @f$ even and identically 0 for @f$ n - m @f$ odd.
261  *
262  * The radial polynomials are related to the Jacobi polynomials:
263  * @f[
264  * R_n^m(\rho) = (-1)^n x^m P_n^{(m,0)}(1-2\rho^2)
265  * @f]
266  * for @f$ 0 <= \rho <= 1 @f$
267  *
268  * The radial polynomials can be related to the Zernike polynomials:
269  * @f[
270  * Z_n^m(\rho,\phi) = R_n^m(\rho) \cos(m\phi)
271  * @f]
272  * @f[
273  * Z_n^{-m}(\rho,\phi) = R_n^m(\rho) \sin(m\phi)
274  * @f]
275  * for non-negative @f$ m, n @f$.
276  * @see zernike for details on the Zernike polynomials.
277  *
278  * @see Principals of Optics, 7th edition, Max Born and Emil Wolf,
279  * Cambridge University Press, 1999, pp 523-525 and 905-910.
280  *
281  * @see Zernike Polynomials: Evaluation, Quadrature, and Interpolation
282  * Philip Greengard, Kirill Serkh,
283  * Technical Report YALEU/DCS/TR-1539, February 20, 2018
284  *
285  * @tparam Tp The real type of the radial coordinate
286  * @param n The non-negative degree.
287  * @param m The non-negative azimuthal order
288  * @param rho The radial argument
289  */
290  template<typename Tp>
291  Tp
292  radial_jacobi(unsigned int n, unsigned int m, Tp rho)
293  {
294  const auto s_NaN = emsr::quiet_NaN(rho);
295 
296  if (std::isnan(rho))
297  return s_NaN;
298 
299  const int nmm = int(n) - int(m);
300  if (nmm < 0)
301  return Tp{0}; // FIXME: Is this true?
302  else if (nmm % 2 == 1)
303  return Tp{0};
304  else
305  {
306  auto k = nmm / 2;
307  return (k % 2 == 0 ? +1 : -1) * std::pow(rho, m)
308  * jacobi_recur(k, Tp(m), Tp{0},
309  Tp{1} - Tp{2} * rho * rho).P_n;
310  }
311  }

313  /**
314  * Return a vector containing the zeros of the radial Jacobi polynomial
315  * @f$ P_n^{(\alpha,\beta)}(1 - 2\rho^2) @f$.
316  *
317  * @tparam Tp The real type of the radial coordinate
318  * @param[in] n The degree of the radial Jacobi polynomial
319  * @param[in] m The order of the radial Jacobi polynomial
320  */
321  template<typename Tp>
322  std::vector<emsr::QuadraturePoint<Tp>>
323  radial_jacobi_zeros(unsigned int n, unsigned int m)
324  {
325  const int nmm = int(n) - int(m);
326  if (nmm % 2 != 0)
327  return std::vector<emsr::QuadraturePoint<Tp>>();
328  else
329  {
330  auto k = (int(n) - int(m)) / 2;
331  auto roots = jacobi_zeros(k, Tp(m), Tp{0});
332  for (auto& z : roots)
333  {
334  z.point = std::sqrt((Tp{1} - z.point) / Tp{2});
335  // weight?
336  }
337  return roots;
338  }
339  }

341  /**
342  * Return the Zernike polynomial @f$ Z_n^m(\rho,\phi) @f$
343  * for non-negative integral degree @f$ n @f$, signed integral order
344  * @f$ m @f$, and real radial argument @f$ \rho @f$ and azimuthal angle
345  * @f$ \phi @f$.
346  *
347  * The even Zernike polynomials are defined by:
348  * @f[
349  * Z_n^m(\rho,\phi) = R_n^m(\rho)\cos(m\phi)
350  * @f]
351  * and the odd Zernike polynomials are defined by:
352  * @f[
353  * Z_n^{-m}(\rho,\phi) = R_n^m(\rho)\sin(m\phi)
354  * @f]
355  * for non-negative degree @f$ m @f$ and @f$ m <= n @f$
356  * and where @f$ R_n^m(\rho) @f$ is the radial polynomial
357  * (@see radial_jacobi).
358  *
359  * @see Principals of Optics, 7th edition, Max Born and Emil Wolf,
360  * Cambridge University Press, 1999, pp 523-525 and 905-910.
361  *
362  * @tparam Tp The real type of the radial coordinate and azimuthal angle
363  * @param n The non-negative integral degree.
364  * @param m The integral azimuthal order
365  * @param rho The radial coordinate
366  * @param phi The azimuthal angle
367  */
368  template<typename Tp>
369  emsr::fp_promote_t<Tp>
370  zernike(unsigned int n, int m, Tp rho, Tp phi)
371  {
372  const auto s_NaN = emsr::quiet_NaN(rho);
373 
374  if (std::isnan(rho) || std::isnan(phi))
375  return s_NaN;
376  else
377  return radial_jacobi(n, std::abs(m), rho)
378  * (m >= 0 ? std::cos(m * phi) : std::sin(m * phi));
379  }
