44  /**
45  * Return the Jacobi polynomial coefficients as a vector.
46  * @param[in] n The order of the Jacobi polynomial
47  * @param[in] alpha1 The first parameter of the Jacobi polynomial
48  * @param[in] beta1 The second parameter of the Jacobi polynomial
49  * @param[in] x The optional scaling of the coordinate; default 1.
50  */
51  template<typename _Tp>
52  std::vector<_Tp>
53  __jacobi_poly(unsigned int __n, _Tp __alpha1, _Tp __beta1)
54  {
55  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__alpha1);
56  const auto __x = _Tp{1};
57  std::vector<_Tp> __poly;
58 
59  if (std::isnan(__alpha1) || std::isnan(__beta1))
60  return __poly;
61 
62  auto __P_nm2 = _Tp{1};
63  __poly.push_back(__P_nm2);
64  if (__n == 0)
65  return __poly;
66 
67  const auto __apb = __alpha1 + __beta1;
68  const auto __amb = __alpha1 - __beta1;
69  auto __P_nm1 = (__amb + (__apb + _Tp{2}) * __x) / _Tp{2};
70  __poly.push_back(__P_nm1);
71  if (__n == 1)
72  return __poly;
73 
74  const auto __a2mb2 = __amb * __apb;
75  const auto __bah = ((__apb + _Tp{2}) + _Tp{2});
76  const auto __poo = (__bah - _Tp{1});
77  auto __P_n = (((__poo * __a2mb2)
78  + ((__poo - _Tp{1}) * __poo * __bah) * __x)
79  * __P_nm1 - (_Tp{2} * (__alpha1 + _Tp{1})
80  * (__beta1 + _Tp{1}) * __bah) * __P_nm2)
81  / (_Tp{4} * (__apb + _Tp{2}) * (__poo - _Tp{1}));
82  __poly.push_back(__P_n);
83  for (auto __k = 3; __k <= __n; ++__k )
84  {
85  const auto __apbpk = __apb + _Tp(__k);
86  const auto __apbp2k = __apbpk + _Tp(__k);
87  const auto __apbp2km1 = __apbp2k - _Tp{1};
88  const auto __apbp2km2 = __apbp2km1 - _Tp{1};
89  const auto __d = _Tp{2} * __k * __apbpk * __apbp2km2;
90  const auto __a = __apbp2km2 * __apbp2km1 * __apbp2k;
91  const auto __b = __apbp2km1 * __a2mb2;
92  const auto __c = _Tp{2} * (__alpha1 + _Tp(__k - 1))
93  * (__beta1 + _Tp(__k - 1)) * __apbp2k;
94  if (__d == _Tp{0})
95  std::__throw_runtime_error(__N("__jacobi_poly: "
96  "Failure in recursion"));
97  __P_nm2 = __P_nm1;
98  __P_nm1 = __P_n;
99  __P_n = ((__b + __a * __x) * __P_nm1 - __c * __P_nm2) / __d;
100  __poly.push_back(__P_n);
101  }
102 
103  return __poly;
104  }

106  /**
107  * Compute the Jacobi polynomial by recursion on @c n:
108  * @f[
109  * 2 n(\alpha + \beta + n) (\alpha + \beta + 2n - 2)
110  * P^{(\alpha, \beta)}_{n}(x)
111  * = (\alpha + \beta + 2n - 1)
112  * ((\alpha^2 - \beta^2)
113  * + x(\alpha + \beta + 2n - 2)(\alpha + \beta + 2n))
114  * P^{(\alpha, \beta)}_{n-1}(x)
115  * - 2 (\alpha + n - 1)(\beta + n - 1)(\alpha + \beta + 2n)
116  * P^{(\alpha, \beta)}_{n-2}(x)
117  * @f]
118  */
119  template<typename _Tp>
120  __gnu_cxx::__jacobi_t<_Tp>
121  __jacobi_recur(unsigned int __n, _Tp __alpha1, _Tp __beta1, _Tp __x)
122  {
123  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__x);
124 
125  if (std::isnan(__alpha1) || std::isnan(__beta1) || std::isnan(__x))
126  return {__n, __alpha1, __beta1, __x, _S_NaN, _S_NaN, _S_NaN};
127 
128  auto __P_nm2 = _Tp{1};
129  if (__n == 0)
130  return {__n, __alpha1, __beta1, __x, __P_nm2, _Tp{0}, _Tp{0}};
131 
132  const auto __apb = __alpha1 + __beta1;
133  const auto __amb = __alpha1 - __beta1;
134  auto __P_nm1 = (__amb + (__apb + _Tp{2}) * __x) / _Tp{2};
135  if (__n == 1)
136  return {__n, __alpha1, __beta1, __x, __P_nm1, __P_nm2, _Tp{0}};
137 
138  const auto __a2mb2 = __amb * __apb;
139  const auto __bah = ((__apb + _Tp{2}) + _Tp{2});
140  const auto __poo = (__bah - _Tp{1});
141  auto __P_n = (((__poo * __a2mb2)
142  + ((__poo - _Tp{1}) * __poo * __bah) * __x)
143  * __P_nm1 - (_Tp{2} * (__alpha1 + _Tp{1})
144  * (__beta1 + _Tp{1}) * __bah) * __P_nm2)
145  / (_Tp{4} * (__apb + _Tp{2}) * (__poo - _Tp{1}));
146  for (auto __k = 3; __k <= __n; ++__k )
147  {
148  const auto __apbpk = __apb + _Tp(__k);
149  const auto __apbp2k = __apbpk + _Tp(__k);
150  const auto __apbp2km1 = __apbp2k - _Tp{1};
151  const auto __apbp2km2 = __apbp2km1 - _Tp{1};
152  const auto __d = _Tp{2} * __k * __apbpk * __apbp2km2;
153  const auto __a = __apbp2km2 * __apbp2km1 * __apbp2k;
154  const auto __b = __apbp2km1 * __a2mb2;
155  const auto __c = _Tp{2} * (__alpha1 + _Tp(__k - 1))
156  * (__beta1 + _Tp(__k - 1)) * __apbp2k;
157  if (__d == _Tp{0})
158  std::__throw_runtime_error(__N("__jacobi_recur: "
159  "Failure in recursion"));
160  __P_nm2 = __P_nm1;
161  __P_nm1 = __P_n;
162  __P_n = ((__b + __a * __x) * __P_nm1 - __c * __P_nm2) / __d;
163  }
164  //auto __Pp_n = (__n * (__alpha1 - __beta1 - __apbp2k * __x) * __P_nm1
165  // + _Tp{2} * (__n + __alpha1) * (__n + __beta1) * __P_nm2)
166  // / (__apbp2k * (_Tp{1} - __x * __x));
167  return {__n, __alpha1, __beta1, __x, __P_n, __P_nm1, __P_nm2};
168  }

183  /**
184  * Return a vector containing the zeros of the Jacobi polynomial
185  * @f$ P_n^{(\alpha,\beta)}@f$.
186  */
187  template<typename _Tp>
188  std::vector<__gnu_cxx::__quadrature_point_t<_Tp>>
189  __jacobi_zeros(unsigned int __n, _Tp __alpha1, _Tp __beta1)
190  {
191  const auto _S_eps = __gnu_cxx::__epsilon(__alpha1);
192  const unsigned int _S_maxit = 1000u;
193 
194  std::vector<__gnu_cxx::__quadrature_point_t<_Tp>> __pt(__n);
195 
196  _Tp __z;
197  _Tp __w;
198  for (auto __i = 1u; __i <= __n; ++__i)
199  {
200  if (__i == 1)
201  {
202  const auto __an = __alpha1 / __n;
203  const auto __bn = __beta1 / __n;
204  const auto __r1 = (1.0 + __alpha1) * (2.78 / (4.0 + __n * __n)
205  + 0.768 * __an / __n);
206  const auto __r2 = 1.0 + 1.48 * __an + 0.96 * __bn
207  + 0.452 * __an * __an + 0.83 * __an * __bn;
208  __z = 1.0 - __r1 / __r2;
209  }
210  else if (__i == 2)
211  {
212  const auto __r1 = (4.1 + __alpha1)
213  / ((1.0 + __alpha1) * (1.0 + 0.156 * __alpha1));
214  const auto __r2 = 1.0
215  + 0.06 * (__n - 8.0) * (1.0 + 0.12 * __alpha1) / __n;
216  const auto __r3 = 1.0
217  + 0.012 * __beta1 * (1.0 + 0.25 * std::abs(__alpha1))
218  / __n;
219  __z -= (1.0 - __z) * __r1 * __r2 * __r3;
220  }
221  else if (__i == 3)
222  {
223  const auto __r1 = (1.67 + 0.28 * __alpha1)
224  / (1.0 + 0.37 * __alpha1);
225  const auto __r2 = 1.0 + 0.22 * (__n - 8.0) / __n;
226  const auto __r3 = 1.0 + 8.0 * __beta1
227  / ((6.28 + __beta1) * __n * __n);
228  __z -= (__pt[0].__point - __z) * __r1 * __r2 * __r3;
229  }
230  else if (__i == __n - 1)
231  {
232  const auto __r1 = (1.0 + 0.235 * __beta1)
233  / (0.766 + 0.119 * __beta1);
234  const auto __r2 = 1.0 / (1.0 + 0.639 * (__n - 4.0)
235  / (1.0 + 0.71 * (__n - 4.0)));
236  const auto __r3 = 1.0 / (1.0 + 20.0 * __alpha1
237  / ((7.5 + __alpha1) * __n * __n));
238  __z += (__z - __pt[__n - 4].__point) * __r1 * __r2 * __r3;
239  }
240  else if (__i == __n)
241  {
242  const auto __r1 = (1.0 + 0.37 * __beta1)
243  / (1.67 + 0.28 * __beta1);
244  const auto __r2 = 1.0 / (1.0 + 0.22 * (__n - 8.0) / __n);
245  const auto __r3 = 1.0 / (1.0 + 8.0 * __alpha1
246  / ((6.28 + __alpha1) * __n * __n));
247  __z += (__z - __pt[__n - 3].__point) * __r1 * __r2 * __r3;
248  }
249  else
250  {
251  __z = 3.0 * __pt[__i - 2].__point
252  - 3.0 * __pt[__i - 3].__point + __pt[__i - 4].__point;
253  }
254 
255  auto __alphabeta = __alpha1 + __beta1;
256  for (auto __its = 1u; __its <= _S_maxit; ++__its)
257  {
258  auto __temp = _Tp{2} + __alphabeta;
259  auto __P1 = (__alpha1 - __beta1 + __temp * __z) / _Tp{2};
260  auto __P2 = _Tp{1};
261  for (auto __j = 2u; __j <= __n; ++__j)
262  {
263  auto __P3 = __P2;
264  __P2 = __P1;
265  __temp = _Tp{2} * __j + __alphabeta;
266  auto __a = _Tp{2} * __j * (__j + __alphabeta)
267  * (__temp - _Tp{2});
268  auto __b = (__temp - _Tp{1})
269  * (__alpha1 * __alpha1 - __beta1 * __beta1
270  + __temp * (__temp - _Tp{2}) * __z);
271  auto __c = _Tp{2} * (__j - 1 + __alpha1)
272  * (__j - 1 + __beta1) * __temp;
273  __P1 = (__b * __P2 - __c * __P3) / __a;
274  }
275  auto __Pp = (__n * (__alpha1 - __beta1 - __temp * __z) * __P1
276  + _Tp{2} * (__n + __alpha1) * (__n + __beta1) * __P2)
277  / (__temp * (_Tp{1} - __z * __z));
278  auto __z1 = __z;
279  __z = __z1 - __P1 / __Pp;
280  if (std::abs(__z - __z1) <= _S_eps)
281  {
282  __w = std::exp(std::lgamma(__alpha1 + _Tp(__n))
283  + std::lgamma(__beta1 + _Tp(__n))
284  - std::lgamma(_Tp(__n + 1))
285  - std::lgamma(_Tp(__n + 1) + __alphabeta))
286  * __temp * std::pow(_Tp{2}, __alphabeta) / (__Pp * __P2);
287  break;
288  }
289  if (__its > _S_maxit)
290  std::__throw_logic_error("__jacobi_zeros: Too many iterations");
291  }
292  __pt[__i - 1].__point = __z;
293  __pt[__i - 1].__weight = __w;
294  }
295 
296  return __pt;
297  }

299  /**
300  * Return the radial polynomial @f$ R_n^m(\rho) @f$ for non-negative
301  * degree @f$ n @f$, order @f$ m <= n @f$, and real radial
302  * argument @f$ \rho @f$.
303  *
304  * The radial polynomials are defined by
305  * @f[
306  * R_n^m(\rho) = \sum_{k=0}^{\frac{n-m}{2}}
307  * \frac{(-1)^k(n-k)!}{k!(\frac{n+m}{2}-k)!(\frac{n-m}{2}-k)!}
308  * \rho^{n-2k}
309  * @f]
310  * for @f$ n - m @f$ even and identically 0 for @f$ n - m @f$ odd.
311  * The radial polynomials can be related to the Zernike polynomials:
312  * @f[
313  * Z_n^m(\rho,\phi) = R_n^m(\rho) \cos(m\phi)
314  * @f]
315  * @f[
316  * Z_n^{-m}(\rho,\phi) = R_n^m(\rho) \sin(m\phi)
317  * @f]
318  * for non-negative @f$ m, n @f$.
319  * @see zernike for details on the Zernike polynomials.
320  *
321  * @see Principals of Optics, 7th edition, Max Born and Emil Wolf,
322  * Cambridge University Press, 1999, pp 523-525 and 905-910.
323  *
324  * @tparam _Tp The real type of the radial coordinate
325  * @param __n The non-negative degree.
326  * @param __m The non-negative azimuthal order
327  * @param __rho The radial argument
328  */
329  template<typename _Tp>
330  _Tp
331  __radial_jacobi(unsigned int __n, unsigned int __m, _Tp __rho)
332  {
333  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__rho);
334 
335  if (std::isnan(__rho))
336  return _S_NaN;
337 
338  if (__m > __n)
339  std::__throw_range_error(__N("radial_jacobi: order > degree"));
340  else if ((__n - __m) % 2 == 1)
341  return _Tp{0};
342  else
343  {
344  auto __k = (__n - __m) / 2;
345  return (__k % 2 == 0 ? +1 : -1) * std::pow(__rho, __m)
346  * __jacobi_recur(__k, _Tp(__m), _Tp{0},
347  _Tp{1} - _Tp{2} * __rho * __rho).__P_n;
348  }
349  }

351  /**
352  * Return the Zernicke polynomial @f$ Z_n^m(\rho,\phi) @f$
353  * for non-negative integral degree @f$ n @f$, signed integral order
354  * @f$ m @f$, and real radial argument @f$ \rho @f$ and azimuthal angle
355  * @f$ \phi @f$.
356  *
357  * The even Zernicke polynomials are defined by:
358  * @f[
359  * Z_n^m(\rho,\phi) = R_n^m(\rho)\cos(m\phi)
360  * @f]
361  * and the odd Zernicke polynomials are defined by:
362  * @f[
363  * Z_n^{-m}(\rho,\phi) = R_n^m(\rho)\sin(m\phi)
364  * @f]
365  * for non-negative degree @f$ m @f$ and @f$ m <= n @f$
366  * and where @f$ R_n^m(\rho) @f$ is the radial polynomial
367  * (@see __radial_jacobi).
368  *
369  * @see Principals of Optics, 7th edition, Max Born and Emil Wolf,
370  * Cambridge University Press, 1999, pp 523-525 and 905-910.
371  *
372  * @tparam _Tp The real type of the radial coordinate and azimuthal angle
373  * @param __n The non-negative integral degree.
374  * @param __m The integral azimuthal order
375  * @param __rho The radial coordinate
376  * @param __phi The azimuthal angle
377  */
378  template<typename _Tp>
379  __gnu_cxx::fp_promote_t<_Tp>
380  __zernike(unsigned int __n, int __m, _Tp __rho, _Tp __phi)
381  {
382  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__rho);
383 
384  if (std::isnan(__rho) || std::isnan(__phi))
385  return _S_NaN;
386  else
387  return __radial_jacobi(__n, std::abs(__m), __rho)
388  * (__m >= 0 ? std::cos(__m * __phi) : std::sin(__m * __phi));
389  }
