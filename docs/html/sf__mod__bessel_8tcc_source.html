61  /**
62  * @brief This routine computes the asymptotic modified cylindrical
63  * Bessel and functions of order nu: @f$ I_{\nu}(x) @f$,
64  * @f$ N_{\nu}(x) @f$. Use this for @f$ x >> \nu^2 + 1 @f$.
65  *
66  * References:
67  * (1) Handbook of Mathematical Functions,
68  * ed. Milton Abramowitz and Irene A. Stegun,
69  * Dover Publications,
70  * Section 9 p. 364, Equations 9.2.5-9.2.10
71  *
72  * @param nu The order of the Bessel functions.
73  * @param x The argument of the Bessel functions.
74  * @return A struct containing the modified cylindrical Bessel functions
75  * of the first and second kinds and their derivatives.
76  */
77  template<typename Tnu, typename Tp>
78  constexpr emsr::cyl_mod_bessel_t<Tnu, Tp, Tp>
79  cyl_bessel_ik_scaled_asymp(Tnu nu, Tp x)
80  {
81  // FIXME: This will promote float to double if Tnu is integral.
82  using Val = emsr::fp_promote_t<Tnu, Tp>;
83  using Real = emsr::num_traits_t<Val>;
84  using bess_t = emsr::cyl_mod_bessel_t<Tnu, Tp, Tp>;
85  const auto s_pi = emsr::pi_v<Real>;
86 
87  const auto sums = cyl_bessel_asymp_sums(nu, x, +1);
88 
89  const auto coef = std::sqrt(Real{1} / (Real{2} * s_pi * x));
90  return bess_t{nu, x,
91  coef * (sums._Psum - sums._Qsum),
92  coef * (sums._Rsum - sums._Ssum),
93  coef * s_pi * (sums._Psum + sums._Qsum),
94  -coef * s_pi * (sums._Rsum + sums._Ssum)};
95  }

97  /**
98  * @param nu The order of the Bessel functions.
99  * @param x The argument of the Bessel functions.
100  * @param do_scaled If true, scale I, I' by exp(-x) and K, K' by exp(+x).
101  */
102  template<typename Tnu, typename Tp>
103  constexpr emsr::cyl_mod_bessel_t<Tnu, Tp, Tp>
104  cyl_bessel_ik_asymp(Tnu nu, Tp x, bool do_scaled = false)
105  {
106  using bess_t = emsr::cyl_mod_bessel_t<Tnu, Tp, Tp>;
107 
108  auto ik = cyl_bessel_ik_scaled_asymp(nu, x);
109 
110  if (do_scaled)
111  return ik;
112  else
113  {
114  const auto exp = std::exp(x);
115  const auto iexp = Tp{1} / exp;
116  // @todo Check for over/under-flow in cyl_bessel_ik_asymp.
117  return bess_t{ik.nu_arg, ik.x_arg,
118  exp * ik.I_value, exp * ik.I_deriv,
119  iexp * ik.K_value, iexp * ik.K_deriv};
120  }
121  }

123  /**
124  * @brief Compute the modified Bessel functions @f$ I_\nu(x) @f$ and
125  * @f$ K_\nu(x) @f$ and their first derivatives
126  * @f$ I'_\nu(x) @f$ and @f$ K'_\nu(x) @f$ respectively.
127  * These four functions are computed together for numerical
128  * stability.
129  *
130  * @param nu The order of the Bessel functions.
131  * @param x The argument of the Bessel functions.
132  * @param do_scaled If true, scale I, I' by exp(-x) and K, K' by exp(+x).
133  * @return A struct containing the modified cylindrical Bessel functions
134  * of the first and second kinds and their derivatives.
135  */
136  template<typename Tp>
137  emsr::cyl_mod_bessel_t<Tp, Tp, Tp>
138  cyl_bessel_ik_steed(Tp nu, Tp x, bool do_scaled = false)
139  {
140  using bess_t = emsr::cyl_mod_bessel_t<Tp, Tp, Tp>;
141  const auto s_eps = emsr::epsilon(x);
142  const auto s_pi = emsr::pi_v<Tp>;
143  const auto s_fp_min = Tp{10} * s_eps;
144  constexpr int s_max_iter = 15000;
145  const auto s_x_min = Tp{2};
146 
147  const int n = std::nearbyint(nu);
148 
149  const auto mu = nu - Tp(n);
150  const auto mu2 = mu * mu;
151  const auto xi = Tp{1} / x;
152  const auto xi2 = Tp{2} * xi;
153  auto h = std::max(s_fp_min, nu * xi);
154  auto b = xi2 * nu;
155  auto d = Tp{0};
156  auto c = h;
157  int i;
158  for (i = 1; i <= s_max_iter; ++i)
159  {
160  b += xi2;
161  d = Tp{1} / (b + d);
162  c = b + Tp{1} / c;
163  const auto del = c * d;
164  h *= del;
165  if (std::abs(del - Tp{1}) < s_eps)
166  break;
167  }
168  if (i > s_max_iter)
169  return cyl_bessel_ik_asymp(nu, x);
170 
171  auto _Inul = s_fp_min;
172  auto _Ipnul = h * _Inul;
173  auto _Inul1 = _Inul;
174  auto _Ipnu1 = _Ipnul;
175  auto fact = nu * xi;
176  for (int l = n; l >= 1; --l)
177  {
178  const auto _Inutemp = fact * _Inul + _Ipnul;
179  fact -= xi;
180  _Ipnul = fact * _Inutemp + _Inul;
181  _Inul = _Inutemp;
182  }
183 
184  const auto f = _Ipnul / _Inul;
185  bool scaled = false;
186  Tp _Kmu, _Knu1;
187  if (x < s_x_min)
188  {
189  const auto x2 = x / Tp{2};
190  const auto pimu = s_pi * mu;
191  const auto fact = (std::abs(pimu) < s_eps
192  ? Tp{1}
193  : pimu / std::sin(pimu));
194  auto d = -std::log(x2);
195  auto e = mu * d;
196  const auto fact2 = (std::abs(e) < s_eps
197  ? Tp{1}
198  : std::sinh(e) / e);
199  const auto gamt = gamma_temme(mu);
200  auto ff = fact
201  * (gamt.gamma_1_value * std::cosh(e)
202  + gamt.gamma_2_value * fact2 * d);
203  auto sum = ff;
204  e = std::exp(e);
205  auto p = e / (Tp{2} * gamt.gamma_plus_value);
206  auto q = Tp{1} / (Tp{2} * e * gamt.gamma_minus_value);
207  auto c = Tp{1};
208  d = x2 * x2;
209  auto sum1 = p;
210  int i;
211  for (i = 1; i <= s_max_iter; ++i)
212  {
213  ff = (i * ff + p + q) / (i * i - mu2);
214  c *= d / Tp(i);
215  p /= Tp(i) - mu;
216  q /= Tp(i) + mu;
217  const auto del = c * ff;
218  sum += del;
219  const auto del1 = c * (p - Tp(i) * ff);
220  sum1 += del1;
221  if (std::abs(del) < s_eps * std::abs(sum))
222  break;
223  }
224  if (i > s_max_iter)
225  throw std::runtime_error("cyl_bessel_ik_steed: K-series failed to converge");
226  _Kmu = sum;
227  _Knu1 = sum1 * xi2;
228  }
229  else
230  {
231  scaled = true;
232  auto b = Tp{2} * (Tp{1} + x);
233  auto d = Tp{1} / b;
234  auto delh = d;
235  auto h = delh;
236  auto q1 = Tp{0};
237  auto q2 = Tp{1};
238  const auto a1 = Tp{0.25L} - mu2;
239  auto q = c = a1;
240  auto a = -a1;
241  auto s = Tp{1} + q * delh;
242  int i;
243  for (i = 2; i <= s_max_iter; ++i)
244  {
245  a -= Tp(2 * (i - 1));
246  c = -a * c / i;
247  const auto qnew = (q1 - b * q2) / a;
248  q1 = q2;
249  q2 = qnew;
250  q += c * qnew;
251  b += Tp{2};
252  d = Tp{1} / (b + a * d);
253  delh = (b * d - Tp{1}) * delh;
254  h += delh;
255  const auto dels = q * delh;
256  s += dels;
257  if (std::abs(dels / s) < s_eps)
258  break;
259  }
260  if (i > s_max_iter)
261  throw std::runtime_error("cyl_bessel_ik_steed: Steed's method failed");
262  h = a1 * h;
263  // We are scaling this branch to prevent under/overflow. Removing...
264  // * std::exp(-x)
265  _Kmu = std::sqrt(s_pi / (Tp{2} * x)) / s;
266  _Knu1 = _Kmu * (mu + x + Tp{0.5L} - h) * xi;
267  }
268 
269  auto _Kpmu = mu * xi * _Kmu - _Knu1;
270  auto _Inumu = xi / (f * _Kmu - _Kpmu);
271  auto _Inu = _Inumu * _Inul1 / _Inul;
272  auto _Ipnu = _Inumu * _Ipnu1 / _Inul;
273  for (int i = 1; i <= n; ++i)
274  _Kmu = std::exchange(_Knu1, (mu + Tp(i)) * xi2 * _Knu1 + _Kmu);
275  auto _Knu = _Kmu;
276  auto _Kpnu = nu * xi * _Kmu - _Knu1;
277 
278  if (do_scaled && !scaled)
279  {
280  const auto exp = std::exp(x);
281  const auto iexp = Tp{1} / exp;
282  _Inu *= iexp;
283  _Ipnu *= iexp;
284  _Knu *= exp;
285  _Kpnu *= exp;
286  }
287  else if (!do_scaled && scaled)
288  {
289  const auto exp = std::exp(x);
290  const auto iexp = Tp{1} / exp;
291  /// @todo Check for over/underflow for large-argument modified Bessel.
292  _Inu *= exp;
293  _Ipnu *= exp;
294  _Knu *= iexp;
295  _Kpnu *= iexp;
296  }
297 
298  return bess_t{nu, x, _Inu, _Ipnu, _Knu, _Kpnu};
299  }

301  /**
302  * @brief Return the modified cylindrical Bessel functions
303  * and their derivatives of order @f$ \nu @f$ by various means.
304  *
305  * @param nu The order of the Bessel functions.
306  * @param x The argument of the Bessel functions.
307  * @return A struct containing the modified cylindrical Bessel functions
308  * of the first and second kinds and their derivatives.
309  */
310  template<typename Tp>
311  emsr::cyl_mod_bessel_t<Tp, Tp, Tp>
312  cyl_bessel_ik(Tp nu, Tp x, bool do_scaled = false)
313  {
314  using bess_t = emsr::cyl_mod_bessel_t<Tp, Tp, Tp>;
315  const auto s_eps = emsr::epsilon(x);
316  const auto s_inf = emsr::infinity(x);
317  const auto s_pi = emsr::pi_v<Tp>;
318  if (nu < Tp{0})
319  {
320  const auto _Bessm = cyl_bessel_ik(-nu, x);
321  const auto sinnupi = emsr::sin_pi(-nu);
322  if (std::abs(sinnupi) < s_eps) // Carefully preserve +-inf.
323  return bess_t{nu, x, _Bessm.I_value, _Bessm.I_deriv,
324  _Bessm.K_value, _Bessm.K_deriv};
325  else
326  return bess_t{nu, x,
327  _Bessm.I_value + Tp{2} * sinnupi * _Bessm.K_value / s_pi,
328  _Bessm.I_deriv + Tp{2} * sinnupi * _Bessm.K_deriv / s_pi,
329  _Bessm.K_value, _Bessm.K_deriv};
330  }
331  else if (x == Tp{0})
332  {
333  Tp _Inu, _Ipnu;
334  if (nu == Tp{0})
335  {
336  _Inu = Tp{1};
337  _Ipnu = Tp{0};
338  }
339  else if (nu == Tp{1})
340  {
341  _Inu = Tp{0};
342  _Ipnu = Tp{0.5L};
343  }
344  else
345  {
346  _Inu = Tp{0};
347  _Ipnu = Tp{0};
348  }
349  return bess_t{nu, x, _Inu, _Ipnu, s_inf, -s_inf};
350  }
351  else if (x > Tp{1000})
352  return cyl_bessel_ik_asymp(nu, x, do_scaled);
353  else
354  return cyl_bessel_ik_steed(nu, x, do_scaled);
355  }

357  /**
358  * @brief Return the regular modified Bessel function of order
359  * @f$ \nu @f$: @f$ I_{\nu}(x) @f$.
360  *
361  * The regular modified cylindrical Bessel function is:
362  * @f[
363  * I_{\nu}(x) = \sum_{k=0}^{\infty}
364  * \frac{(x/2)^{\nu + 2k}}{k!\Gamma(\nu+k+1)}
365  * @f]
366  *
367  * @param nu The order of the regular modified Bessel function.
368  * @param x The argument of the regular modified Bessel function.
369  * @return The output regular modified Bessel function.
370  */
371  template<typename Tp>
372  Tp
373  cyl_bessel_i(Tp nu, Tp x)
374  {
375  if (x < Tp{0})
376  throw std::domain_error("cyl_bessel_i: Argument < 0");
377  else if (std::isnan(nu) || std::isnan(x))
378  return emsr::quiet_NaN(x);
379  else if (nu >= Tp{0} && x * x < Tp{10} * (nu + Tp{1}))
380  return detail::cyl_bessel_ij_series(nu, x, +1, 200);
381  else
382  return cyl_bessel_ik(nu, x).I_value;
383  }
384 
385  template<typename Tp>
386  Tp
387  cyl_bessel_i_scaled(Tp nu, Tp x)
388  {
389  if (x < Tp{0})
390  throw std::domain_error("cyl_bessel_i: Argument < 0");
391  else if (std::isnan(nu) || std::isnan(x))
392  return emsr::quiet_NaN(x);
393  else
394  return cyl_bessel_ik(nu, x, true).I_value;
395  }

397  /**
398  * @brief Return the irregular modified Bessel function
399  * @f$ K_{\nu}(x) @f$ of order @f$ \nu @f$.
400  *
401  * The irregular modified Bessel function is defined by:
402  * @f[
403  * K_{\nu}(x) = \frac{\pi}{2}
404  * \frac{I_{-\nu}(x) - I_{\nu}(x)}{\sin \nu\pi}
405  * @f]
406  * where for integral @f$ \nu = n @f$ a limit is taken:
407  * @f$ lim_{\nu \to n} @f$.
408  * For negative argument we have simply:
409  * @f[
410  * K_{-\nu}(x) = K_{\nu}(x)
411  * @f]
412  *
413  * @param nu The order of the irregular modified Bessel function.
414  * @param x The argument of the irregular modified Bessel function.
415  * @return The output irregular modified Bessel function.
416  */
417  template<typename Tp>
418  Tp
419  cyl_bessel_k(Tp nu, Tp x)
420  {
421  if (x < Tp{0})
422  throw std::domain_error("cyl_bessel_k: Argument < 0");
423  else if (std::isnan(nu) || std::isnan(x))
424  return emsr::quiet_NaN(x);
425  else
426  return cyl_bessel_ik(nu, x).K_value;
427  }

441  /**
442  * @brief Compute the spherical modified Bessel functions
443  * @f$ i_n(x) @f$ and @f$ k_n(x) @f$ and their first
444  * derivatives @f$ i'_n(x) @f$ and @f$ k'_n(x) @f$
445  * respectively.
446  *
447  * @param n The order of the modified spherical Bessel function.
448  * @param x The argument of the modified spherical Bessel function.
449  * @return A struct containing the modified spherical Bessel functions
450  * of the first and second kinds and their derivatives.
451  */
452  template<typename Tp>
453  emsr::sph_mod_bessel_t<unsigned int, Tp, Tp>
454  sph_bessel_ik(unsigned int n, Tp x)
455  {
456  using sph_t = emsr::sph_mod_bessel_t<unsigned int, Tp, Tp>;
457  const auto s_NaN = emsr::quiet_NaN(x);
458 
459  if (std::isnan(x))
460  return sph_t{n, x, s_NaN, s_NaN, s_NaN, s_NaN};
461  else if (x == Tp{0})
462  {
463  const auto s_inf = emsr::infinity(x);
464  if (n == 0)
465  return sph_t{n, x, Tp{1}, Tp{0}, s_inf, -s_inf};
466  else
467  return sph_t{n, x, Tp{0}, Tp{0}, s_inf, -s_inf};
468  }
469  else
470  {
471  const auto nu = Tp(n + 0.5L);
472  auto _Bess = cyl_bessel_ik(nu, x);
473 
474  const auto factor = (emsr::sqrtpi_v<Tp> / emsr::sqrt2_v<Tp>)
475  / std::sqrt(x);
476 
477  const auto i_n = factor * _Bess.I_value;
478  const auto ip_n = factor * _Bess.I_deriv
479  - i_n / (Tp{2} * x);
480  const auto k_n = factor * _Bess.K_value;
481  const auto kp_n = factor * _Bess.K_deriv
482  - k_n / (Tp{2} * x);
483 
484  return sph_t{n, x, i_n, ip_n, k_n, kp_n};
485  }
486  }

489  /**
490  * @brief Compute the Airy functions
491  * @f$ Ai(x) @f$ and @f$ Bi(x) @f$ and their first
492  * derivatives @f$ Ai'(x) @f$ and @f$ Bi(x) @f$
493  * respectively.
494  *
495  * @param z The argument of the Airy functions.
496  * @return A struct containing the Airy functions
497  * of the first and second kinds and their derivatives.
498  */
499  template<typename Tp>
500  emsr::airy_t<Tp, Tp>
501  airy(Tp z)
502  {
503  using ai_t = emsr::airy_t<Tp, Tp>;
504  const auto s_NaN = emsr::quiet_NaN(z);
505  const auto s_inf = emsr::infinity(z);
506  const auto s_pi = emsr::pi_v<Tp>;
507  const auto s_sqrt3 = emsr::sqrt3_v<Tp>;
508  const auto absz = std::abs(z);
509  const auto rootz = std::sqrt(absz);
510  const auto xi = Tp{2} * absz * rootz / Tp{3};
511 
512  if (std::isnan(z))
513  return ai_t{z, s_NaN, s_NaN, s_NaN, s_NaN};
514  else if (z == s_inf)
515  return ai_t{z, Tp{0}, Tp{0}, s_inf, s_inf};
516  else if (z == -s_inf)
517  return ai_t{z, Tp{0}, Tp{0}, Tp{0}, Tp{0}};
518  else if (z > Tp{0})
519  {
520  const auto _Bess13 = cyl_bessel_ik(Tp{1} / Tp{3}, xi);
521  const auto _Ai = rootz * _Bess13.K_value / (s_sqrt3 * s_pi);
522  const auto _Bi = rootz * (_Bess13.K_value / s_pi
523  + Tp{2} * _Bess13.I_value / s_sqrt3);
524 
525  const auto _Bess23 = cyl_bessel_ik(Tp{2} / Tp{3}, xi);
526  const auto _Aip = -z * _Bess23.K_value / (s_sqrt3 * s_pi);
527  const auto _Bip = z * (_Bess23.K_value / s_pi
528  + Tp{2} * _Bess23.I_value / s_sqrt3);
529 
530  return ai_t{z, _Ai, _Aip, _Bi, _Bip};
531  }
532  else if (z < Tp{0})
533  {
534  const auto _Bess13 = emsr::detail::cyl_bessel_jn(Tp{1} / Tp{3}, xi);
535  const auto _Ai = +rootz * (_Bess13.J_value
536  - _Bess13.N_value / s_sqrt3) / Tp{2};
537  const auto _Bi = -rootz * (_Bess13.N_value
538  + _Bess13.J_value / s_sqrt3) / Tp{2};
539 
540  const auto _Bess23 = emsr::detail::cyl_bessel_jn(Tp{2} / Tp{3}, xi);
541  const auto _Aip = absz * (_Bess23.N_value / s_sqrt3
542  + _Bess23.J_value) / Tp{2};
543  const auto _Bip = absz * (_Bess23.J_value / s_sqrt3
544  - _Bess23.N_value) / Tp{2};
545 
546  return ai_t{z, _Ai, _Aip, _Bi, _Bip};
547  }
548  else
549  {
550  // Reference:
551  // Abramowitz & Stegun, page 446 section 10.4.4 on Airy functions.
552  // The number is Ai(0) = 3^{-2/3}/\Gamma(2/3).
553  const auto _Ai
554  = Tp{0.3550280538878172392600631860041831763979791741991772L};
555  const auto _Bi = _Ai * s_sqrt3;
556 
557  // Reference:
558  // Abramowitz & Stegun, page 446 section 10.4.5 on Airy functions.
559  // The number is Ai'(0) = -3^{-1/3}/\Gamma(1/3).
560  const auto _Aip
561  = -Tp{0.25881940379280679840518356018920396347909113835493L};
