63  /**
64  * Evolve the backwards recurrence for F, F'.
65  * @f[
66  * F_{l-1} = (S_l F_l + F_l') / R_l
67  * F_{l-1}' = (S_l F_{l-1} - R_l F_l)
68  * @f]
69  * where
70  * @f[
71  * R_l = \sqrt{1 + (\eta / l)^2}
72  * S_l = l / x + \eta / l
73  * @f]
74  */
75  template<typename _Tp>
76  std::pair<_Tp, _Tp>
77  __coulomb_f_recur(unsigned int __l_min, unsigned int __k_max,
78  _Tp __eta, _Tp __x,
79  _Tp _F_l_max, _Tp _Fp_l_max)
80  {
81  const auto __x_inv = _Tp{1}/__x;
82  auto __fcl = _F_l_max;
83  auto __fpl = _Fp_l_max;
84  const auto __l_max = __l_min + __k_max;
85  auto __l = __l_max;
86 
87  for (int __k = __k_max - 1; __k >= 0; --__k)
88  {
89  const auto __el = __eta / __l;
90  const auto __rl = std::hypot(_Tp{1}, __el);
91  const auto __sl = __el + __l * __x_inv;
92  const auto __fc_lm1 = (__fcl * __sl + __fpl) / __rl;
93  __fpl = __fc_lm1 * __sl - __fcl * __rl;
94  __fcl = __fc_lm1;
95  __l -= 1;
96  }
97 
98  return std::make_pair(__fcl, __fpl);
99  }

101  /**
102  * Evolve the forward recurrence for G, G'.
103  * @f[
104  * G_{l+1} = (S_l G_l - G_l')/R_l
105  * G_{l+1}' = R_{l+1} G_l - S_l G_{l+1}
106  * @f]
107  * where
108  * @f[
109  * R_l = \sqrt{1 + (\eta / l)^2}
110  * S_l = l / x + \eta / l
111  * @f]
112  */
113  template<typename _Tp>
114  std::pair<_Tp, _Tp>
115  __coulomb_g_recur(unsigned int __l_min, unsigned int __k_max,
116  _Tp __eta, _Tp __x,
117  _Tp _G_l_min, _Tp _Gp_l_min)
118  {
119  const auto __x_inv = _Tp{1} / __x;
120  auto __gcl = _G_l_min;
121  auto __gpl = _Gp_l_min;
122  auto __l = __l_min + 1;
123 
124  for (int k = 1; k <= __k_max; ++k)
125  {
126  const auto __el = __eta / __l;
127  const auto __rl = std::hypot(_Tp{1}, __el);
128  const auto __sl = __el + __l * __x_inv;
129  const auto __gc_lm1 = (__sl * __gcl - __gpl) / __rl;
130  __gpl = __rl * __gcl - __sl * __gc_lm1;
131  __gcl = __gc_lm1;
132  __l += 1;
133  }
134 
135  return std::make_pair(__gcl, __gpl);
136  }

138  /**
139  * Evaluate the first continued fraction, giving the ratio F'/F
140  * at the upper l value.
141  * We also determine the sign of F at that point, since it is the sign
142  * of the last denominator in the continued fraction.
143  */
144  template<typename _Tp>
145  std::pair<_Tp, _Tp>
146  __coulomb_CF1(unsigned int __l,
147  _Tp __eta, _Tp __x)
148  {
149  const auto _CF1_small = 1.e-30;
150  const auto _CF1_abort = 1.0e+05;
151  const auto _CF1_acc = _Tp{2} * std::numeric_limits<_Tp>::epsilon();
152  const auto __x_inv = _Tp{1} / __x;
153  const auto __px = __l + _Tp{1} + _CF1_abort;
154 
155  auto __pk = __l + 1;
156  auto __F = __eta / __pk + __pk * __x_inv;
157 
158  auto __fcl_sign = _Tp{1};
159 
160  if (std::abs(__F) < _CF1_small)
161  __F = _CF1_small;
162  auto __D = _Tp{0};
163  auto __C = __F;
164 
165  _Tp __df;
166  do
167  {
168  const auto __pk1 = __pk + 1;
169  const auto __ek = __eta / __pk;
170  const auto __rk2 = _Tp{1} + __ek * __ek;
171  const auto __tk = (__pk + __pk1) * (__x_inv + __ek / __pk1);
172  __D = __tk - __rk2 * __D;
173  __C = __tk - __rk2 / __C;
174  if (std::abs(__C) < _CF1_small)
175  __C = _CF1_small;
176  if (std::abs(__D) < _CF1_small)
177  __D = _CF1_small;
178  __D = _Tp{1} / __D;
179  __df = __D * __C;
180  __F *= __df;
181  if (__D < _Tp{0})
182  {
183  // sign of result depends on sign of denominator.
184  __fcl_sign = -__fcl_sign;
185  }
186  __pk = __pk1;
187  if (__pk > __px)
188  std::__throw_runtime_error("__coulomb_CF1: Too many iterations.");
189  }
190  while (std::abs(__df - _Tp{1}) > _CF1_acc);
191 
192  return std::make_pair(__F, __fcl_sign);
193  }

195  /**
196  * Evaluate the second continued fraction to obtain the ratio
197  * @f[
198  * (G' + i F') / (G + i F) := P + i Q
199  * @f]
200  * at the specified l value.
201  */
202  template<typename _Tp>
203  std::complex<_Tp>
204  __coulomb_CF2(unsigned int __l, _Tp __eta, _Tp __x)
205  {
206  const auto _S_i = std::complex<_Tp>{0, 1};
207  const auto _CF2_acc = _Tp{4} * std::numeric_limits<_Tp>::epsilon();
208  const auto _CF2_abort = 2.0e+05;
209 
210  const auto __wi = _Tp{2} * __eta;
211  const auto __x_inv = _Tp{1} / __x;
212  const auto __e2mm1 = __eta * __eta + _Tp(__l * (__l + 1));
213 
214  auto __a = std::complex<_Tp>(-__e2mm1, __eta);
215  auto __b = _Tp{2} * std::complex<_Tp>(__x - __eta, _Tp{2});
216  auto __d = std::conj(__b) / std::norm(__b);
217 
218  auto __dpq = __x_inv * std::conj(__a * __d);
219 
220  auto __pk = _Tp{0};
221  auto __PQ = std::complex<_Tp>(_Tp{0}, _Tp{1} - __eta * __x_inv);
222 
223  do
224  {
225  __PQ += __dpq;
226  __pk += _Tp{2};
227  __a += std::complex<_Tp>(__pk, __wi);
228  __b += _Tp{2} * _S_i;
229  __d = __b + __a * __d;
230  __d = std::conj(__d) / std::norm(__d);
231  __dpq *= __b * __d - _Tp{1};
232  if (__pk > _CF2_abort)
233  std::__throw_runtime_error("__coulomb_CF2: Too many iterations.");
234  }
235  while (std::abs(__dpq) > (std::abs(__PQ)) * _CF2_acc);
236 
237  //if (Q < CF2_abort * std::numeric_limits<_Tp>::epsilon() * std::abs(P))
238  //status = GSL_ELOSS;
239 
240  return __PQ;
241  }

246  template <typename _Tp>
247  std::complex<_Tp>
248  __hydrogen(unsigned int __n,
249  unsigned int __l, unsigned int __m,
250  _Tp __Z, _Tp __r, _Tp __theta, _Tp __phi)
251  {
252  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__r);
253 
254  if (std::isnan(__Z) || std::isnan(__r)
255  || std::isnan(__theta) || std::isnan(__phi))
256  return std::complex<_Tp>{_S_NaN, _S_NaN};
257  else if(__n < 1)
258  std::__throw_domain_error(__N("__hydrogen: "
259  "level number less than one"));
260  else if(__l > __n - 1)
261  std::__throw_domain_error(__N("__hydrogen: "
262  "angular momentum number too large"));
263  else if(__Z <= _Tp(0))
264  std::__throw_domain_error(__N("__hydrogen: non-positive charge"));
265  else if(__r < _Tp(0))
266  std::__throw_domain_error(__N("__hydrogen: negative radius"));
267  else
268  {
269  const auto __A = _Tp(2) * __Z / __n;
270 
271  const auto __pre = std::sqrt(__A * __A * __A / (_Tp(2) * __n));
272  const auto __ln_a = __log_gamma(__n + __l + 1);
273  const auto __ln_b = __log_gamma(__n - __l);
274  const auto __ex = std::exp((__ln_b - __ln_a) / _Tp(2));
275  const auto __norm = __pre * __ex;
276 
277  const auto __rho = __A * __r;
278  const auto __ea = std::exp(-__rho / _Tp(2));
279  const auto __pp = std::pow(__rho, __l);
280  const auto __lag = __assoc_laguerre(__n - __l - 1, 2 * __l + 1,
281  __rho);
282  const auto __sphh = __sph_legendre(__l, __m, __theta)
283  * std::polar(_Tp(1), _Tp(__m) * __phi);
284 
285  const auto __psi = __norm * __ea * __pp * __lag * __sphh;
286 
287  return __psi;
288  }
