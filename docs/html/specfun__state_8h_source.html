649  /**
650  * @brief A structure for the gamma functions required by the Temme series
651  * expansions of @f$ N_\nu(x) @f$ and @f$ K_\nu(x) @f$.
652  * @f[
653  * \Gamma_1 = \frac{1}{2\mu}
654  * \left[\frac{1}{\Gamma(1 - \mu)} - \frac{1}{\Gamma(1 + \mu)}\right]
655  * @f]
656  * and
657  * @f[
658  * \Gamma_2 = \frac{1}{2}
659  * \left[\frac{1}{\Gamma(1 - \mu)} + \frac{1}{\Gamma(1 + \mu)}\right]
660  * @f]
661  * where @f$ -1/2 <= \mu <= 1/2 @f$ is @f$ \mu = \nu - N @f$ and @f$ N @f$.
662  * is the nearest integer to @f$ \nu @f$.
663  * The values of @f$ \Gamma(1 + \mu) @f$ and @f$ \Gamma(1 - \mu) @f$
664  * are returned as well.
665  *
666  * The accuracy requirements on this are high for @f$ |\mu| < 0 @f$.
667  */
668  template<typename _Tp>
669  struct __gamma_temme_t
670  {
671  /// The input parameter of the gamma functions
672  _Tp __mu_arg;
673 
674  /// The output function @f$ 1/\Gamma(1 + \mu) @f$
675  _Tp __gamma_plus_value;
676 
677  /// The output function @f$ 1/\Gamma(1 - \mu) @f$
678  _Tp __gamma_minus_value;
679 
680  /// The output function @f$ \Gamma_1(\mu) @f$
681  _Tp __gamma_1_value;
682 
683  /// The output function @f$ \Gamma_2(\mu) @f$
684  _Tp __gamma_2_value;
685  };
