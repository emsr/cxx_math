115  /**
116  * A type describing the state of a Legendre polynomial.
117  *
118  * The method lobatto() will return the Lobatto polynomial:
119  * @f[
120  * Lo_l(x) = (1 - x^2)P'_l(x) = l \left[ P_{l - 1}(x) - x P_l(x) \right]
121  * @f]
122  */
123  template<typename _Tp>
124  struct __legendre_p_t
125  {
126  unsigned int __l;
127  _Tp __x;
128  _Tp __P_l; /// P_l(x)
129  _Tp __P_lm1; /// P_{l-1}(x)
130  _Tp __P_lm2; /// P_{l-2}(x)
131 
132  // Return the Lobatto polynomial.
133  _Tp
134  lobatto() const
135  { return __l * (__P_l - __x * __P_lm1); }
136 
137  _Tp
138  deriv() const
139  {
140  if (std::abs(__x) == _Tp{1})
141  {
142  const auto __sgn = __x == _Tp{+1}
143  ? _Tp{+1}
144  : (__l % 2 == 0 ? _Tp{-1} : _Tp{+1});
145  return __sgn * _Tp(__l) * _Tp(__l + 1) / _Tp{2};
146  }
147  else
148  return __l * (__x * __P_l - __P_lm1)
149  / ((_Tp{1} - __x) * (_Tp{1} + __x));
150  }
151  };

153  /**
154  * A type describing the state of an associated Legendre function.
155  */
156  template<typename _Tp>
157  struct __assoc_legendre_p_t
158  {
159  unsigned int __l;
160  unsigned int __m;
161  _Tp __x;
162  _Tp __P_lm; /// P_l^{(m)}(x)
163  _Tp __P_lm1m; /// P_{l-1}^{(m)}(x)
164  _Tp __P_lm2m; /// P_{l-2}^{(m)}(x)
165  _Tp __phase = 1; // -1 For Condon-Shortley.
166 
167  _Tp
168  deriv() const
169  {
170  if (std::abs(__x) == _Tp{1})
171  {
172  const auto __sgn = __x == _Tp{+1}
173  ? _Tp{+1}
174  : (__l % 2 == 0 ? _Tp{-1} : _Tp{+1});
175  if (__m == 0)
176  return __sgn * _Tp(__l) * _Tp(__l + 1) / _Tp{2};
177  else if (__m == 1)
178  {
179  const auto __sgn = __x == _Tp{+1}
180  ? _Tp{+1}
181  : (__l % 2 == 0 ? _Tp{+1} : _Tp{-1});
182  return -__phase * __sgn * std::numeric_limits<_Tp>::infinity();
183  }
184  else if (__m == 2)
185  return -__sgn * _Tp(__l + 2) * _Tp(__l + 1) / _Tp{2}
186  * _Tp(__l) * _Tp(int(__l) - 1) / _Tp{2};
187  else
188  return _Tp{0};
189  }
190  else
191  return -__phase * ((__l + __m) * __P_lm1m - __l * __x * __P_lm)
192  / ((_Tp{1} - __x) * (_Tp{1} + __x));
193  }
194  };
