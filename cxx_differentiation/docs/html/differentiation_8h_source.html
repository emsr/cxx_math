71  * @tparam Func A function type callable with a numeric type

73  * @tparam Tp The foating point type of the argument and stepsize.
74  *
75  * @param func The function to be differentated.
76  * @param x The location at which the derivative is required.
77  * @param h The stepsize.

79  template<typename Func, typename Tp>

103  /**
104  * Compute the derivative using the 4-point rule (x - h/4, x - h/2,
105  * x - 3h/4, x - h).
106  *
107  * Compute the error using the difference between the 4-point and
108  * the 2-point rule (x - h/2, x - h).
109  *
110  * i.e. run the forward differentiator with negative stepsize.
111  *
112  * @tparam Func A function type callable with a numeric type
113  * and returning the same.
114  * @tparam Tp The foating point type of the argument and stepsize.
115  *
116  * @param func The function to be differentated.
117  * @param x The location at which the derivative is required.
118  * @param h The stepsize.
119  */
120  template<typename Func, typename Tp>
121  inline derivative_t<Tp>
122  derivative_backward(Func func, Tp x, Tp h)
123  { return derivative_forward(func, x, -h); }

145  /**
146  * Compute the derivative of a function func at a point x by automatic
147  * differentiation:
148  * @f[
149  * f'(x) = Im[f(x + i\epsilon)] / \epsilon
150  * @f]
151  * This routine ignores the input stepsize and uses machine epsilon.
152  *
153  * If your function has complex overloads and works at least very near
154  * the real axis this is a very accurate routine.
155  *
156  * @tparam Func A function type callable with a numeric type
157  * and returning the same.
158  * @tparam Tp The foating point type of the argument and stepsize.
159  *
160  * @param func The function to be differentated.
161  * @param x The (real) location at which the derivative is required.
162  * @param h The initial stepsize.
163  */
164  template<typename Func, typename Tp>
165  derivative_t<Tp>
166  derivative_automatic(Func func, Tp x, Tp);
