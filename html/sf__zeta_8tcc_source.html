173  /**
174  * @brief Compute the dilogarithm function @f$ Li_2(x) @f$
175  * by summation for x <= 1.
176  *
177  * The Riemann zeta function is defined by:
178  * @f[
179  * Li_2(x) = \sum_{k=1}^{\infty} \frac{1}{k^{s}} for s > 1
180  * @f]
181  * For |x| near 1 use the reflection formulae:
182  * @f[
183  * Li_2(-x) + Li_2(1-x) = \frac{\pi^2}{6} - \ln(x) \ln(1-x)
184  * @f]
185  * @f[
186  * Li_2(-x) - Li_2(1-x) - \frac{1}{2}Li_2(1-x^2)
187  * = -\frac{\pi^2}{12} - \ln(x) \ln(1-x)
188  * @f]
189  * For x < 1 use the reflection formula:
190  * @f[
191  * Li_2(1-x) - Li_2(1-\frac{1}{1-x}) - \frac{1}{2}(\ln(x))^2
192  * @f]
193  */
194  template<typename _Tp>
195  _Tp
196  __dilog(_Tp __x)
197  {
198  constexpr unsigned long long _S_maxit = 100000ULL;
199  const auto _S_eps = 10 * __gnu_cxx::__epsilon(std::real(__x));
200  const auto _S_pipio6 = __gnu_cxx::__const_pi_sqr_div_6(std::real(__x));
201  if (__isnan(__x))
202  return __gnu_cxx::__quiet_NaN(std::real(__x));
203  else if (__x > _Tp{+1})
204  std::__throw_range_error(__N("dilog: argument greater than one"));
205  else if (__x < _Tp{-1})
206  {
207  auto __lnfact = std::log(_Tp{1} - __x);
208  return -__dilog(_Tp{1} - _Tp{1} / (_Tp{1} - __x))
209  - _Tp{0.5L} * __lnfact * __lnfact;
210  }
211  else if (__x == _Tp{0})
212  return _Tp{0};
213  else if (__x == _Tp{1})
214  return _S_pipio6;
215  else if (__x == -_Tp{1})
216  return -_Tp{0.5L} * _S_pipio6;
217  else if (__x > _Tp{0.5L})
218  return _S_pipio6 - std::log(__x) * std::log(_Tp{1} - __x)
219  - __dilog(_Tp{1} - __x);
220  else if (__x < -_Tp{0.5L})
221  return -_Tp{0.5L} * _S_pipio6 - std::log(_Tp{1} - __x) * std::log(-__x)
222  + __dilog(_Tp{1} + __x) -_Tp{0.5L} * __dilog(_Tp{1} - __x * __x);
223  else
224  {
225  _Tp __sum = 0;
226  _Tp __fact = 1;
227  for (auto __i = 1ULL; __i < _S_maxit; ++__i)
228  {
229  __fact *= __x;
230  auto __term = __fact / (__i * __i);
231  __sum += __term;
232  if (std::abs(__term) < _S_eps * std::abs(__sum))
233  break;
234  if (__i + 1 == _S_maxit)
235  std::__throw_runtime_error(__N("__dilog: sum failed"));
236  }
237  return __sum;
238  }
239  }

242  /**
243  * @brief Compute the Riemann zeta function @f$ \zeta(s) @f$
244  * by summation for s > 1.
245  *
246  * The Riemann zeta function is defined by:
247  * @f[
248  * \zeta(s) = \sum_{k=1}^{\infty} \frac{1}{k^{s}} for s > 1
249  * @f]
250  * For s < 1 use the reflection formula:
251  * @f[
252  * \zeta(s) = 2^s \pi^{s-1} \Gamma(1-s) \zeta(1-s)
253  * @f]
254  */
255  template<typename _Tp>
256  _Tp
257  __riemann_zeta_sum(_Tp __s)
258  {
259  using _Val = _Tp;
260  using _Real = __num_traits_t<_Val>;
261  // A user shouldn't get to this.
262  if (std::real(__s) < _Real{1})
263  std::__throw_domain_error(__N("__riemann_zeta_sum: "
264  "Bad argument in zeta sum."));
265 
266  constexpr unsigned int _S_max_iter = 10000;
267  const auto _S_eps = __gnu_cxx::__epsilon(std::real(__s));
268  auto __zeta = _Val{1};
269  for (unsigned int __k = 2; __k < _S_max_iter; ++__k)
270  {
271  auto __term = std::pow(_Val(__k), -__s);
272  __zeta += __term;
273  if (std::abs(__term) < _S_eps * std::abs(__zeta)
274  || std::abs(__term) < _S_eps
275  && std::abs(__zeta) < _Real{100} * _S_eps)
276  break;
277  }
278 
279  return __zeta;
280  }

283  /**
284  * @brief Evaluate the Riemann zeta function @f$ \zeta(s) @f$
285  * by an alternate series for s > 0.
286  *
287  * This is a specialization of the code for the Hurwitz zeta function.
288  */
289  template<typename _Tp>
290  _Tp
291  __riemann_zeta_euler_maclaurin(_Tp __s)
292  {
293  using _Val = _Tp;
294  using _Real = __num_traits_t<_Val>;
295  const auto _S_eps = __gnu_cxx::__epsilon(std::real(__s));
296  const auto _S_N = 10 + __gnu_cxx::__digits10(std::real(__s)) / _Tp{2};
297  const auto _S_jmax = 99;
298 
299  const auto __pmax = std::pow(_Val{_S_N + 1}, -__s);
300  const auto __denom = _Val{_S_N + 1} * _Val{_S_N + 1};
301  auto __ans = __pmax * (_Val{_S_N + 1} / (__s - _Val{1}) + _Val{0.5L});
302  for (auto __k = 0; __k < _S_N; ++__k)
303  __ans += std::pow(_Val(__k + 1), -__s);
304 
305  auto __fact = __pmax * __s / _Val{_S_N + 1};
306  auto __delta_prev = __gnu_cxx::__max(__s);
307  for (auto __j = 0; __j <= _S_jmax; ++__j)
308  {
309  auto __delta = _S_Euler_Maclaurin_zeta[__j + 1] * __fact;
310  if (std::abs(__delta) > __delta_prev)
311  break;
312  __delta_prev = std::abs(__delta);
313  __ans += __delta;
314  if (std::abs(__delta) < _Real{0.5L} * _S_eps * std::abs(__ans)
315  || std::abs(__delta) < _S_eps
316  && std::abs(__ans) < _Real{100} * _S_eps)
317  break;
318  __fact *= (__s + _Val(2 * __j + 1)) * (__s + _Val(2 * __j + 2))
319  / __denom;
320  }
321 
322  return __ans;
323  }

327  * @brief Evaluate the Riemann zeta function @f$ \zeta(s) @f$
328  * by an alternate series for s > 0.
329  *
330  * The series is:
331  * @f[
332  * \zeta(s) = \frac{1}{1-2^{1-s}}
333  * \sum_{n=1}^{\infty} \frac{(-1)^{n-1}}{n^s}
334  * @f]
335  *
336  * The Riemann zeta function is defined by:
337  * @f[
338  * \zeta(s) = \sum_{k=1}^{\infty} \frac{1}{k^{s}} for s > 1
339  * @f]
340  * For s < 1 use the reflection formula:
341  * @f[
342  * \zeta(s) = 2^s \pi^{s-1} \Gamma(1-s) \zeta(1-s)
343  * @f]
344  */
345  template<typename _Tp>
346  _Tp
347  __riemann_zeta_alt(_Tp __s)
348  {
349  using _Val = _Tp;
350  using _Real = __num_traits_t<_Val>;
351  const auto _S_eps = __gnu_cxx::__epsilon(std::real(__s));
352  const unsigned int _S_max_iter = 10000000;
353  auto __sgn = _Real{1};
354  auto __zeta = _Val{0};
355  for (unsigned int __i = 1; __i < _S_max_iter; ++__i)
356  {
357  auto __term = __sgn / std::pow(_Val(__i), __s);
358  __zeta += __term;
359  __sgn = -__sgn;
360  if (std::abs(__term) < _S_eps * std::abs(__zeta)
361  || std::abs(__term) < _S_eps
362  && std::abs(__zeta) < _Real{100} * _S_eps)
363  break;
364  }
365  __zeta /= _Val{1} - std::pow(_Val{2}, _Val{1} - __s);
366 
367  return __zeta;

371  /**
372  * @brief Evaluate the Riemann zeta function by series for all s != 1.
373  * Convergence is great until largish negative numbers.
374  * Then the convergence of the > 0 sum gets better.
375  *
376  * The series is:
377  * @f[
378  * \zeta(s) = \frac{1}{1-2^{1-s}}
379  * \sum_{n=0}^{\infty} \frac{1}{2^{n+1}}
380  * \sum_{k=0}^{n} (-1)^k \frac{n!}{(n-k)!k!} (k+1)^{-s}
381  * @f]
382  * Havil 2003, p. 206.
383  *
384  * The Riemann zeta function is defined by:
385  * @f[
386  * \zeta(s) = \sum_{k=1}^{\infty} \frac{1}{k^{s}} for s > 1
387  * @f]
388  * For s < 1 use the reflection formula:
389  * @f[
390  * \zeta(s) = 2^s \pi^{s-1} \Gamma(1-s) \zeta(1-s)
391  * @f]
392  */
393  template<typename _Tp>
394  _Tp
395  __riemann_zeta_glob(_Tp __s)
396  {
397  using _Val = _Tp;
398  using _Real = __num_traits_t<_Val>;
399 
400  auto __zeta = _Val{0};
401 
402  const auto _S_eps = __gnu_cxx::__epsilon(std::real(__s));
403  // Max e exponent before overflow.
404  const auto __max_bincoeff
405  = std::exp(__gnu_cxx::__max_exponent10(std::real(__s))
406  * std::log(_Real{10}) - _Real{1});
407  const auto _S_pi = __gnu_cxx::__const_pi(std::real(__s));
408  const auto _S_pi_2 = __gnu_cxx::__const_pi_half(std::real(__s));
409  // This series works until the binomial coefficient blows up
410  // so use reflection.
411  if (std::real(__s) < _Real{0})
412  {
413  if (std::imag(__s) == _Real{0}
414  && std::fmod(std::real(__s), _Real{2}) == _Real{0})
415  return _Val{0};
416  else
417  {
418  auto __zeta = __riemann_zeta_glob(_Val{1} - __s);
419  __zeta *= std::pow(_Real{2} * _S_pi, __s)
420  * __sin_pi(_Real{0.5L} * __s)
421  * __gamma(_Val{1} - __s) / _S_pi;
422  return __zeta;
423  }
424  }
425 
426  auto __num = _Real{0.25L};
427  const unsigned int __maxit = 10000;
428  __zeta = _Val{0.5L};
429  // This for loop starts at 1 because we already calculated the
430  // value of the zeroeth order in __zeta above
431  for (unsigned int __i = 1; __i < __maxit; ++__i)
432  {
433  bool __punt = false;
434  auto __term = _Val{1};
435  auto __bincoeff = _Real{1};
436  // This for loop starts at 1 because we already calculated the value
437  // of the zeroeth order in __term above.
438  for (unsigned int __j = 1; __j <= __i; ++__j)
439  {
440  auto __incr = _Real(__i - __j + 1) / _Real(__j);
441  __bincoeff *= -__incr;
442  if(std::abs(__bincoeff) > __max_bincoeff )
443  {
444  // This only gets hit for x << 0.
445  __punt = true;
446  break;
447  }
448  __term += __bincoeff * std::pow(_Val(1 + __j), -__s);
449  }
450  if (__punt)
451  break;
452  __term *= __num;
453  __zeta += __term;
454  if (std::abs(__term) < _S_eps * std::abs(__zeta)
455  || std::abs(__term) < _S_eps
456  && std::abs(__zeta) < _Real{100} * _S_eps)
457  break;
458  __num *= _Real{0.5L};
459  }
460 
461  __zeta /= _Real{1} - std::pow(_Real{2}, _Val{1} - __s);
462 
463  return __zeta;
