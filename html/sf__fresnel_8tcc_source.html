45  /**
46  * @brief This function returns the Fresnel cosine and sine integrals
47  * as a pair by series expansion for positive argument.
48  */
49  template <typename _Tp>
50  void
51  __fresnel_series(const _Tp __ax, _Tp & _Cf, _Tp & _Sf)
52  {
53  const auto _S_max_iter = 100;
54  const auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon(__ax);
55  const auto _S_pi = __gnu_cxx::__const_pi(__ax);
56  const auto _S_pi_2 = _S_pi / _Tp{2};
57 
58  // Evaluate S and C by series expansion.
59  auto __sum = _Tp{0};
60  auto _Ssum = _Tp{0};
61  auto _Csum = __ax;
62  auto __sign = _Tp{1};
63  auto __fact = _S_pi_2 * __ax * __ax;
64  auto __odd = true;
65  auto __term = __ax;
66  auto __n = 3;
67  auto __k = 0;
68  for (__k = 1; __k <= _S_max_iter; ++__k)
69  {
70  __term *= __fact / __k;
71  __sum += __sign * __term / __n;
72  _Tp __test = std::abs(__sum) * _S_eps;
73  if (__odd)
74  {
75  __sign = -__sign;
76  _Ssum = __sum;
77  __sum = _Csum;
78  }
79  else
80  {
81  _Csum = __sum;
82  __sum = _Ssum;
83  }
84 
85  if (__term < __test)
86  break;
87 
88  __odd = ! __odd;
89 
90  __n += 2;
91  }
92  if (__k > _S_max_iter)
93  std::__throw_runtime_error(__N("__fresnel_series: "
94  "series evaluation failed"));
95 
96  _Cf = _Csum;
97  _Sf = _Ssum;
98 
99  return;
100  }

103  /**
104  * @brief This function computes the Fresnel cosine and sine integrals
105  * by continued fractions for positive argument.
106  */
107  template <typename _Tp>
108  void
109  __fresnel_cont_frac(const _Tp __ax, _Tp & _Cf, _Tp & _Sf)
110  {
111  const auto _S_max_iter = 100;
112  const auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon(__ax);
113  const auto _S_fp_min = __gnu_cxx::__min(__ax);
114  const auto _S_pi = __gnu_cxx::__const_pi(__ax);
115 
116  // Evaluate S and C by Lentz's complex continued fraction method.
117  const auto __pix2 = _S_pi * __ax * __ax;
118  std::complex<_Tp> __b(_Tp{1}, -__pix2);
119  std::complex<_Tp> __cc(_Tp{1} / _S_fp_min, _Tp{0});
120  auto __h = _Tp{1} / __b;
121  auto __d = __h;
122  auto __n = -1;
123  auto __k = 0;
124  for (__k = 2; __k <= _S_max_iter; ++__k)
125  {
126  __n += 2;
127  const auto __a = -_Tp(__n * (__n + 1));
128  __b += _Tp{4};
129  __d = _Tp{1} / (__a * __d + __b);
130  __cc = __b + __a / __cc;
131  const auto __del = __cc * __d;
132  __h *= __del;
133  if (std::abs(__del.real() - _Tp{1})
134  + std::abs(__del.imag()) < _S_eps)
135  break;
136  }
137  if (__k > _S_max_iter)
138  std::__throw_runtime_error(__N("__fresnel_cont_frac: "
139  "continued fraction evaluation failed"));
140 
141  __h *= std::complex<_Tp>(__ax, -__ax);
142  auto __phase = std::polar(_Tp{1}, __pix2/_Tp{2});
143  auto __cs = std::complex<_Tp>(_Tp{0.5L}, _Tp{0.5L})
144  * (_Tp{1} - __phase * __h);
145  _Cf = __cs.real();
146  _Sf = __cs.imag();
147 
148  return;
149  }

152  /**
153  * @brief Return the Fresnel cosine and sine integrals
154  * as a complex number $f[ C(x) + iS(x) $f].
155  *
156  * The Fresnel cosine integral is defined by:
157  * @f[
158  * C(x) = \int_0^x \cos(\frac{\pi}{2}t^2) dt
159  * @f]
160  *
161  * The Fresnel sine integral is defined by:
162  * @f[
163  * S(x) = \int_0^x \sin(\frac{\pi}{2}t^2) dt
164  * @f]
165  *
166  * @param __x The argument
167  */
168  template <typename _Tp>
169  std::complex<_Tp>
170  __fresnel(const _Tp __x)
171  {
172  const auto _S_fp_min = __gnu_cxx::__min(__x);
173  const auto _S_x_min = _Tp{1.5L};
174  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__x);
175  if (__isnan(__x))
176  return std::complex<_Tp>{_S_NaN, _S_NaN};
177 
178  auto _Cf = _Tp{0};
179  auto _Sf = _Tp{0};
180 
181  const _Tp __ax = std::abs(__x);
182  if (__ax < std::sqrt(_S_fp_min))
183  {
184  _Cf = __ax;
185  _Sf = _Tp{0};
186  }
187  else if (__ax < _S_x_min)
188  __fresnel_series(__ax, _Cf, _Sf);
189  else
190  __fresnel_cont_frac(__ax, _Cf, _Sf);
191 
192  if (__x < _Tp{0})
193  {
194  _Cf = -_Cf;
195  _Sf = -_Sf;
196  }
197 
198  return std::complex<_Tp>(_Cf, _Sf);
199  }
