42  /**
43  * @brief Compute the Dawson integral using the series expansion.
44  */
45  template<typename _Tp>
46  _Tp
47  __dawson_series(_Tp __x)
48  {
49  auto __x2 = __x * __x;
50  _Tp __sum(1);
51  auto __k = 1;
52  _Tp __term(1);
53  while (true)
54  {
55  __term *= -(_Tp{2} / _Tp(2 * __k + 1)) * __x2;
56  __sum += __term;
57  ++__k;
58  if (std::abs(__term) < __gnu_cxx::__epsilon<_Tp>())
59  break;
60  }
61  return __x * __sum;
62  }

217  /**
218  * @brief Return the Dawson integral, @f$ F(x) @f$, for real argument @c x.
219  *
220  * The Dawson integral is defined by:
221  * @f[
222  * F(x) = e^{-x^2} \int_0^x e^{y^2} dy
223  * @f]
224  * and it's derivative is:
225  * @f[
226  * F'(x) = 1 - 2xF(x)
227  * @f]
228  *
229  * @param __x The argument @f$ -inf < x < inf @f$.
230  */
231  template<typename _Tp>
232  _Tp
233  __dawson(_Tp __x)
234  {
235  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
236  constexpr _Tp _S_x_min{0.2L};
237 
238  if (__isnan(__x))
239  return _S_NaN;
240  else if (std::abs(__x) < _S_x_min)
241  return __dawson_series(__x);
242  else
243  return __dawson_cont_frac(__x);
244  }
