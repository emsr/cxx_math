45  * @f[
46  * sinc_a(x) = \frac{\sin(\pi x / a)}{(\pi x / a)}
47  * @f]

51  __sinc(_Tp __a, _Tp __x)
52  {

54  if (__isnan(__a) || __isnan(__x))
55  return __gnu_cxx::__quiet_NaN<_Tp>();
56  else if (std::abs(__a) < __gnu_cxx::__min<_Tp>())
57  std::__throw_domain_error(__N("__sinc: Zero normalization a"));
58  else if (std::abs(__x) == __gnu_cxx::__infinity<_Tp>())
59  return _Tp{0};

65  else
