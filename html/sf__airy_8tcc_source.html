45  /**
46  * @brief This function evaluates @f$ Ai(z) @f$ and @f$ Ai'(z) @f$ from their asymptotic
47  * expansions for @f$ |arg(z)| < 2*\pi/3 @f$. For speed, the number
48  * of terms needed to achieve about 16 decimals accuracy is tabled
49  * and determined from abs(z).
50  *
51  * Note that for speed and since this function
52  * is called by another, checks for valid arguments are not
53  * made.
54  *
55  * @see Digital Library of Mathematical Functions Sec. 9.7 Asymptotic Expansions
56  * http://dlmf.nist.gov/9.7
57  *
58  * @param[in] __z Complex input variable set equal to the value at which
59  * @f$ Ai(z) @f$ and @f$ Bi(z) @f$ and their derivative are evaluated.
60  * This function assumes @f$ |z| > 15 @f$ and @f$ |arg(z)| < 2\pi/3 @f$.
61  * @param[inout] _Ai The value computed for @f$ Ai(z) @f$.
62  * @param[inout] _Aip The value computed for @f$ Ai'(z) @f$.
63  * @param[in] __sign The sign of the series terms amd exponent.
64  * The default (-1) gives the Airy Ai functions
65  * for @f$ |arg(z)| < \pi @f$.
66  * The value +1 gives the Airy Bi functions
67  * for @f$ |arg(z)| < \pi/3 @f$.

255  // Set up working value of z.
256  __z = -__z;
257  // Compute zeta and z^(1/4).
258  auto __z1d4 = std::sqrt(__z);
259  auto __zeta = __z * __z1d4;

261  __z1d4 = std::sqrt(__z1d4);

263  // Compute sine and cosine factors in the expansions.
264  auto __zetaarg = __zeta + _S_pid4;
265  auto __sinzeta = std::sin(__zetaarg);
266  auto __coszeta = std::cos(__zetaarg);
267 
268  // Determine number of terms to use.
269  auto __nterm = _S_nterms[std::min(_S_numnterms - 1,
270  (int(std::abs(__z)) - 10) / 5)];
271  // Initialize for modified Horner's rule evaluation of sums
272  // it is assumed that at least three terms are used.
273  __z = std::pow(_S_zone / __z, _Tp(3));

286  auto __bec = _S_u_cos[__index];
287  auto __beprs = _S_v_sin[__index];
288  auto __beprc = _S_v_cos[__index];

290

292  for (int __k = __index; __k < _S_ncoeffs; ++__k)
293  {
294  auto __term = __s * __als;
295  __als = __bes + __r * __als;
296  __bes = _S_u_sin[__k] - __term;
297  __term = __s * __alc;
298  __alc = __bec + __r * __alc;
299  __bec = _S_u_cos[__k] - __term;
300  __term = __s * __alprs;
301  __alprs = __beprs + __r * __alprs;
302  __beprs = _S_v_sin[__k] - __term;
303  __term = __s * __alprc;
304  __alprc = __beprc + __r * __alprc;
305  __beprc = _S_v_cos[__k] - __term;
306  }
307 
308  // Complete evaluation of the Airy functions.
309  __zeta = _S_zone / __zeta;
310  _Ai = __sinzeta * __als * __z + __bes
311  - __zeta * __coszeta * __alc * __z + __bec;
312  _Ai *= _S_pimh / __z1d4;
313  _Aip = __coszeta * __alprc * __z + __beprc
314  + __zeta * __sinzeta * __alprs * __z + __beprs;
315  _Aip *= -_S_pimh * __z1d4;
316 
317  return;

321  /**
322  * Compute the modified Bessel functions of the first kind of
323  * orders +-1/3 and +-2/3 needed to compute the Airy functions
324  * and their derivatives from their representation in terms of the
325  * modified Bessel functions. This function is only used for z
326  * less than two in modulus and in the closed right half plane.
327  * This stems from the fact that the values of the modified
328  * Bessel functions occuring in the representations of the Airy
329  * functions and their derivatives are almost equal for z large
330  * in the right half plane. This means that loss of significance
331  * occurs if these representations are used for z to large in
332  * magnitude. This algorithm is also not used for z too small,
333  * since a low order rational approximation can be used instead.
334  *
335  * This routine is an implementation of a modified version of
336  * Miller's backward recurrence algorithm for computation by
337  * from the recurrence relation
338  * @f[
339  * I_{\nu-1} = (2\nu/z)I_\nu + I_{\nu+1}
340  * @f]
341  * satisfied by the modified Bessel functions of the first kind.
342  * the normalization relationship used is
343  * @f[
344  * \frac{z/2)^\nu e^z}{\Gamma(\nu+1)} = I_\nu(z)
345  * + 2\sum_{k=1}^{\infty}\frac{(k+\nu)\Gamma(2\nu+k)}{k!\Gamma(1+2\nu)}
346  * I_{\nu+k}(z).
347  * @f]
348  *
349  * This modification of the algorithm is given in part in
350  *
351  * Olver, F. W. J. and Sookne, D. J., Note on Backward Recurrence
352  * Algorithms, Math. of Comp., Vol. 26, no. 120, Oct. 1972.
353  *
354  * And further elaborated for the Bessel functions in
355  *
356  * Sookne, D. J., Bessel Functions I and J of Complex Argument
357  * and Integer Order, J. Res. NBS - Series B, Vol 77B, Nos.
358  * 3 & 4, July-December, 1973.
359  *
360  * Insight was also gained from
361  *
362  * Cody, W. J., Preliminary Report on Software for the Modified
363  * Bessel Functions of the First Kind, Argonne National
364  * Laboratory, Applied Mathematics Division, Tech. Memo.
365  * No. 357, August, 1980.
366  *
367  * Cody implements the algorithm of Sookne for fractional order
368  * and nonnegative real argument. Like Cody, we do not change
369  * the convergence testing mechanism in any substantial way.
370  * However, we do trim the overhead by making the additional
371  * assumption that performing the convergence test for the
372  * functions of order 2/3 will suffice for order 1/3 as well.
373  * This assumption has not been established by rigourous
374  * analysis at this time. For speed the convergence
375  * tests are performed in the 1-norm instead of the usual Euclidean
376  * norm used in the complex plane using the inequality
377  * @f[
378  * |x| + |y| <= \sqrt(2) \sqrt(x^2 + y^2)
379  * @f]
380  *
381  * @param[in] __z The argument of the modified Bessel functions.
382  * @param[in] __eps The maximum relative error required in the results.
383  * @param[out] _Ip1d3 The value of @f$ I_(+1/3)(z) @f$.
384  * @param[out] _Im1d3 The value of @f$ I_(-1/3)(z) @f$.
385  * @param[out] _Ip2d3 The value of @f$ I_(+2/3)(z) @f$.
386  * @param[out] _Im2d3 The value of @f$ I_(-2/3)(z) @f$.
387  */
388  template<typename _Tp>
389  void
390  __airy_bessel_i(const std::complex<_Tp>& __z, _Tp __eps,
391  std::complex<_Tp>& _Ip1d3,
392  std::complex<_Tp>& _Im1d3,
393  std::complex<_Tp>& _Ip2d3,
394  std::complex<_Tp>& _Im2d3)
395  {
396  using __cmplx = std::complex<_Tp>;
397 
398  constexpr __cmplx _S_zero{0}, _S_zone{1};
399  constexpr _Tp
400  _S_1d3 {_Tp{1} / _Tp{3}}, _S_2d3 {_Tp{2} / _Tp{3}},
401  _S_4d3 {_Tp{4} / _Tp{3}}, _S_5d3 {_Tp{5} / _Tp{3}},
402  _S_8d3 {_Tp{8} / _Tp{3}}, _S_10d3 {_Tp{10} / _Tp{3}},
403  _S_14d3 {_Tp{14} / _Tp{3}}, _S_16d3 {_Tp{16} / _Tp{3}},
404  _S_gamma4d3{8.929795115692492112185643136582258813769e-1L},
405  _S_gamma5d3{9.027452929509336112968586854363425236809e-1L},
406  _S_sqrt2 = __gnu_cxx::__math_constants<_Tp>::__root_2;
407 
408  // Compute 1/z for use in recurrence for speed and abs(z).
409  auto __1dz = __safe_div(_Tp{1}, __z);
410 
411  // Initialize for forward recursion based on order 2/3.
412  int __n = 0;
413  auto __d2n = _Tp(2 * __n) + _S_4d3;
414  auto __plast2 = _S_zone;
415  auto __p2 = __d2n * __1dz;
416 
417  auto __weak_test = 20 * _S_sqrt2 / __eps;
418  bool __converged = false;
419 
420  while (true)
421  {
422  do
423  {
424  // Update n dependent quantities.
425  ++__n;
426  __d2n += 2;
427  // Interchange values.
428  auto __pold2 = __plast2;
429  __plast2 = __p2;
430  // Recur forward one step.
431  __p2 = __1dz * __d2n * __plast2 + __pold2;
432  }
433  while (__l1_norm(__p2) < __weak_test);
434 
435  // If strong convergence, then weak and strong convergence.
436  if (__converged)
437  break;
438 
439  // Calculate strong convergence test.
440  // See the Olver and Sookne papers cited for details.
441  auto __rep2 = std::real(__p2);
442  auto __imp2 = std::imag(__p2);
443  auto __replast2 = std::real(__plast2);
444  auto __implast2 = std::imag(__plast2);
445  // Compute scale factor to avoid possible overflow.
446  auto __lamn = __linf_norm(__p2);
447  // Compute the kappa_n of strong convergence lemma.
448  auto __kapn = std::sqrt(((__rep2 / __lamn) * (__rep2 / __lamn)
449  + (__imp2 / __lamn) * (__imp2 / __lamn))
450  / ((__replast2 / __lamn) * (__replast2 / __lamn)
451  + (__implast2 / __lamn) * (__implast2 / __lamn)));
452  // Compute quantity needed for lambda_n of strong convergence lemma
453  __lamn = _Tp(__n + 1) / std::abs(__z);
454  // Determine appropriate value for rho_n of lemma
455  if (__kapn + 1 / __kapn > 2 * __lamn)
456  __kapn = __lamn + std::sqrt(__lamn * __lamn - 1);
457  // Compute test value - sqrt(2) multiple already included.
458  __weak_test *= std::sqrt(__kapn - 1 / __kapn);
459  // Set strong convergence test flag.
460  __converged = true;
461  }
462 
463  // Prepare for backward recurrence for both orders 1/3 and 2/3.
464  auto __rn = _Tp(__n);
465  ++__n;
466  __d2n = _Tp(2 * __n);
467  auto __plast1 = _S_zero;
468  __plast2 = _S_zero;
469  // Carefully compute 1/p2 to avoid overflow in complex divide.
470  auto __p1 = __safe_div(_Tp{1}, __p2);
471  __p2 = __p1;
472  // Set up n dependent parameters used in normalization sum.
473  auto __rnpn1 = __rn + _S_1d3;
474  auto __rnpn2 = __rn + _S_2d3;
475  auto __rnp2n1 = (__rn - _Tp(1)) + _S_2d3;
476  auto __rnp2n2 = (__rn - _Tp(1)) + _S_4d3;
477  // Initialize normalization sum
478  auto __fact1 = __rnpn1 * __rnp2n1 / __rn;
479  auto __sum1 = __fact1 * __p1;
480  auto __fact2 = __rnpn2 * __rnp2n2 / __rn;
481  auto __sum2 = __fact2 * __p2;
482  // Set ending loop index to correspond to k=1 term of the
483  // normalization relationship.
484  auto __nend = __n - 3;
485 
486  // If backward recurrence loop will be nontrivial...
487  if (__nend > 0)
488  {
489  // ...loop until backward recursion to k=1 term of normalization.
490  for (int __l = 1; __l <= __nend; ++__l)
491  {
492  // Update n dependent quantities.
493  --__n;
494  __d2n -= 2;
495  __fact1 = __d2n + _S_2d3;
496  __fact2 = __d2n + _S_4d3;
497  // Interchanges for order 1/3 recurrence.
498  auto __pold1 = __plast1;
499  __plast1 = __p1;
500  // Recur back one step for order 1/3.
501  __p1 = __1dz * __fact1 * __plast1 + __pold1;
502  // Interchanges for order 2/3 recurrence
503  auto __pold2 = __plast2;
504  __plast2 = __p2;
505  // Recur back one step for order 2/3.
506  __p2 = __1dz * __fact2 * __plast2 + __pold2;
507  // Update quantities for computing normalization sums.
508  __rn -= _Tp(1);
509  __rnpn1 = __rn + _S_1d3;
510  __rnp2n1 = __rn - _S_1d3;
511  __rnpn2 = __rn + _S_2d3;
512  __rnp2n2 = __rnpn1;
513  __fact1 = __rnp2n1 / __rn;
514  __fact2 = __rnp2n2 / __rn;
515  // Update normalization sums.
516  __sum1 += __rnpn1 * __p1;
517  __sum1 *= __fact1;
518  __sum2 += __rnpn2 * __p2;
519  __sum2 *= __fact2;
520  }
521  }
522 
523  // Perform last two recurrence steps for order 1/3
524  auto __pold1 = __plast1;
525  __plast1 = __p1;
526  __p1 = _S_14d3 * __plast1 * __1dz + __pold1;
527  __sum1 += _S_4d3 * __p1;
528  __pold1 = __plast1;
529  __plast1 = __p1;
530  __p1 = _S_8d3 * __plast1 * __1dz + __pold1;
531  __sum1 = _Tp(2) * __sum1 + __p1;
532 
533  // Compute scale factor and scale results for order 1/3 case
534  auto __zd2pow = std::pow(_Tp(0.5L) * __z, -_S_1d3);
535  __pold1 = __zd2pow * std::exp(-__z);
536  __sum1 *= _S_gamma4d3 * __pold1;
537  __plast1 /= __sum1;
538  _Ip1d3 = __p1 / __sum1;
539 
540  // Perform last two recurrence steps for order 2/3
541  auto __pold2 = __plast2;
542  __plast2 = __p2;
543  __p2 = _S_16d3 * __plast2 * __1dz + __pold2;
544  __sum2 += _S_5d3 * __p2;
545  __pold2 = __plast2;
546  __plast2 = __p2;
547  __p2 = _S_10d3 * __plast2 * __1dz + __pold2;
548  __sum2 = _Tp(2) * __sum2 + __p2;
549

551  __sum2 *= _S_gamma5d3 * __zd2pow * __pold1;
552  __plast2 /= __sum2;
553  _Ip2d3 = __p2 / __sum2;
554 
555  // Recur back one step from order +1/3 to get order -2/3
556  _Im2d3 = _S_2d3 * _Ip1d3 * __1dz + __plast1;
557 
558  // Recur back one step from order +2/3 to get order -1/3
559  _Im1d3 = _S_4d3 * _Ip2d3 * __1dz + __plast2;
560 
561  return;

565  /**
566  * @brief Compute approximations to the modified Bessel functions
567  * of the second kind of orders 1/3 and 2/3 for moderate arguments.
568  *
569  * This routine computes
570  * @f[
571  * E_\nu(z) = \exp{z} \sqrt{2 z/\pi} K_\nu(z), for \nu = 1/3 and \nu = 2/3
572  * @f]
573  * using a rational approximation given in
574  *
575  * Luke, Y. L., Mathematical functions and their approximations,
576  * Academic Press, pp 366-367, 1975.
577  *
578  * Though the approximation converges in @f$ |\arg(z)| <= pi @f$,
579  * The convergence weakens as abs(arg(z)) increases. Also, in
580  * the case of real order between 0 and 1, convergence weakens
581  * as the order approaches 1. For these reasons, we only use
582  * this code for @f$ |\arg(z)| <= 3pi/4 @f$ and the convergence test
583  * is performed only for order 2/3.
584  *
585  * The coding of this function is also influenced by the fact
586  * that it will only be used for about @f$ 2 <= |z| <= 15 @f$.
587  * Hence, certain considerations of overflow, underflow, and
588  * loss of significance are unimportant for our purpose.
589  *
590  * @param[in] __z The value for which the quantity E_nu is to be computed.
591  * it is recommended that abs(z) not be too small and that
592  * @f$ |\arg(z)| <= 3pi/4 @f$.
593  * @param[in] __eps The maximum relative error allowable in the computed
594  * results. The relative error test is based on the
595  * comparison of successive iterates.
596  * @param[out] _Kp1d3 The value computed for @f$ E_{+1/3}(z) @f$.
597  * @param[out] _Kp2d3 The value computed for @f$ E_{+2/3}(z) @f$.
598  *
599  * @note In the worst case, say, @f$ z=2 @f$ and @f$ \arg(z) = 3pi/4 @f$,
600  * 20 iterations should give 7 or 8 decimals of accuracy.
601  */
602  template<typename _Tp>
603  void
604  __airy_bessel_k(const std::complex<_Tp>& __z, _Tp __eps,
605  std::complex<_Tp>& _Kp1d3,
606  std::complex<_Tp>& _Kp2d3)
607  {
608  using __cmplx = std::complex<_Tp>;
609 
610  constexpr _Tp _S_an1i{ _Tp{437} / _Tp{9}},
611  _S_an2i{ _Tp{425} / _Tp{9}},
612  _S_p12i{ _Tp{283} / _Tp{9}},
613  _S_p22i{ _Tp{295} / _Tp{9}},
614  _S_p13i{- _Tp{25} / _Tp{27}},
615  _S_p23i{ _Tp{35} / _Tp{27}},
616  _S_p11i{-_Tp{2135} / _Tp{27}},
617  _S_p21i{-_Tp{2195} / _Tp{27}};
618 
619  constexpr std::complex<_Tp> _S_zone{1};
620  constexpr int _S_iter_max = 100;

622  constexpr _Tp
623  _S_f[8]
624  { 144, 77, 62208, 95472, 17017, 65, 90288, 13585 };

627  _S_phi[6]
628  { 67, 91152, 12697, 79, 96336, 19633 };

632  auto __f10 = _S_zone;
633  auto __f20 = _S_zone;
634  auto __f11 = _S_zone + _S_f[0] * __z / _S_f[1];
635  auto __f12 = __z * (_S_f[2] * __z + _S_f[3]);
636  __f12 = _S_zone + __f12 / _S_f[4];
637  auto __f21 = _S_zone + _S_f[0] * __z / _S_f[5];
638  auto __f22 = __z * (_S_f[2] * __z + _S_f[6]);
639  __f22 = _S_zone + __f22 / _S_f[7];
640 
641  auto __phi10 = _S_zone;
642  auto __phi20 = _S_zone;
643  auto __phi11 = __cmplx((_S_f[0] * std::real(__z) + _S_phi[0]) / _S_f[1],
644  std::imag(__f11));
645  auto __phi12 = __z * (_S_f[2] * __z + _S_phi[1]);
646  __phi12 = (__phi12 + _S_phi[2]) / _S_f[4];
647  auto __phi21 = __cmplx((_S_f[0] * std::real(__z) + _S_phi[3]) / _S_f[5],
648  std::imag(__f21));
649  auto __phi22 = __z * (_S_f[2] * __z + _S_phi[4]);
650  __phi22 = (__phi22 + _S_phi[5]) / _S_f[7];
651 
652  // Initialize for recursion
653  auto __ratold = __phi22 / __f22;
654  auto __rat1 = __phi12 / __f12;
655  auto __delt = _Tp{32};
656  auto __an1 = _S_an1i;
657  auto __an2 = _S_an2i;
658  auto __p11 = _S_p11i;
659  auto __p12 = _S_p12i;
660  auto __p13 = _S_p13i;
661  auto __p21 = _S_p21i;
662  auto __p22 = _S_p22i;
663  auto __p23 = _S_p23i;
664  auto __eta = _Tp{24};
665  auto __gamm = _Tp{3};
666  auto __gam = _Tp{5};
667  auto __q = _Tp{16} * __gam;
668 
669  // Loop until maximum iterations used or convergence.
670  for (int __i = 1; __i < _S_iter_max; ++__i)
671  {
672  // Evaluate next term in recurrence for order 1/3 polynomials.
673  auto __qz = __q * __z;
674  auto __fact1 = __qz - __p11;
675  auto __fact2 = __qz - __p12;
676  auto __f13 = __fact1 * __f12
677  + __fact2 * __f11
678  - __p13 * __f10;
679  __f13 /= __an1;
680  auto __phi13 = __fact1 * __phi12
681  + __fact2 * __phi11
682  - __p13 * __phi10;
683  __phi13 /= __an1;
684 
685  // Evaluate next term in recurrence for order 2/3 polynomials.
686  __fact1 = __qz - __p21;
687  __fact2 = __qz - __p22;
688  auto __f23 = __fact1 * __f22
689  + __fact2 * __f21
690  - __p23 * __f20;
691  __f23 /= __an2;
692  auto __phi23 = __fact1 * __phi22
693  + __fact2 * __phi21
694  - __p23 * __phi20;
695  __phi23 /= __an2;
696 
697  // Check for convergence.
698  auto __ratnew = __phi23 / __f23;
699  __rat1 = __phi13 / __f13;
700 
701  if (std::abs(__ratnew - __ratold) < __eps * std::abs(__ratnew))
702  {
703  // Convergence.
704  _Kp2d3 = __ratnew;
705  _Kp1d3 = __phi13 / __f13;
706  return;
707  }
708 
709  // Prepare for next iteration.
710  __ratold = __ratnew;
711  __f20 = __f21;
712  __f21 = __f22;
713  __f22 = __f23;
714  __phi20 = __phi21;
715  __phi21 = __phi22;
716  __phi22 = __phi23;
717  __f10 = __f11;
718  __f11 = __f12;
719  __f12 = __f13;
720  __phi10 = __phi11;
721  __phi11 = __phi12;
722  __phi12 = __phi13;
723  __delt = __delt + _Tp{24};
724  __p12 = __p12 + __delt;
725  __p22 = __p22 + __delt;
726  __eta += _Tp{8};
727  __an1 += __eta;
728  __an2 += __eta;
729  auto __anm1 = __an1 - __delt - _Tp{16};
730  auto __anm2 = __an2 - __delt - _Tp{16};
731  __gamm = __gam;
732  __gam += _Tp{2};
733  __p23 = -__gam / __gamm;
734  __p13 = __p23 * __anm1;
735  __p23 = __p23 * __anm2;
736  __p11 = -__an1 - __p12 - __p13;
737  __p21 = -__an2 - __p22 - __p23;
738  __q = _Tp{16} * __gam;
739  }
740 
741  std::__throw_runtime_error(__N("__airy_bessel_k:"
742  " maximum iterations exceeded"));
743 
744  return;
745  }

748  /**
749  * @brief This function computes rational approximations
750  * to the hypergeometric functions related to the modified Bessel
751  * functions of orders @f$ \nu = +-1/3 @f$ and @f$ \nu +- 2/3 @f$. That is,
752  * @f$ A(z)/B(z) @f$, Where @f$ A(z) @f$ and @f$ B(z) @f$ are
753  * cubic polynomials with real coefficients, approximates
754  * @f[
755  * \frac{\Gamma(\nu+1)}{(z/2)^nu}I_\nu(z) = _0F_1 (;\nu+1;z^2/4),
756  * @f]
757  * where the function on the right is a
758  * confluent hypergeometric limit function. For @f$ |z| <= 1/4 @f$ and
759  * @f$ |arg(z)| <= pi/2 @f$, the approximations are accurate to
760  * about 16 decimals.
761  *
762  * For further details including the four term recurrence
763  * relation satisfied by the numerator and denominator poly-
764  * nomials in the higher order approximants, see
765  *
766  * Luke, Y.L., Mathematical Functions and their Approximations,
767  * Academic Press, pp 361-363, 1975.
768  *
769  * An asymptotic expression for the error is given as well as
770  * other useful expressions in the event one wants to extend
771  * this function to incorporate higher order approximants.
772  *
773  * Note also that for speed and since this function is called by another,
774  * checks that are not absolutely necessary are not made.
775  *
776  * @param[in] __z The argument at which the hypergeometric given
777  * above is to be evaluated. Since the approximation
778  * is of fixed order, @f$ |z| @f$ must be small to ensure
779  * sufficient accuracy of the computed results.
780  * @param[out] _Ai The Airy function @f$ Ai(z) @f$.
781  * @param[out] _Aip The Airy function derivative @f$ Ai'(z) @f$.
782  * @param[out] _Bi The Airy function @f$ Bi(z) @f$.
783  * @param[out] _Bip The Airy function derivative @f$ Bi'(z) @f$.
784  */
785  template<typename _Tp>
786  void
787  __airy_hyperg_rational(const std::complex<_Tp>& __z,
788  std::complex<_Tp>& _Ai,
789  std::complex<_Tp>& _Aip,
790  std::complex<_Tp>& _Bi,
791  std::complex<_Tp>& _Bip)
792  {
793  using __cmplx = std::complex<_Tp>;
794 
795  constexpr __cmplx _S_zone{1};
796 
797  constexpr _Tp _S_ap1d3[4]{ 81, 32400, 2585520, 37920960};
798  constexpr _Tp _S_bp1d3[4]{ -35, 5040, -574560, 37920960};
799  constexpr _Tp _S_am1d3[4]{ 81, 22680, 1156680, 7711200};
800  constexpr _Tp _S_bm1d3[4]{ -10, 1260, -128520, 7711200};
801  constexpr _Tp _S_ap2d3[4]{ 162, 75735, 7270560, 139352400};
802  constexpr _Tp _S_bp2d3[4]{-110, 16830, -2019600, 139352400};
803  constexpr _Tp _S_am2d3[4]{ 162, 36855, 1415232, 4481568};
804  constexpr _Tp _S_bm2d3[4]{ -7, 819, -78624, 4481568};
805  constexpr _Tp _S_Ai0{3.550280538878172392600631860041831763980e-1L};
806  constexpr _Tp _S_Aip0{-2.588194037928067984051835601892039634793e-1L};
807  constexpr _Tp _S_Bi0{6.149266274460007351509223690936135535960e-1L};
808  constexpr _Tp _S_Bip0{4.482883573538263579148237103988283908668e-1L};
809 
810  // Check to see if z^3 will underflow and act accordingly.
811  auto __zzz = __z * __z * __z;
812 
813  // The confluent hypergeometric limit functions related to
814  // the modified Bessel functions of order +1/3, -1/3, +2/3, and -2/3
815  // respectively.
816  std::complex<_Tp> _Fp1d3, _Fm1d3, _Fp2d3, _Fm2d3;
817  if (std::abs(__zzz) < _Tp{10} * __gnu_cxx::__min<_Tp>())
818  {
819  _Fp1d3 = _S_zone;
820  _Fm1d3 = _S_zone;
821  _Fp2d3 = _S_zone;
822  _Fm2d3 = _S_zone;
823  }
824  else
825  {
826  auto __r = _Tp{2} * std::real(__zzz);
827  auto __s = std::norm(__zzz);
828 
829  // The following polynomial evaluations are done using
830  // a modified of Horner's rule which exploits the fact that
831  // the polynomial coefficients are all real.
832  // The algorithm is discussed in detail in:
833  // Knuth, D. E., The Art of Computer Programming: Seminumerical
834  // Algorithms (Vol. 2) Third Ed., Addison-Wesley, pp 486-488, 1998.
835  //
836  // If n is the degree of the polynomial, n - 3 multiplies are
837  // saved and 4 * n - 6 additions are saved.
838  auto __horner
839  {
840  [__r, __s, __zzz](const auto (&_S_c)[4])
841  {
842  auto __aa = _S_c[0];
843  auto __t = __s * __aa;
844  __aa = _S_c[1] + __r * __aa;
845  auto __bb = _S_c[2] - __t;
846  __t = __s * __aa;
847  __aa = __bb + __r * __aa;
848  __bb = _S_c[3] - __t;
849  return __aa * __zzz + __bb;
850  }
851  };
852 
853  _Fp1d3 = __horner(_S_ap1d3) / __horner(_S_bp1d3);
854  _Fm1d3 = __horner(_S_am1d3) / __horner(_S_bm1d3);
855  _Fp2d3 = __horner(_S_ap2d3) / __horner(_S_bp2d3);
856  _Fm2d3 = __horner(_S_am2d3) / __horner(_S_bm2d3);
857  }
858 
859  _Ai = _S_Ai0 * _Fm1d3 + _S_Aip0 * __z * _Fp1d3;
860  _Aip = _S_Ai0 * __z * __z * _Fp2d3 / _Tp{2} + _S_Aip0 * _Fm2d3;
861  _Bi = _S_Bi0 * _Fm1d3 + _S_Bip0 * __z * _Fp1d3;
862  _Bip = _S_Bi0 * __z * __z * _Fp2d3 / _Tp{2} + _S_Bip0 * _Fm2d3;
863 
864  return;
865  }

868  /**
869  * @brief This function computes the Airy function @f$ Ai(z) @f$
870  * and its first derivative in the complex z-plane.
871  *
872  * The algorithm used exploits numerous representations of the
873  * Airy function and its derivative.
874  * The representations are recorded here for reference:
875  *
876  * @f[
877  * (1a) Ai(z) = \frac{\sqrt{z}}{3}(I_{-1/3}(\zeta) - I_{1/3}(\zeta))
878  * @f]
879  * @f[
880  * (1b) Bi(z) = \sqrt{\frac{z}{3}}(I_{-1/3}(\zeta) + I_{1/3}(\zeta))
881  * @f]
882  *
883  * @f[
884  * (2) Ai(z) = \frac{\sqrt{z/3}}{\pi} K_{1/3}(\zeta)
885  *
886  * = \frac{2^{2/3}3^{-5/6}}{\sqrt(\pi)}
887  * z \exp(-\zeta) U(5/6; 5/3; 2 \zeta)
888  * @f]
889  *
890  * @f[
891  * (3a) Ai(-z) = \frac{\sqrt{z}}{3}(J_{-1/3}(\zeta) + J_{1/3}(\zeta))
892  * @f]
893  * @f[
894  * (3b) Bi(-z) = \sqrt{\frac{z}{3}}(J_{-1/3}(\zeta) - J_{1/3}(\zeta))
895  * @f]
896  *
897  * @f[
898  * (4a) Ai'(z) = \frac{z}{3}(I_{2/3}(\zeta) - I_{-2/3}(\zeta))
899  * @f]
900  * @f[
901  * (4b) Bi'(z) = \frac{z}{\sqrt{3}}(I_{-2/3}(\zeta) + I_{2/3}(\zeta))
902  * @f]
903  *
904  * @f[
905  * (5a) Ai'(z) = -\frac{z}{\pi\sqrt(3)} K_(2/3)(zeta)
906  *
907  * = -\frac{4^{2/3}3^{-7/6}}{\sqrt(\pi)}
908  * z^2 \exp(-\zeta) U(7/6; 7/3; 2 \zeta)
909  * @f]
910  *
911  * @f[
912  * (6a) Ai'(-z) = \frac{z}{3}(J_{2/3}(\zeta) - J_{-2/3}(\zeta)) ,
913  * @f]
914  * @f[
915  * (6b) Bi'(-z) = \frac{z}{\sqrt{3}}(J_{-2/3}(\zeta) + J_{2/3}(\zeta)) ,
916  * @f]
917  * Where @f$ \zeta = - \frac{2}{3}z^{3/2} @f$ and @f$ U(a;b;z) @f$ is
918  * the confluent hypergeometric function defined in
919  *
920  * @see Stegun, I. A. and Abramowitz, M., Handbook of Mathematical Functions,
921  * Natl. Bureau of Standards, AMS 55, pp 504-515, 1964.
922  *
923  * The asymptotic expansions derivable from these representations and
924  * Hankel's asymptotic expansions for the Bessel functions are used for
925  * large modulus of z. The implementation has taken advantage of the
926  * error bounds given in
927  *
928  * @see Olver, F. W. J., Error Bounds for Asymptotic Expansions,
929  * with an Application to Cylinder Functions of Large Argument,
930  * in Asymptotic Solutions of Differential Equations (Wilcox, Ed.),
931  * Wiley and Sons, pp 163-183, 1964
932  *
933  * @see Olver, F. W. J., Asymptotics and Special Functions, Academic Press,
934  * pp 266-268, 1974.
935  *
936  * For small modulus of z, a rational approximation is used. This
937  * approximant is derived from
938  *
939  * Luke, Y. L., Mathematical Functions and their Approximations,
940  * Academic Press, pp 361-363, 1975.
941  *
942  * The identities given below are for Bessel functions of the first
943  * kind in terms of modified Bessel functions of the first kind are
944  * also used with the rational approximant.
945  *
946  * For moderate modulus of z, three techniques are used. Two use
947  * a backward recursion algorithm with (1), (3), (4), and (6). The
948  * third uses the confluent hypergeometric representations given by
949  * (2) and (5). The backward recursion algorithm generates values of
950  * the modified Bessel functions of the first kind of orders + or -
951  * 1/3 and + or - 2/3 for z in the right half plane. Values for
952  * the corresponding Bessel functions of the first kind are recovered
953  * via the identities
954  * @f[
955  * J_\nu(z) = exp(\nu \pi i/2) I_\nu(z exp(-\pi i/2)),
956  * 0 <= arg(z) <= \pi/2
957  * @f]
958  * and
959  * @f[
960  * J_\nu(z) = exp(-\nu \pi i/2) I_\nu(z exp(\pi i/2)),
961  * -\pi/2 <= arg(z) < 0 .
962  * @f]
963  *
964  * The particular backward recursion algorithm used is discussed in
965  *
966  * @see Olver, F. W. J, Numerical solution of second-order linear
967  * difference equations, NBS J. Res., Series B, VOL 71B,
968  * pp 111-129, 1967.
969  *
970  * @see Olver, F. W. J. and Sookne, D. J., Note on backward recurrence
971  * algorithms, Math. Comp. Vol 26, No. 120, pp 941-947,
972  * Oct. 1972
973  *
974  * @see Sookne, D. J., Bessel Functions I and J of Complex Argument and
975  * Integer Order, NBS J. Res., Series B, Vol 77B, Nos 3& 4,
976  * pp 111-113, July-December, 1973.
977  *
978  * The following paper was also useful
979  *
980  * @see Cody, W. J., Preliminary report on software for the modified
981  * Bessel functions of the first kind, Applied Mathematics
982  * Division, Argonne National Laboratory, Tech. Memo. no. 357.
983  *
984  * A backward recursion algorithm is also used to compute the
985  * confluent hypergeometric function. The recursion relations
986  * and a convergence theorem are given in
987  *
988  * @see Wimp, J., On the computation of Tricomi's psi function, Computing,
989  * Vol 13, pp 195-203, 1974.
990  *
991  * @param[in] __z The argument at which the Airy function and its derivative
992  * are computed.
993  * @param[in] __eps Relative error required. Currently, eps is used only
994  * in the backward recursion algorithms.
995  * @param[out] _Ai The value computed for Ai(z).
996  * @param[out] _Aip The value computed for Ai'(z).
997  * @param[out] _Bi The value computed for Bi(z).
998  * @param[out] _Bip The value computed for Bi'(z).
999  */
1000  template<typename _Tp>
1001  void
1002  __airy(const std::complex<_Tp>& __z, _Tp __eps,
1003  std::complex<_Tp>& _Ai,
1004  std::complex<_Tp>& _Aip,

1014  _S_empid6{_S_sqrt3 / _Tp{2}, -_Tp{0.5L}},
1015  _S_eppid3{_Tp{0.5L}, _S_sqrt3 / _Tp{2}},
1016  _S_empid3{_Tp{0.5L}, -_S_sqrt3 / _Tp{2}};
1017  static constexpr _Tp
1018  _S_2d3{_Tp{2} / _Tp{3}},
1019  _S_rsqpi{2.820947917738781434740397257803862929219e-01L}, // 1/(2sqrt(pi))

1023  auto __absz = std::abs(__z);

1025  if (__absz < _S_big)
1026  {
1027  // Moderate or small abs(z)
1028  // Check argument for right or left half plane.
1029  if (std::real(__z) >= _Tp{0})
1030  {
1031  // Argument in closed right half plane. Compute zeta as defined
1032  // in the representations in terms of Bessel functions.
