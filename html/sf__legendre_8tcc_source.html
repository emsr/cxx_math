58  /**
59  * @brief Return the Legendre polynomial by upward recursion
60  * on order @f$ l @f$.
61  *
62  * The Legendre function of order @f$ l @f$ and argument @f$ x @f$,
63  * @f$ P_l(x) @f$, is defined by:
64  * @f[
65  * P_l(x) = \frac{1}{2^l l!}\frac{d^l}{dx^l}(x^2 - 1)^{l}
66  * @f]
67  *
68  * @param l The order of the Legendre polynomial. @f$l >= 0@f$.
69  * @param x The argument of the Legendre polynomial. @f$|x| <= 1@f$.
70  */
71  template<typename _Tp>
72  _Tp
73  __poly_legendre_p(unsigned int __l, _Tp __x)
74  {
75  if ((__x < -_Tp{1}) || (__x > +_Tp{1}))
76  std::__throw_domain_error(__N("__poly_legendre_p: argument out of range"));
77  else if (__isnan(__x))
78  return __gnu_cxx::__quiet_NaN<_Tp>();
79  else if (__x == +_Tp{1})
80  return +_Tp{1};
81  else if (__x == -_Tp{1})
82  return (__l % 2 == 1 ? -_Tp{1} : +_Tp{1});
83  else
84  {
85  auto _P_lm2 = _Tp{1};
86  if (__l == 0)
87  return _P_lm2;
88 
89  auto _P_lm1 = __x;
90  if (__l == 1)
91  return _P_lm1;
92 
93  auto _P_l = _Tp{0};
94  for (unsigned int __ll = 2; __ll <= __l; ++__ll)
95  {
96  // This arrangement is supposed to be better for roundoff
97  // protection, Arfken, 2nd Ed, Eq 12.17a.
98  _P_l = _Tp{2} * __x * _P_lm1 - _P_lm2
99  - (__x * _P_lm1 - _P_lm2) / _Tp(__ll);
100  _P_lm2 = _P_lm1;
101  _P_lm1 = _P_l;
102  }
103 
104  return _P_l;
105  }
106  }

108  /**
109  * @brief Return the Legendre function of the second kind
110  * by upward recursion on order @f$ l @f$.
111  *
112  * The Legendre function of order @f$ l @f$ and argument @f$ x @f$,
113  * @f$ Q_l(x) @f$, is defined by:
114  * @f[
115  * Q_l(x) = \frac{1}{2^l l!}\frac{d^l}{dx^l}(x^2 - 1)^{l}
116  * @f]
117  *
118  * @param l The order of the Legendre polynomial. @f$l >= 0@f$.
119  * @param x The argument of the Legendre polynomial. @f$|x| <= 1@f$.
120  */
121  template<typename _Tp>
122  _Tp
123  __poly_legendre_q(unsigned int __l, _Tp __x)
124  {
125  if ((__x < -_Tp{1}) || (__x > +_Tp{1}))
126  std::__throw_domain_error(__N("__poly_legendre_q: argument out of range"));
127  else if (__isnan(__x))
128  return __gnu_cxx::__quiet_NaN<_Tp>();
129  else if (__x == +_Tp{1})
130  return +_Tp{1};
131  else if (__x == -_Tp{1})
132  return (__l % 2 == 1 ? -_Tp{1} : +_Tp{1});
133  else
134  {
135  auto _Q_lm2 = _Tp{0.5L} * std::log((_Tp{1} + __x) / (_Tp{1} - __x));
136  if (__l == 0)
137  return _Q_lm2;
138  auto _Q_lm1 = __x * _Q_lm2 - _Tp{1};
139  if (__l == 1)
140  return _Q_lm1;
141  auto _Q_l = _Tp{0};
142  for (unsigned int __ll = 2; __ll <= __l; ++__ll)
143  {
144  // This arrangement is supposed to be better for roundoff
145  // protection, Arfken, 2nd Ed, Eq 12.17a.
146  _Q_l = _Tp{2} * __x * _Q_lm1 - _Q_lm2
147  - (__x * _Q_lm1 - _Q_lm2) / _Tp(__ll);
148  _Q_lm2 = _Q_lm1;
149  _Q_lm1 = _Q_l;
150  }
151 
152  return _Q_l;
153  }

156  /**
157  * @brief Return the associated Legendre function by recursion
158  * on @f$ l @f$ and downward recursion on m.
159  *
160  * The associated Legendre function is derived from the Legendre function
161  * @f$ P_l(x) @f$ by the Rodrigues formula:
162  * @f[
163  * P_l^m(x) = (1 - x^2)^{m/2}\frac{d^m}{dx^m}P_l(x)
164  * @f]
165  *
166  * @param l The order of the associated Legendre function.
167  * @f$ l >= 0 @f$.
168  * @param m The order of the associated Legendre function.
169  * @f$ m <= l @f$.
170  * @param x The argument of the associated Legendre function.
171  * @f$ |x| <= 1 @f$.
172  */
173  template<typename _Tp>
174  _Tp
175  __assoc_legendre_p(unsigned int __l, unsigned int __m, _Tp __x)
176  {
177  if (__x < -_Tp{1} || __x > +_Tp{1})
178  std::__throw_domain_error(__N("__assoc_legendre_p: "
179  "argument out of range"));
180  else if (__m > __l)
181  std::__throw_domain_error(__N("__assoc_legendre_p: "
182  "degree out of range"));
183  else if (__isnan(__x))
184  return __gnu_cxx::__quiet_NaN<_Tp>();
185  else if (__m == 0)
186  return __poly_legendre_p(__l, __x);
187  else
188  {
189  _Tp _P_mm = _Tp{1};
190  if (__m > 0)
191  {
192  // Two square roots seem more accurate more of the time
193  // than just one.
194  _Tp __root = std::sqrt(_Tp{1} - __x) * std::sqrt(_Tp{1} + __x);
195  _Tp __fact = _Tp{1};
196  for (unsigned int __i = 1; __i <= __m; ++__i)
197  {
198  _P_mm *= -__fact * __root;
199  __fact += _Tp{2};
200  }
201  }
202  if (__l == __m)
203  return _P_mm;
204 
205  _Tp _P_mp1m = _Tp(2 * __m + 1) * __x * _P_mm;
206  if (__l == __m + 1)
207  return _P_mp1m;
208 
209  _Tp _P_lm2m = _P_mm;
210  _Tp _P_lm1m = _P_mp1m;
211  _Tp _P_lm = _Tp{0};
212  for (unsigned int __j = __m + 2; __j <= __l; ++__j)
213  {
214  _P_lm = (_Tp(2 * __j - 1) * __x * _P_lm1m
215  - _Tp(__j + __m - 1) * _P_lm2m) / _Tp(__j - __m);
216  _P_lm2m = _P_lm1m;
217  _P_lm1m = _P_lm;
218  }
219 
220  return _P_lm;
221  }
222  }

225  /**
226  * @brief Return the spherical associated Legendre function.
227  *
228  * The spherical associated Legendre function of @f$ l @f$, @f$ m @f$,
229  * and @f$ \theta @f$ is defined as @f$ Y_l^m(\theta,0) @f$ where
230  * @f[
231  * Y_l^m(\theta,\phi) = (-1)^m[\frac{(2l+1)}{4\pi}
232  * \frac{(l-m)!}{(l+m)!}]
233  * P_l^m(\cos\theta) \exp^{im\phi}
234  * @f]
235  * is the spherical harmonic function and @f$ P_l^m(x) @f$ is the
236  * associated Legendre function.
237  *
238  * This function differs from the associated Legendre function by
239  * argument (@f$x = \cos(\theta)@f$) and by a normalization factor
240  * but this factor is rather large for large @f$ l @f$ and @f$ m @f$
241  * and so this function is stable for larger differences of @f$ l @f$
242  * and @f$ m @f$.
243  *
244  * @param l The order of the spherical associated Legendre function.
245  * @f$ l >= 0 @f$.
246  * @param m The order of the spherical associated Legendre function.
247  * @f$ m <= l @f$.
248  * @param theta The radian polar angle argument
249  * of the spherical associated Legendre function.
250  */
251  template<typename _Tp>
252  _Tp
253  __sph_legendre(unsigned int __l, unsigned int __m, _Tp __theta)
254  {
255  if (__isnan(__theta))
256  return __gnu_cxx::__quiet_NaN<_Tp>();
257 
258  const auto __x = std::cos(__theta);
259 
260  if (__l < __m)
261  std::__throw_domain_error(__N("__sph_legendre: bad argument"));
262  else if (__m == 0)
263  {
264  _Tp _P_l = __poly_legendre_p(__l, __x);
265  _Tp __fact = std::sqrt(_Tp(2 * __l + 1)
266  / (_Tp{4} * __gnu_cxx::__math_constants<_Tp>::__pi));
267  _P_l *= __fact;
268  return _P_l;
269  }
270  else if (__x == _Tp{1} || __x == -_Tp{1})
271  return _Tp{0}; // m > 0 here
272  else
273  {
274  // m > 0 and |x| < 1 here
275 
276  // Starting value for recursion.
277  // Y_m^m(x) = sqrt( (2m+1)/(4pi m) gamma(m+1/2)/gamma(m) )
278  // (-1)^m (1-x^2)^(m/2) / pi^(1/4)
279  const auto __sgn = (__m % 2 == 1 ? -_Tp{1} : _Tp{1});
280  const auto _Y_mp1m_factor = __x * std::sqrt(_Tp(2 * __m + 3));
281  const auto __lncirc = std::log1p(-__x * __x);
282  // Gamma(m+1/2) / Gamma(m)
283  const auto __lnpoch = __log_gamma(_Tp(__m + 0.5L))
284  - __log_gamma(_Tp(__m));
285  const auto __lnpre_val =
286  -_Tp{0.25L} * __gnu_cxx::__math_constants<_Tp>::__ln_pi
287  + _Tp{0.5L} * (__lnpoch + __m * __lncirc);
288  _Tp __sr = std::sqrt((_Tp{2} + _Tp{1} / __m)
289  / (_Tp{4} * __gnu_cxx::__math_constants<_Tp>::__pi));
290  _Tp _Y_mm = __sgn * __sr * std::exp(__lnpre_val);
291  _Tp _Y_mp1m = _Y_mp1m_factor * _Y_mm;
292 
293  if (__l == __m)
294  {
295  return _Y_mm;
296  }
297  else if (__l == __m + 1)
298  {
299  return _Y_mp1m;
300  }
301  else
302  {
303  _Tp _Y_lm = _Tp{0};
304 
305  // Compute Y_l^m, l > m+1, upward recursion on l.
306  for ( int __ll = __m + 2; __ll <= __l; ++__ll)
307  {
308  const auto __rat1 = _Tp(__ll - __m) / _Tp(__ll + __m);
309  const auto __rat2 = _Tp(__ll - __m - 1) / _Tp(__ll + __m - 1);
310  const auto __fact1 = std::sqrt(__rat1 * _Tp(2 * __ll + 1)
311  * _Tp(2 * __ll - 1));
312  const auto __fact2 = std::sqrt(__rat1 * __rat2
313  * _Tp(2 * __ll + 1)
314  / _Tp(2 * __ll - 3));
315  _Y_lm = (__x * _Y_mp1m * __fact1
316  - _Tp(__ll + __m - 1) * _Y_mm * __fact2)
317  / _Tp(__ll - __m);
318  _Y_mm = _Y_mp1m;
319  _Y_mp1m = _Y_lm;
320  }
321 
322  return _Y_lm;
323  }
324  }
325  }

328  /**
329  * @brief Return the spherical harmonic function.
330  *
331  * The spherical harmonic function of @f$ l @f$, @f$ m @f$,
332  * and @f$ \theta @f$, @f$ \phi @f$ is defined by:
333  * @f[
334  * Y_l^m(\theta,\phi) = (-1)^m[\frac{(2l+1)}{4\pi}
335  * \frac{(l-m)!}{(l+m)!}]
336  * P_l^{|m|}(\cos\theta) \exp^{im\phi}
337  * @f]
338  *
339  * @param l The order of the spherical harmonic function.
340  * @f$ l >= 0 @f$.
341  * @param m The order of the spherical harmonic function.
342  * @f$ m <= l @f$.
343  * @param theta The radian polar angle argument
344  * of the spherical harmonic function.
345  * @param phi The radian azimuthal angle argument
346  * of the spherical harmonic function.
347  */
348  template<typename _Tp>
349  std::complex<_Tp>
350  __sph_harmonic(unsigned int __l, int __m, _Tp __theta, _Tp __phi)
351  {
352  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
353  if (__isnan(__theta) || __isnan(__phi))
354  return std::complex<_Tp>{_S_NaN, _S_NaN};
355 
356  return __sph_legendre(__l, std::abs(__m), __theta)
357  * std::polar(_Tp{1}, _Tp(__m) * __phi);
358  }
