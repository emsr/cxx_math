59  /**
60  * @brief This routine computes the asymptotic modified cylindrical
61  * Bessel and functions of order nu: @f$ I_{\nu}(x) @f$,
62  * @f$ N_{\nu}(x) @f$. Use this for @f$ x >> nu^2 + 1 @f$.
63  *
64  * References:
65  * (1) Handbook of Mathematical Functions,
66  * ed. Milton Abramowitz and Irene A. Stegun,
67  * Dover Publications,
68  * Section 9 p. 364, Equations 9.2.5-9.2.10
69  *
70  * @param __nu The order of the Bessel functions.
71  * @param __x The argument of the Bessel functions.
72  * @param _Inu The output regular modified Bessel function.
73  * @param _Knu The output irregular modified Bessel function.
74  * @param _Ipnu The output derivative of the regular
75  * modified Bessel function.
76  * @param _Kpnu The output derivative of the irregular
77  * modified Bessel function.
78  */
79  template<typename _Tp>
80  void
81  __cyl_bessel_ik_asymp(_Tp __nu, _Tp __x,
82  _Tp & _Inu, _Tp & _Knu,
83  _Tp & _Ipnu, _Tp & _Kpnu)
84  {
85  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
86  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
87  constexpr auto _S_pi_2 = __gnu_cxx::__math_constants<long double>::__pi_half;
88  const auto __2nu = _Tp{2} * __nu;
89  const auto __4nu2 = __2nu * __2nu;
90  const auto __8x = _Tp{8} * __x;
91  auto __k = 0;
92  auto __bk_xk = _Tp{1};
93  auto _Rsum = __bk_xk;
94  auto __ak_xk = _Tp{1};
95  auto _Psum = __ak_xk;
96  ++__k;
97  auto __2km1 = 1;
98  __bk_xk *= (__4nu2 + __2km1 * (__2km1 + 2)) / __8x;
99  auto _Ssum = __bk_xk;
100  __ak_xk *= (__2nu - __2km1) * (__2nu + __2km1) / __8x;
101  auto _Qsum = __ak_xk;
102  do
103  {
104  ++__k;
105  __2km1 += 2;
106  __bk_xk = (__4nu2 + __2km1 * (__2km1 + 2)) * __ak_xk / (__k * __8x);
107  _Rsum += __bk_xk;
108  __ak_xk *= (__2nu - __2km1) * (__2nu + __2km1) / (__k * __8x);
109  _Psum += __ak_xk;
110  auto __convP = std::abs(__ak_xk) < _S_eps * std::abs(_Psum);
111 
112  ++__k;
113  __2km1 += 2;
114  __bk_xk = (__4nu2 + __2km1 * (__2km1 + 2)) * __ak_xk / (__k * __8x);
115  _Ssum += __bk_xk;
116  __ak_xk *= (__2nu - __2km1) * (__2nu + __2km1) / (__k * __8x);
117  _Qsum += __ak_xk;
118  auto __convQ = std::abs(__ak_xk) < _S_eps * std::abs(_Qsum);
119 
120  if (__convP && __convQ && __k > (__nu / _Tp{2}))
121  break;
122  }
123  while (__k < _Tp{100} * __nu);
124 
125  auto __coef = std::sqrt(_Tp{1} / (_Tp{2} * _S_pi * __x));
126  _Inu = __coef * std::exp(__x) * (_Psum - _Qsum);
127  _Knu = _S_pi * __coef * std::exp(-__x) * (_Psum + _Qsum);
128  _Ipnu = __coef * std::exp(__x) * (_Rsum - _Ssum);
129  _Kpnu = -_S_pi * __coef * std::exp(-__x) * (_Rsum + _Ssum);
130 
131  return;
132  }

134  /**
135  * @brief Compute the modified Bessel functions @f$ I_\nu(x) @f$ and
136  * @f$ K_\nu(x) @f$ and their first derivatives
137  * @f$ I'_\nu(x) @f$ and @f$ K'_\nu(x) @f$ respectively.
138  * These four functions are computed together for numerical
139  * stability.
140  *
141  * @param __nu The order of the Bessel functions.
142  * @param __x The argument of the Bessel functions.
143  * @param _Inu The output regular modified Bessel function.
144  * @param _Knu The output irregular modified Bessel function.
145  * @param _Ipnu The output derivative of the regular
146  * modified Bessel function.
147  * @param _Kpnu The output derivative of the irregular
148  * modified Bessel function.
149  */
150  template<typename _Tp>
151  void
152  __cyl_bessel_ik_steed(_Tp __nu, _Tp __x,
153  _Tp & _Inu, _Tp & _Knu, _Tp & _Ipnu, _Tp & _Kpnu)
154  {
155  constexpr auto _S_inf = __gnu_cxx::__infinity<_Tp>();
156  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
157  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
158  constexpr auto _S_fp_min = _Tp{10} * _S_eps;
159  constexpr int _S_max_iter = 15000;
160  constexpr auto _S_x_min = _Tp{2};
161 
162  const int __n = std::nearbyint(__nu);
163 
164  const auto __mu = __nu - _Tp(__n);
165  const auto __mu2 = __mu * __mu;
166  const auto __xi = _Tp{1} / __x;
167  const auto __xi2 = _Tp{2} * __xi;
168  auto __h = __nu * __xi;
169  if (__h < _S_fp_min)
170  __h = _S_fp_min;
171  auto __b = __xi2 * __nu;
172  auto __d = _Tp{0};
173  auto __c = __h;
174  int __i;
175  for (__i = 1; __i <= _S_max_iter; ++__i)
176  {
177  __b += __xi2;
178  __d = _Tp{1} / (__b + __d);
179  __c = __b + _Tp{1} / __c;
180  const auto __del = __c * __d;
181  __h *= __del;
182  if (std::abs(__del - _Tp{1}) < _S_eps)
183  break;
184  }
185  if (__i > _S_max_iter)
186  {
187  // Don't throw with message "try asymptotic expansion" - Just do it!
188  __cyl_bessel_ik_asymp(__nu, __x, _Inu, _Knu, _Ipnu, _Kpnu);
189  return;
190  }
191 
192  auto _Inul = _S_fp_min;
193  auto _Ipnul = __h * _Inul;
194  auto _Inul1 = _Inul;
195  auto _Ipnu1 = _Ipnul;
196  auto __fact = __nu * __xi;
197  for (int __l = __n; __l >= 1; --__l)
198  {
199  const auto _Inutemp = __fact * _Inul + _Ipnul;
200  __fact -= __xi;
201  _Ipnul = __fact * _Inutemp + _Inul;
202  _Inul = _Inutemp;
203  }
204 
205  auto __f = _Ipnul / _Inul;
206  _Tp _Kmu, _Knu1;
207  if (__x < _S_x_min)
208  {
209  const auto __x2 = __x / _Tp{2};
210  const auto __pimu = _S_pi * __mu;
211  const auto __fact = (std::abs(__pimu) < _S_eps
212  ? _Tp{1}
213  : __pimu / std::sin(__pimu));
214  auto __d = -std::log(__x2);
215  auto __e = __mu * __d;
216  const auto __fact2 = (std::abs(__e) < _S_eps
217  ? _Tp{1}
218  : std::sinh(__e) / __e);
219  _Tp __gam1, __gam2, __gampl, __gammi;
220  __gamma_temme(__mu, __gam1, __gam2, __gampl, __gammi);
221  auto __ff = __fact
222  * (__gam1 * std::cosh(__e) + __gam2 * __fact2 * __d);
223  auto __sum = __ff;
224  __e = std::exp(__e);
225  auto __p = __e / (_Tp{2} * __gampl);
226  auto __q = _Tp{1} / (_Tp{2} * __e * __gammi);
227  auto __c = _Tp{1};
228  __d = __x2 * __x2;
229  auto __sum1 = __p;
230  int __i;
231  for (__i = 1; __i <= _S_max_iter; ++__i)
232  {
233  __ff = (__i * __ff + __p + __q) / (__i * __i - __mu2);
234  __c *= __d / _Tp(__i);
235  __p /= _Tp(__i) - __mu;
236  __q /= _Tp(__i) + __mu;
237  const auto __del = __c * __ff;
238  __sum += __del;
239  const auto __del1 = __c * (__p - _Tp(__i) * __ff);
240  __sum1 += __del1;
241  if (std::abs(__del) < _S_eps * std::abs(__sum))
242  break;
243  }
244  if (__i > _S_max_iter)
245  std::__throw_runtime_error(__N("__cyl_bessel_ik_steed: "
246  "K-series failed to converge"));
247  _Kmu = __sum;
248  _Knu1 = __sum1 * __xi2;
249  }
250  else
251  {
252  auto __b = _Tp{2} * (_Tp{1} + __x);
253  auto __d = _Tp{1} / __b;
254  auto __delh = __d;
255  auto __h = __delh;
256  auto __q1 = _Tp{0};
257  auto __q2 = _Tp{1};
258  auto __a1 = _Tp{0.25L} - __mu2;
259  auto __q = __c = __a1;
260  auto __a = -__a1;
261  auto __s = _Tp{1} + __q * __delh;
262  int __i;
263  for (__i = 2; __i <= _S_max_iter; ++__i)
264  {
265  __a -= _Tp{2 * (__i - 1)};
266  __c = -__a * __c / __i;
267  const auto __qnew = (__q1 - __b * __q2) / __a;
268  __q1 = __q2;
269  __q2 = __qnew;
270  __q += __c * __qnew;
271  __b += _Tp{2};
272  __d = _Tp{1} / (__b + __a * __d);
273  __delh = (__b * __d - _Tp{1}) * __delh;
274  __h += __delh;
275  const auto __dels = __q * __delh;
276  __s += __dels;
277  if (std::abs(__dels / __s) < _S_eps)
278  break;
279  }
280  if (__i > _S_max_iter)
281  std::__throw_runtime_error(__N("__cyl_bessel_ik_steed: "
282  "Steed's method failed"));
283  __h = __a1 * __h;
284  _Kmu = std::sqrt(_S_pi / (_Tp{2} * __x))
285  * std::exp(-__x) / __s;
286  _Knu1 = _Kmu * (__mu + __x + _Tp{0.5L} - __h) * __xi;
287  }
288 
289  auto _Kpmu = __mu * __xi * _Kmu - _Knu1;
290  auto _Inumu = __xi / (__f * _Kmu - _Kpmu);
291  _Inu = _Inumu * _Inul1 / _Inul;
292  _Ipnu = _Inumu * _Ipnu1 / _Inul;
293  for (int __i = 1; __i <= __n; ++__i)
294  _Kmu = std::exchange(_Knu1, (__mu + _Tp(__i)) * __xi2 * _Knu1 + _Kmu);
295  _Knu = _Kmu;
296  _Kpnu = __nu * __xi * _Kmu - _Knu1;
297 
298  return;
299  }

301  /**
302  * @brief Return the modified cylindrical Bessel functions
303  * and their derivatives of order @f$ \nu @f$ by various means.
304  *
305  * @param __nu The order of the Bessel functions.
306  * @param __x The argument of the Bessel functions.
307  * @param _Inu The output regular modified Bessel function.
308  * @param _Knu The output irregular modified Bessel function.
309  * @param _Ipnu The output derivative of the regular
310  * modified Bessel function.
311  * @param _Kpnu The output derivative of the irregular
312  * modified Bessel function.
313  */
314  template<typename _Tp>
315  void
316  __cyl_bessel_ik(_Tp __nu, _Tp __x,
317  _Tp & _Inu, _Tp & _Knu, _Tp & _Ipnu, _Tp & _Kpnu)
318  {
319  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
320  constexpr auto _S_inf = __gnu_cxx::__infinity<_Tp>();
321  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
322  if (__nu < _Tp{0})
323  {
324  _Tp _I_mnu, _K_mnu, _Ip_mnu, _Kp_mnu;
325  __cyl_bessel_ik(-__nu, __x, _I_mnu, _K_mnu, _Ip_mnu, _Kp_mnu);
326  auto __arg = -__nu * _S_pi;
327  auto __sinnupi = std::sin(__arg);
328  if (std::abs(__sinnupi) < _S_eps)
329  { // Carefully preserve +-inf.
330  _Inu = _I_mnu;
331  _Knu = _K_mnu;
332  _Ipnu = _Ip_mnu;
333  _Kpnu = _Kp_mnu;
334  }
335  else
336  {
337  _Inu = _I_mnu + _Tp{2} * __sinnupi * _K_mnu / _S_pi;
338  _Knu = _K_mnu;
339  _Ipnu = _Ip_mnu + _Tp{2} * __sinnupi * _Kp_mnu / _S_pi;
340  _Kpnu = _Kp_mnu;
341  }
342  }
343  else if (__x == _Tp{0})
344  {
345  if (__nu == _Tp{0})
346  {
347  _Inu = _Tp{1};
348  _Ipnu = _Tp{0};
349  }
350  else if (__nu == _Tp{1})
351  {
352  _Inu = _Tp{0};
353  _Ipnu = _Tp{0.5L};
354  }
355  else
356  {
357  _Inu = _Tp{0};
358  _Ipnu = _Tp{0};
359  }
360  _Knu = _S_inf;
361  _Kpnu = -_S_inf;
362  return;
363  }
364  else if (__x > _Tp{1000})
365  __cyl_bessel_ik_asymp(__nu, __x, _Inu, _Knu, _Ipnu, _Kpnu);
366  else
367  __cyl_bessel_ik_steed(__nu, __x, _Inu, _Knu, _Ipnu, _Kpnu);
368  }

372  * @f$ \nu @f$: @f$ I_{\nu}(x) @f$.

375  * @f[
376  * I_{\nu}(x) = \sum_{k=0}^{\infty}
377  * \frac{(x/2)^{\nu + 2k}}{k!\Gamma(\nu+k+1)}
378  * @f]
379  *
380  * @param __nu The order of the regular modified Bessel function.
381  * @param __x The argument of the regular modified Bessel function.
382  * @return The output regular modified Bessel function.
383  */

386  __cyl_bessel_i(_Tp __nu, _Tp __x)
387  {
388  if (__x < _Tp{0})
389  std::__throw_domain_error(__N("__cyl_bessel_i: bad argument"));

391  return __gnu_cxx::__quiet_NaN<_Tp>();
392  else if (__nu >= _Tp{0} && __x * __x < _Tp{10} * (__nu + _Tp{1}))
393  return __cyl_bessel_ij_series(__nu, __x, +_Tp{1}, 200);
394  else
395  {
396  _Tp _I_nu, _K_nu, _Ip_nu, _Kp_nu;

399  }

402  /**
403  * @brief Return the irregular modified Bessel function
404  * @f$ K_{\nu}(x) @f$ of order @f$ \nu @f$.
405  *
406  * The irregular modified Bessel function is defined by:
407  * @f[
408  * K_{\nu}(x) = \frac{\pi}{2}
409  * \frac{I_{-\nu}(x) - I_{\nu}(x)}{\sin \nu\pi}
410  * @f]

412  * @f$ lim_{\nu \to n} @f$.
413  * For negative argument we have simply:
414  * @f[
415  * K_{-\nu}(x) = K_{\nu}(x)
416  * @f]
417  *
418  * @param __nu The order of the irregular modified Bessel function.
419  * @param __x The argument of the irregular modified Bessel function.
420  * @return The output irregular modified Bessel function.
421  */
422  template<typename _Tp>
423  _Tp
424  __cyl_bessel_k(_Tp __nu, _Tp __x)
425  {
426  if (__x < _Tp{0})
427  std::__throw_domain_error(__N("__cyl_bessel_k: Bad argument"));
428  else if (__isnan(__nu) || __isnan(__x))
429  return __gnu_cxx::__quiet_NaN<_Tp>();
430  else
431  {
432  _Tp _I_nu, _K_nu, _Ip_nu, _Kp_nu;
433  __cyl_bessel_ik(__nu, __x, _I_nu, _K_nu, _Ip_nu, _Kp_nu);

481  /**
482  * @brief Compute the Airy functions
483  * @f$ Ai(x) @f$ and @f$ Bi(x) @f$ and their first
484  * derivatives @f$ Ai'(x) @f$ and @f$ Bi(x) @f$
485  * respectively.
486  *
487  * @param __z The argument of the Airy functions.
488  * @param _Ai The output Airy function of the first kind.
489  * @param _Bi The output Airy function of the second kind.
490  * @param _Aip The output derivative of the Airy function
491  * of the first kind.
492  * @param _Bip The output derivative of the Airy function
493  * of the second kind.
494  */
495  template<typename _Tp>
496  void
497  __airy(_Tp __z, _Tp & _Ai, _Tp & _Bi, _Tp & _Aip, _Tp & _Bip)
498  {
499  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
500  constexpr auto _S_inf = __gnu_cxx::__infinity<_Tp>();
501  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
502  constexpr auto _S_sqrt3 = __gnu_cxx::__math_constants<_Tp>::__root_3;
503  const auto __absz = std::abs(__z);
504  const auto __rootz = std::sqrt(__absz);
505  const auto __xi = _Tp{2} * __absz * __rootz / _Tp{3};
506 
507  if (__isnan(__z))
508  _Ai = _Bi = _Aip = _Bip = _S_NaN;
509  else if (__z == _S_inf)
510  {
511  _Ai = _Aip = _Tp{0};
512  _Bi = _Bip = _S_inf;
513  }
514  else if (__z == -_S_inf)
515  _Ai = _Bi = _Aip = _Bip = _Tp{0};
516  else if (__z > _Tp{0})
517  {
518  _Tp _I_nu, _K_nu, _Ip_nu, _Kp_nu;
519 
520  __cyl_bessel_ik(_Tp{1} / _Tp{3}, __xi, _I_nu, _K_nu, _Ip_nu, _Kp_nu);
521  _Ai = __rootz * _K_nu / (_S_sqrt3 * _S_pi);
522  _Bi = __rootz * (_K_nu / _S_pi + _Tp{2} * _I_nu / _S_sqrt3);
523 
524  __cyl_bessel_ik(_Tp{2} / _Tp{3}, __xi, _I_nu, _Ip_nu, _K_nu, _Kp_nu);
525  _Aip = -__z * _K_nu / (_S_sqrt3 * _S_pi);
526  _Bip = __z * (_K_nu / _S_pi + _Tp{2} * _I_nu / _S_sqrt3);
527  }
528  else if (__z < _Tp{0})
529  {
530  _Tp _J_nu, _N_nu, _Jp_nu, _Np_nu;
