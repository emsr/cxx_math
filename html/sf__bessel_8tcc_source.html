59  /**
60  * @brief This routine computes the asymptotic cylindrical Bessel
61  * and Neumann functions of order nu: \f$ J_{\nu}(x) \f$,
62  * \f$ N_{\nu}(x) \f$. Use this for \f$ x >> nu^2 + 1 \f$.
63  *
64  * References:
65  * (1) Handbook of Mathematical Functions,
66  * ed. Milton Abramowitz and Irene A. Stegun,
67  * Dover Publications,
68  * Section 9 p. 364, Equations 9.2.5-9.2.10
69  *
70  * @param __nu The order of the Bessel functions.
71  * @param __x The argument of the Bessel functions.
72  * @param _Jnu The output Bessel function of the first kind.
73  * @param _Nnu The output Neumann function (Bessel function of the second kind).
74  */
75  template<typename _Tp>
76  void
77  __cyl_bessel_jn_asymp(_Tp __nu, _Tp __x,
78  _Tp & _Jnu, _Tp & _Nnu,
79  _Tp & _Jpnu, _Tp & _Npnu)
80  {
81  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
82  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
83  constexpr auto _S_pi_2 = __gnu_cxx::__math_constants<long double>::__pi_half;
84  const auto __2nu = _Tp{2} * __nu;
85  const auto __4nu2 = __2nu * __2nu;
86  const auto __8x = _Tp{8} * __x;
87  auto __k = 0;
88  auto __bk_xk = _Tp{1};
89  auto _Rsum = __bk_xk;
90  auto __ak_xk = _Tp{1};
91  auto _Psum = __ak_xk;
92  ++__k;
93  auto __2km1 = 1;
94  __bk_xk *= (__4nu2 + __2km1 * (__2km1 + 2)) / __8x;
95  auto _Ssum = __bk_xk;
96  __ak_xk *= (__2nu - __2km1) * (__2nu + __2km1) / __8x;
97  auto _Qsum = __ak_xk;
98  do
99  {
100  ++__k;
101  __2km1 += 2;
102  __bk_xk = -(__4nu2 + __2km1 * (__2km1 + 2)) * __ak_xk / (__k * __8x);
103  _Rsum += __bk_xk;
104  __ak_xk *= -(__2nu - __2km1) * (__2nu + __2km1) / (__k * __8x);
105  _Psum += __ak_xk;
106  auto __convP = std::abs(__ak_xk) < _S_eps * std::abs(_Psum);
107 
108  ++__k;
109  __2km1 += 2;
110  __bk_xk = (__4nu2 + __2km1 * (__2km1 + 2)) * __ak_xk / (__k * __8x);
111  _Ssum += __bk_xk;
112  __ak_xk *= (__2nu - __2km1) * (__2nu + __2km1) / (__k * __8x);
113  _Qsum += __ak_xk;
114  auto __convQ = std::abs(__ak_xk) < _S_eps * std::abs(_Qsum);
115 
116  if (__convP && __convQ && __k > (__nu / _Tp{2}))
117  break;
118  }
119  while (__k < _Tp{100} * __nu);
120 
121  auto __omega = __x - (__nu + 0.5L) * _S_pi_2;
122  auto __c = std::cos(__omega);
123  auto __s = std::sin(__omega);
124 
125  auto __coef = std::sqrt(_Tp{2} / (_S_pi * __x));
126  _Jnu = __coef * (__c * _Psum - __s * _Qsum);
127  _Nnu = __coef * (__s * _Psum + __c * _Qsum);
128  _Jpnu = -__coef * (__s * _Rsum + __c * _Ssum);
129  _Npnu = __coef * (__c * _Rsum - __s * _Ssum);
130 
131  return;
132  }

134  /**
135  * @brief Compute the gamma functions required by the Temme series
136  * expansions of @f$ N_\nu(x) @f$ and @f$ K_\nu(x) @f$.
137  * @f[
138  * \Gamma_1 = \frac{1}{2\mu}
139  * [\frac{1}{\Gamma(1 - \mu)} - \frac{1}{\Gamma(1 + \mu)}]
140  * @f]
141  * and
142  * @f[
143  * \Gamma_2 = \frac{1}{2}
144  * [\frac{1}{\Gamma(1 - \mu)} + \frac{1}{\Gamma(1 + \mu)}]
145  * @f]
146  * where @f$ -1/2 <= \mu <= 1/2 @f$ is @f$ \mu = \nu - N @f$ and @f$ N @f$.
147  * is the nearest integer to @f$ \nu @f$.
148  * The values of \f$ \Gamma(1 + \mu) \f$ and \f$ \Gamma(1 - \mu) \f$
149  * are returned as well.
150  *
151  * The accuracy requirements on this are exquisite.
152  *
153  * @param __mu The input parameter of the gamma functions.
154  * @param __gam1 The output function \f$ \Gamma_1(\mu) \f$
155  * @param __gam2 The output function \f$ \Gamma_2(\mu) \f$
156  * @param __gampl The output function \f$ \Gamma(1 + \mu) \f$
157  * @param __gammi The output function \f$ \Gamma(1 - \mu) \f$
158  */
159  template<typename _Tp>
160  void
161  __gamma_temme(_Tp __mu,
162  _Tp & __gam1, _Tp & __gam2, _Tp & __gampl, _Tp & __gammi)
163  {
164  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
165  constexpr auto _S_gamma_E = __gnu_cxx::__math_constants<_Tp>::__gamma_e;
166  __gampl = _Tp{1} / std::tgamma(_Tp{1} + __mu);
167  __gammi = _Tp{1} / std::tgamma(_Tp{1} - __mu);
168 
169  if (std::abs(__mu) < _S_eps)
170  __gam1 = -_S_gamma_E;
171  else
172  __gam1 = (__gammi - __gampl) / (_Tp{2} * __mu);
173 
174  __gam2 = (__gammi + __gampl) / _Tp{2};
175 
176  return;
177  }

179  /**
180  * @brief Compute the Bessel @f$ J_\nu(x) @f$ and Neumann
181  * @f$ N_\nu(x) @f$ functions and their first derivatives
182  * @f$ J'_\nu(x) @f$ and @f$ N'_\nu(x) @f$ respectively.
183  * These four functions are computed together for numerical
184  * stability.
185  *
186  * @param __nu The order of the Bessel functions.
187  * @param __x The argument of the Bessel functions.
188  * @param _Jnu The output Bessel function of the first kind.
189  * @param _Nnu The output Neumann function (Bessel function of the second kind).
190  * @param _Jpnu The output derivative of the Bessel function of the first kind.
191  * @param _Npnu The output derivative of the Neumann function.
192  */
193  template<typename _Tp>
194  void
195  __cyl_bessel_jn_steed(_Tp __nu, _Tp __x,
196  _Tp & _Jnu, _Tp & _Nnu, _Tp & _Jpnu, _Tp & _Npnu)
197  {
198  constexpr auto _S_inf = __gnu_cxx::__infinity<_Tp>();
199  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
200  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
201  // When the multiplier is N i.e.
202  // fp_min = N * min()
203  // Then J_0 and N_0 tank at x = 8 * N (J_0 = 0 and N_0 = nan)!
204  //const _Tp _S_fp_min = _Tp{20} * __gnu_cxx::__min<_Tp>();
205  constexpr auto _S_fp_min = __gnu_cxx::__sqrt_min<_Tp>();
206  constexpr int _S_max_iter = 15000;
207  constexpr auto _S_x_min = _Tp{2};
208 
209  const int __n = (__x < _S_x_min
210  ? std::nearbyint(__nu)
211  : std::max(0,
212  static_cast<int>(__nu - __x + _Tp{1.5L})));
213 
214  const auto __mu = __nu - __n;
215  const auto __mu2 = __mu * __mu;
216  const auto __xi = _Tp{1} / __x;
217  const auto __xi2 = _Tp{2} * __xi;
218  const auto _Wronski = __xi2 / _S_pi;
219  int __isign = 1;
220  _Tp __h = __nu * __xi;
221  if (__h < _S_fp_min)
222  __h = _S_fp_min;
223  auto __b = __xi2 * __nu;
224  auto __d = _Tp{0};
225  auto __c = __h;
226  int __i;
227  for (__i = 1; __i <= _S_max_iter; ++__i)
228  {
229  __b += __xi2;
230  __d = __b - __d;
231  if (std::abs(__d) < _S_fp_min)
232  __d = _S_fp_min;
233  __c = __b - _Tp{1} / __c;
234  if (std::abs(__c) < _S_fp_min)
235  __c = _S_fp_min;
236  __d = _Tp{1} / __d;
237  const auto __del = __c * __d;
238  __h *= __del;
239  if (__d < _Tp{0})
240  __isign = -__isign;
241  if (std::abs(__del - _Tp{1}) < _S_eps)
242  break;
243  }
244  if (__i > _S_max_iter)
245  {
246  // Don't throw with message "try asymptotic expansion" - Just do it!
247  __cyl_bessel_jn_asymp(__nu, __x, _Jnu, _Nnu, _Jpnu, _Npnu);
248  return;
249  }
250  auto _Jnul = __isign * _S_fp_min;
251  auto _Jpnul = __h * _Jnul;
252  auto _Jnul1 = _Jnul;
253  auto _Jpnu1 = _Jpnul;
254  auto __fact = __nu * __xi;
255  for (int __l = __n; __l >= 1; --__l)
256  {
257  const auto _Jnutemp = __fact * _Jnul + _Jpnul;
258  __fact -= __xi;
259  _Jpnul = __fact * _Jnutemp - _Jnul;
260  _Jnul = _Jnutemp;
261  }
262  if (_Jnul == _Tp{0})
263  _Jnul = _S_eps;
264 
265  auto __f = _Jpnul / _Jnul;
266  _Tp _Nmu, _Nnu1, _Npmu, _Jmu;
267  if (__x < _S_x_min)
268  {
269  const auto __x2 = __x / _Tp{2};
270  const auto __pimu = _S_pi * __mu;
271  const auto __fact = (std::abs(__pimu) < _S_eps
272  ? _Tp{1}
273  : __pimu / std::sin(__pimu));
274  auto __d = -std::log(__x2);
275  auto __e = __mu * __d;
276  const auto __fact2 = (std::abs(__e) < _S_eps
277  ? _Tp{1}
278  : std::sinh(__e) / __e);
279  _Tp __gam1, __gam2, __gampl, __gammi;
280  __gamma_temme(__mu, __gam1, __gam2, __gampl, __gammi);
281  auto __ff = (_Tp{2} / _S_pi) * __fact
282  * (__gam1 * std::cosh(__e) + __gam2 * __fact2 * __d);
283  __e = std::exp(__e);
284  auto __p = __e / (_S_pi * __gampl);
285  auto __q = _Tp{1} / (__e * _S_pi * __gammi);
286  const auto __pimu2 = __pimu / _Tp{2};
287  auto __fact3 = (std::abs(__pimu2) < _S_eps
288  ? _Tp{1} : std::sin(__pimu2) / __pimu2 );
289  auto __r = _S_pi * __pimu2 * __fact3 * __fact3;
290  auto __c = _Tp{1};
291  __d = -__x2 * __x2;
292  auto __sum = __ff + __r * __q;
293  auto __sum1 = __p;
294  int __i;
295  for (__i = 1; __i <= _S_max_iter; ++__i)
296  {
297  __ff = (__i * __ff + __p + __q) / (__i * __i - __mu2);
298  __c *= __d / _Tp(__i);
299  __p /= _Tp(__i) - __mu;
300  __q /= _Tp(__i) + __mu;
301  const auto __del = __c * (__ff + __r * __q);
302  __sum += __del;
303  const auto __del1 = __c * __p - _Tp(__i) * __del;
304  __sum1 += __del1;
305  if (std::abs(__del) < _S_eps * (_Tp{1} + std::abs(__sum)))
306  break;
307  }
308  if (__i > _S_max_iter)
309  std::__throw_runtime_error(__N("__cyl_bessel_jn_steed: "
310  "Y-series failed to converge"));
311  _Nmu = -__sum;
312  _Nnu1 = -__sum1 * __xi2;
313  _Npmu = __mu * __xi * _Nmu - _Nnu1;
314  _Jmu = _Wronski / (_Npmu - __f * _Nmu);
315  }
316  else
317  {
318  auto __a = _Tp{0.25L} - __mu2;
319  auto __q = _Tp{1};
320  auto __p = -__xi / _Tp{2};
321  auto __br = _Tp{2} * __x;
322  auto __bi = _Tp{2};
323  auto __fact = __a * __xi / (__p * __p + __q * __q);
324  auto __cr = __br + __q * __fact;
325  auto __ci = __bi + __p * __fact;
326  auto __den = __br * __br + __bi * __bi;
327  auto __dr = __br / __den;
328  auto __di = -__bi / __den;
329  auto __dlr = __cr * __dr - __ci * __di;
330  auto __dli = __cr * __di + __ci * __dr;
331  auto __temp = __p * __dlr - __q * __dli;
332  __q = __p * __dli + __q * __dlr;
333  __p = __temp;
334  int __i;
335  for (__i = 2; __i <= _S_max_iter; ++__i)
336  {
337  __a += _Tp{2 * (__i - 1)};
338  __bi += _Tp{2};
339  __dr = __a * __dr + __br;
340  __di = __a * __di + __bi;
341  if (std::abs(__dr) + std::abs(__di) < _S_fp_min)
342  __dr = _S_fp_min;
343  __fact = __a / (__cr * __cr + __ci * __ci);
344  __cr = __br + __cr * __fact;
345  __ci = __bi - __ci * __fact;
346  if (std::abs(__cr) + std::abs(__ci) < _S_fp_min)
347  __cr = _S_fp_min;
348  __den = __dr * __dr + __di * __di;
349  __dr /= __den;
350  __di /= -__den;
351  __dlr = __cr * __dr - __ci * __di;
352  __dli = __cr * __di + __ci * __dr;
353  __temp = __p * __dlr - __q * __dli;
354  __q = __p * __dli + __q * __dlr;
355  __p = __temp;
356  if (std::abs(__dlr - _Tp{1}) + std::abs(__dli) < _S_eps)
357  break;
358  }
359  if (__i > _S_max_iter)
360  std::__throw_runtime_error(__N("__cyl_bessel_jn_steed: "
361  "Lentz's method failed"));
362  const auto __gam = (__p - __f) / __q;
363  _Jmu = std::sqrt(_Wronski / ((__p - __f) * __gam + __q));
364  _Jmu = std::copysign(_Jmu, _Jnul);
365  _Nmu = __gam * _Jmu;
366  _Npmu = (__p + __q / __gam) * _Nmu;
367  _Nnu1 = __mu * __xi * _Nmu - _Npmu;
368  }
369  __fact = _Jmu / _Jnul;
370  _Jnu = __fact * _Jnul1;
371  _Jpnu = __fact * _Jpnu1;
372  for (int __i = 1; __i <= __n; ++__i)
373  _Nmu = std::exchange(_Nnu1, (__mu + __i) * __xi2 * _Nnu1 - _Nmu);
374  _Nnu = _Nmu;
375  _Npnu = __nu * __xi * _Nmu - _Nnu1;
376 
377  return;
378  }

381  /**
382  * @brief This routine returns the cylindrical Bessel functions
383  * of order \f$ \nu \f$: \f$ J_{\nu} \f$ or \f$ I_{\nu} \f$
384  * by series expansion.
385  *
386  * The modified cylindrical Bessel function is:
387  * @f[
388  * Z_{\nu}(x) = \sum_{k=0}^{\infty}
389  * \frac{\sigma^k (x/2)^{\nu + 2k}}{k!\Gamma(\nu+k+1)}
390  * @f]
391  * where \f$ \sigma = +1 \f$ or\f$ -1 \f$ for
392  * \f$ Z = I \f$ or \f$ J \f$ respectively.
393  *
394  * See Abramowitz & Stegun, 9.1.10
395  * Abramowitz & Stegun, 9.6.7
396  * (1) Handbook of Mathematical Functions,
397  * ed. Milton Abramowitz and Irene A. Stegun,
398  * Dover Publications,
399  * Equation 9.1.10 p. 360 and Equation 9.6.10 p. 375
400  *
401  * @param __nu The order of the Bessel function.
402  * @param __x The argument of the Bessel function.
403  * @param __sgn The sign of the alternate terms
404  * -1 for the Bessel function of the first kind.
405  * +1 for the modified Bessel function of the first kind.
406  * @return The output Bessel function.
407  */
408  template<typename _Tp>
409  _Tp
410  __cyl_bessel_ij_series(_Tp __nu, _Tp __x, _Tp __sgn,
411  unsigned int __max_iter)
412  {
413  constexpr auto _S_eps = __gnu_cxx::__epsilon<_Tp>();
414  const auto __x2 = __x / _Tp{2};
415  _Tp __fact = __nu * std::log(__x2);
416  __fact -= __log_gamma(__nu + _Tp{1});
417  __fact = std::exp(__fact);
418  const auto __xx4 = __sgn * __x2 * __x2;
419  _Tp _Jn = _Tp{1};
420  _Tp __term = _Tp{1};
421 
422  for (unsigned int __i = 1; __i < __max_iter; ++__i)
423  {
424  __term *= __xx4 / (_Tp(__i) * (__nu + _Tp(__i)));
425  _Jn += __term;
426  if (std::abs(__term / _Jn) < _S_eps)
427  break;
428  }
429 
430  return __fact * _Jn;
431  }

433  /**
434  * @brief Return the cylindrical Bessel functions and their derivatives
435  * of order \f$ \nu \f$ by various means.
436  */
437  template<typename _Tp>
438  void
439  __cyl_bessel_jn(_Tp __nu, _Tp __x,
440  _Tp & _Jnu, _Tp & _Nnu, _Tp & _Jpnu, _Tp & _Npnu)
441  {
442  constexpr auto _S_inf = __gnu_cxx::__infinity<_Tp>();
443  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
444  if (__nu < _Tp{0})
445  {
446  _Tp _J_mnu, _N_mnu, _Jp_mnu, _Np_mnu;
447  __cyl_bessel_jn(-__nu, __x, _J_mnu, _N_mnu, _Jp_mnu, _Np_mnu);
448  auto __arg = -__nu * _S_pi;
449  auto __sinnupi = std::sin(__arg);
450  auto __cosnupi = std::cos(__arg);
451  _Jnu = __cosnupi * _J_mnu - __sinnupi * _N_mnu;
452  _Nnu = __sinnupi * _J_mnu + __cosnupi * _N_mnu;
453  _Jpnu = __cosnupi * _Jp_mnu - __sinnupi * _Np_mnu;
454  _Npnu = __sinnupi * _Jp_mnu + __cosnupi * _Np_mnu;
455  }
456  else if (__x == _Tp{0})
457  {
458  if (__nu == _Tp{0})
459  {
460  _Jnu = _Tp{1};
461  _Jpnu = _Tp{0};
462  }
463  else if (__nu == _Tp{1})
464  {
465  _Jnu = _Tp{0};
466  _Jpnu = _Tp{0.5L};
467  }
468  else
469  {
470  _Jnu = _Tp{0};
471  _Jpnu = _Tp{0};
472  }
473  _Nnu = -_S_inf;
474  _Npnu = _S_inf;
475  return;
476  }
477  else if (__x > _Tp{1000})
478  __cyl_bessel_jn_asymp(__nu, __x, _Jnu, _Nnu, _Jpnu, _Npnu);
479  else
480  __cyl_bessel_jn_steed(__nu, __x, _Jnu, _Nnu, _Jpnu, _Npnu);
481  }

517  /**
518  * @brief Return the Neumann function of order \f$ \nu \f$:
519  * \f$ N_{\nu}(x) \f$.
520  *
521  * The Neumann function is defined by:
