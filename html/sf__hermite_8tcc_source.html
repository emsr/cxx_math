53  /**
54  * @brief This routine returns the Hermite polynomial
55  * of order n: \f$ H_n(x) \f$ by recursion on n.
56  *
57  * The Hermite polynomial is defined by:
58  * @f[
59  * H_n(x) = (-1)^n e^{x^2} \frac{d^n}{dx^n} e^{-x^2}
60  * @f]
61  *
62  * @param __n The order of the Hermite polynomial.
63  * @param __x The argument of the Hermite polynomial.
64  * @return The value of the Hermite polynomial of order n
65  * and argument x.
66  */
67  template<typename _Tp>
68  _Tp
69  __poly_hermite_recursion(unsigned int __n, _Tp __x)
70  {
71  // Compute H_0.
72  auto __H_nm2 = _Tp{1};
73  if (__n == 0)
74  return __H_nm2;
75 
76  // Compute H_1.
77  auto __H_nm1 = _Tp{2} * __x;
78  if (__n == 1)
79  return __H_nm1;
80 
81  // Compute H_n.
82  _Tp __H_n;
83  for (unsigned int __i = 2; __i <= __n; ++__i)
84  {
85  __H_n = _Tp{2} * (__x * __H_nm1 - _Tp(__i - 1) * __H_nm2);
86  __H_nm2 = __H_nm1;
87  __H_nm1 = __H_n;
88  }
89 
90  return __H_n;
91  }

93  /**
94  * @brief This routine returns the Hermite polynomial
95  * of large order n: \f$ H_n(x) \f$. We assume here
96  * that x >= 0.
97  *
98  * The Hermite polynomial is defined by:
99  * @f[
100  * H_n(x) = (-1)^n e^{x^2} \frac{d^n}{dx^n} e^{-x^2}
101  * @f]
102  *
103  * @see "Asymptotic analysis of the Hermite polynomials
104  * from their differential-difference equation",
105  * Diego Dominici, arXiv:math/0601078v1 [math.CA] 4 Jan 2006
106  * @param __n The order of the Hermite polynomial.
107  * @param __x The argument of the Hermite polynomial.
108  * @return The value of the Hermite polynomial of order n
109  * and argument x.
110  */
111  template<typename _Tp>
112  _Tp
113  __poly_hermite_asymp(unsigned int __n, _Tp __x)
114  {
115  constexpr auto _S_pi = __gnu_cxx::__math_constants<_Tp>::__pi;
116  constexpr auto _S_sqrt_2 = __gnu_cxx::__math_constants<_Tp>::__root_2;
117  constexpr auto _S_sqrt_2pi = _S_sqrt_2
118  * __gnu_cxx::__math_constants<_Tp>::__root_pi;
119  // __x >= 0 in this routine.
120  const auto __xturn = std::sqrt(_Tp(2 * __n));
121  if (std::abs(__x - __xturn) < _Tp{0.05L} * __xturn)
122  {
123  // Transition region x ~ sqrt(2n).
124  const auto __n_2 = _Tp(__n) / _Tp{2};
125  const auto __n6th = std::pow(_Tp(__n), _Tp{1} / _Tp{6});
126  const auto __exparg = __n * std::log(__xturn) - _Tp{3} * __n_2
127  + __xturn * __x;
128  const auto __airyarg = _S_sqrt_2 * (__x - __xturn) * __n6th;
129  _Tp _Ai, _Bi, _Aip, _Bip;
130  __airy(__airyarg, _Ai, _Bi, _Aip, _Bip);
131  return _S_sqrt_2pi * __n6th * std::exp(__exparg) * _Ai;
132  }
133  else if (__x < __xturn)
134  {
135  // Oscillatory region |x| < sqrt(2n).
136  const auto __theta = std::asin(__x / __xturn);
137  const auto __2theta = _Tp{2} * __theta;
138  const auto __n_2 = _Tp(__n) / _Tp{2};
139  const auto __exparg = __n_2 * (_Tp{2} * std::log(__xturn)
140  - std::cos(__2theta));
141  const auto __arg = __theta / _Tp{2}
142  + __n_2 * (std::sin(__2theta) + __2theta - _S_pi);
143  return std::sqrt(_Tp{2} / std::cos(__theta))
144  * std::exp(__exparg) * std::cos(__arg);
145  }
146  else
147  {
148  // Exponential region |x| > sqrt(2n).
149  const auto __sigma = std::sqrt((__x - __xturn) * (__x + __xturn));
150  const auto __exparg = _Tp{0.5L} * (__x * (__x - __sigma) - __n)
151  + __n * std::log(__sigma + __x);
152  return std::exp(__exparg)
153  * std::sqrt(_Tp{0.5L} * (_Tp{1} + __x / __sigma));
154  }
155  }

158  /**
159  * @brief This routine returns the Hermite polynomial
160  * of order n: \f$ H_n(x) \f$.
161  *
162  * The Hermite polynomial is defined by:
163  * @f[
164  * H_n(x) = (-1)^n e^{x^2} \frac{d^n}{dx^n} e^{-x^2}
165  * @f]
166  *
167  * The Hermite polynomial obeys a reflection formula:
168  * @f[
169  * H_n(-x) = (-1)^n H_n(x)
170  * @f]
171  *
172  * @param __n The order of the Hermite polynomial.
173  * @param __x The argument of the Hermite polynomial.
174  * @return The value of the Hermite polynomial of order n
175  * and argument x.
176  */
177  template<typename _Tp>
178  _Tp
179  __poly_hermite(unsigned int __n, _Tp __x)
180  {
181  if (__isnan(__x))
182  return __gnu_cxx::__quiet_NaN<_Tp>();
183  else if (__x < _Tp{0})
184  return (__n % 2 == 1 ? -1 : +1) * __poly_hermite(__n, -__x);
185  else if (__n > 100)
186  return __poly_hermite_asymp(__n, __x);
187  else
188  return __poly_hermite_recursion(__n, __x);
189  }
