46  /**
47  * Compute and return the exponential @f$ \theta_2 @f$ function
48  * by series expansion.
49  */
50  template<typename _Tp>
51  _Tp
52  __theta_2_sum(_Tp __nu, _Tp __x)
53  {
54  using _Real = __num_traits_t<_Tp>;
55  const auto _S_eps = __gnu_cxx::__epsilon(std::abs(__x));
56  const auto _S_NaN = __gnu_cxx::__quiet_NaN(std::abs(__x));
57  const auto _S_pi = __gnu_cxx::__const_pi(std::abs(__x));
58  auto __sum = std::exp(-__nu * __nu / __x);
59  auto __sign = _Tp{-1};
60  for (auto __k = 1; __k < 20; ++__k)
61  {
62  auto __nup = __nu + _Tp(__k);
63  auto __termp = __sign * std::exp(-__nup * __nup / __x);
64  auto __num = __nu - _Tp(__k);
65  auto __termm = __sign * std::exp(-__num * __num / __x);
66  __sum += __termp + __termm;
67  __sign = -__sign;
68  if (std::abs(__termp) < _S_eps * std::abs(__sum)
69  && std::abs(__termm) < _S_eps * std::abs(__sum))
70  break;
71  }
72  return __sum / std::sqrt(_S_pi * __x);
73  }

104  * by asymptotic series expansion.
105  */

116  {

118  auto __cosarg = __nu * __thing;

335  * given the elliptic argument k.
336  * @f[
337  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
338  * @f]
339  * where @f$ k' = \sqrt{1 - k^2} @f$ is the complementary elliptic argument
340  * and @f$ @f$ is the Legendre elliptic integral of the first kind.
341  */
342  template<typename _Tp>
343  _Tp
344  __ellnome_series(_Tp __k)
345  {
346  auto __m = __k * __k;
347  return __m * ((_Tp{1} / _Tp{16})
348  + __m * ((_Tp{1} / _Tp{32})
349  + __m * ((_Tp{21} / _Tp{1024})
350  + __m * ((_Tp{31} / _Tp{2048})
351  + __m * (_Tp{6257} / _Tp{524288})))));

356  * given the elliptic argument k.
357  * @f[
358  * q(k) = exp\left(-\pi\frac{K(k')}{K(k)}\right)
359  * @f]
