42  /**
43  * Compute the Jacobi polynomial by recursion on x:
44  * @f[
45  * 2 k(\alpha + \beta + k) (\alpha + \beta + 2k - 2)
46  * P^{(\alpha, \beta)}_{k}(x)
47  * = (\alpha + \beta + 2k - 1)
48  * ((\alpha^2 - \beta^2)
49  * + x(\alpha + \beta + 2k - 2)(\alpha + \beta + 2k))
50  * P^{(\alpha, \beta)}_{k-1}(x)
51  * - 2 (\alpha + k - 1)(\beta + k - 1)(\alpha + \beta + 2k)
52  * P^{(\alpha, \beta)}_{k-2}(x)
53  * @f]
54  */
55  template<typename _Tp>
56  _Tp
57  __poly_jacobi(unsigned int __n, _Tp __alpha, _Tp __beta, _Tp __x)
58  {
59  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
60 
61  if (__isnan(__alpha) || __isnan(__beta) || __isnan(__x))
62  return _S_NaN;
63 
64  auto _Pm2 = _Tp{1};
65  if (__n == 0)
66  return _Pm2;
67 
68  auto __apb = __alpha + __beta;
69  auto __amb = __alpha - __beta;
70  auto _Pm1 = (__amb + (_Tp{2} + __apb) * __x) / _Tp{2};
71  if (__n == 1)
72  return _Pm1;
73 
74  auto _Pm0 = _Tp{0};
75  auto __a2mb2 = __amb * __apb;
76  for (auto __k = 2; __k <= __n; ++__k )
77  {
78  auto __apbpk = __apb + _Tp(__k);
79  auto __apbp2k = __apbpk + _Tp(__k);
80  auto __apbp2km1 = __apbp2k - _Tp{1};
81  auto __apbp2km2 = __apbp2km1 - _Tp{1};
82  auto __d = _Tp{2} * __k * __apbpk * __apbp2km2;
83  auto __a = __apbp2km2 * __apbp2km1 * __apbp2k;
84  auto __b = __apbp2km1 * __a2mb2;
85  auto __c = _Tp{2} * (__alpha + _Tp(__k - 1))
86  * (__beta + _Tp(__k - 1)) * __apbp2k;
87  if (__d == _Tp{0})
88  std::__throw_runtime_error("__poly_jacobi: error in recursion");
89  _Pm0 = ((__b + __a * __x) * _Pm1 - __c * _Pm2) / __d;
90  _Pm2 = _Pm1;
91  _Pm1 = _Pm0;
92  }
93  return _Pm0;
94  }

109  template<typename _Tp>
110  _Tp
111  __poly_radial_jacobi(unsigned int __n, unsigned int __m, _Tp __rho)
112  {
113  constexpr auto _S_NaN = __gnu_cxx::__quiet_NaN<_Tp>();
114 
115  if (__isnan(__rho))
116  return _S_NaN;
117 
118  if (__m > __n)
119  throw std::range_error("poly_radial_jacobi: m > n");
120  else if ((__n - __m) % 2 == 1)
121  return _Tp{0};
122  else
123  {
124  auto __k = (__n - __m) / 2;
125  return (__k % 2 == 0 ? +1 : -1) * std::pow(__rho, __m)
126  * __poly_jacobi(__k, _Tp(__m), _Tp{0},
127  _Tp{1} - _Tp{2} * __rho * __rho);
128  }
129  }
