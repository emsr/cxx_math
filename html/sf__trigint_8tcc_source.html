45  /**
46  * @brief This function computes the sine @f$ Si(x) @f$
47  * and cosine @f$ Ci(x) @f$ integrals by continued fraction
48  * for positive argument.
49  */
50  template<typename _Tp>
51  void
52  __sincosint_cont_frac(_Tp __t, _Tp& _Si, _Tp& _Ci)
53  {
54  const auto _S_max_iter = 100;
55  const auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon(__t);
56  const auto _S_fp_min = __gnu_cxx::__min(__t);
57  const auto _S_pi_2 = __gnu_cxx::__const_pi_half(__t);
58 
59  // Evaluate Ci and Si by Lentz's modified method of continued fractions.
60  std::complex<_Tp> __b(_Tp{1}, __t);
61  std::complex<_Tp> __c(_Tp{1} / _S_fp_min);
62  std::complex<_Tp> __d(_Tp{1} / __b);
63  std::complex<_Tp> __h(__d);
64  int __i = 2;
65  while (true)
66  {
67  auto __a = -_Tp(__i - 1) * _Tp(__i - 1);
68  __b += _Tp{2};
69  __d = _Tp{1} / (__a * __d + __b);
70  __c = __b + __a / __c;
71  std::complex<_Tp> __del = __c * __d;
72  __h *= __del;
73  if (std::abs(__del - _Tp{1}) < _S_eps)
74  break;
75  if (__i > _S_max_iter)
76  std::__throw_runtime_error(__N("__sincosint_cont_frac: "
77  "continued fraction evaluation failed"));
78  ++__i;
79  }
80  __h *= std::polar(_Tp{1}, -__t);
81  _Ci = -__h.real();
82  _Si = _S_pi_2 + __h.imag();
83 
84  return;
85  }

88  /**
89  * @brief This function computes the sine @f$ Si(x) @f$
90  * and cosine @f$ Ci(x) @f$ integrals by series summation
91  * for positive argument.
92  */
93  template<typename _Tp>
94  void
95  __sincosint_series(_Tp __t, _Tp& _Si, _Tp& _Ci)
96  {
97  const auto _S_max_iter = 100;
98  const auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon(__t);
99  const auto _S_fp_min = __gnu_cxx::__min(__t);
100  const auto _S_gamma_e = __gnu_cxx::__const_gamma_e(__t);
101 
102  // Evaluate Ci and Si by series simultaneously.
103  _Tp __sumc(0), __sums(0);
104  if (__t * __t < _S_fp_min)
105  {
106  // Avoid underflow.
107  __sumc = _Tp{0};
108  __sums = __t;
109  }
110  else
111  {
112  // Evaluate Si and Ci by series expansion.
113  _Tp __sum(0);
114  _Tp __sign(1), __fact(1);
115  bool __odd = true;
116  unsigned int __k = 1;
117  while (true)
118  {
119  __fact *= __t / __k;
120  _Tp __term = __fact / __k;
121  __sum += __sign * __term;
122  _Tp __err = __term / std::abs(__sum);
123  if (__odd)
124  {
125  __sign = -__sign;
126  __sums = __sum;
127  __sum = __sumc;
128  }
129  else
130  {
131  __sumc = __sum;
132  __sum = __sums;
133  }
134  if (__err < _S_eps)
135  break;
136  __odd = !__odd;
137  ++__k;
138  if (__k > _S_max_iter)
139  std::__throw_runtime_error(__N("__sincosint_series: "
140  "series evaluation failed"));
141  }
142  }
143  _Si = __sums;
144  _Ci = _S_gamma_e + std::log(__t) + __sumc;
145 
146  return;
147  }

150  /**
151  * @brief This function computes the sine @f$ Si(x) @f$
152  * and cosine @f$ Ci(x) @f$ integrals by asymptotic series summation
153  * for positive argument.
154  *
155  * The asymptotic series is very good for x > 50.
156  */
157  template<typename _Tp>
158  void
159  __sincosint_asymp(_Tp __t, _Tp& _Si, _Tp& _Ci)
160  {
161  const auto _S_max_iter = 100;
162  const auto _S_eps = _Tp{5} * __gnu_cxx::__epsilon(__t);
163  const auto _S_pi_2 = __gnu_cxx::__const_pi_half(__t);
164 
165  auto __invt = _Tp{1} / __t;
166  auto __term = _Tp{1}; // 0!
167  auto __sume = _Tp{__term};
168  __term *= __invt; // 1! / t
169  auto __sumo = _Tp{__term};
170  auto __sign = _Tp{1};
171  auto __even = true;
172  auto __k = 2;
173  while (true)
174  {
175  __term *= __k * __invt;
176 
177  if (__even)
178  {
179  __sign = -__sign;
180  __sume += __sign * __term;
181 
182  }
183  else
184  {
185  __sumo += __sign * __term;
186  if (__term / std::abs(__sumo) < _S_eps)
187  break;
188  }
189 
190  __even = !__even;
191 
192  if (__k > _S_max_iter)
193  std::__throw_runtime_error(__N("__sincosint_asymp: "
194  "Series evaluation failed"));
195  ++__k;
196  }
197 
198  auto __sint = std::sin(__t);
199  auto __cost = std::cos(__t);
200  _Si = _S_pi_2
201  - __cost * __invt * __sume
202  - __sint * __invt * __sumo;
203  _Ci = __sint * __invt * __sume
204  - __cost * __invt * __sumo;
205 
206  return;
207  }

210  /**
211  * @brief This function returns the sine @f$ Si(x) @f$
212  * and cosine @f$ Ci(x) @f$ integrals as a @c pair.
213  *
214  * The sine integral is defined by:
215  * @f[
216  * Si(x) = \int_0^x dt \frac{\sin(t)}{t}
217  * @f]
218  *
219  * The cosine integral is defined by:
220  * @f[
221  * Ci(x) = \gamma_E + \log(x) + \int_0^x dt \frac{\cos(t) - 1}{t}
222  * @f]
223  */
224  template<typename _Tp>
225  std::pair<_Tp, _Tp>
226  __sincosint(_Tp __x)
227  {
228  const auto _S_NaN = __gnu_cxx::__quiet_NaN(__x);
229  if (__isnan(__x))
230  return std::make_pair(_S_NaN, _S_NaN);
231 
232  auto __t = std::abs(__x);
233  _Tp _Ci, _Si;
234  if (__t == _Tp{0})
235  {
236  _Si = _Tp{0};
237  _Ci = -__gnu_cxx::__infinity(__x);
238  }
239  else if (__t > _Tp{1000}) // Check this!
240  __sincosint_asymp(__t, _Si, _Ci);
241  else if (__t > _Tp{2})
242  __sincosint_cont_frac(__t, _Si, _Ci);
243  else
244  __sincosint_series(__t, _Si, _Ci);
245 
246  if (__x < _Tp{0})
247  _Si = -_Si;
248 
249  return std::make_pair(_Si, _Ci);
250  }
